54void Statemachine_task(void* pvParameters)
55{
56 McuSSD1306_Init();
57 McuGDisplaySSD1306_Init();

59 uint8_t pinvalue;
60 currentpkg.transID = 0;
61 currentpkg.cmd = 0;
62 bool initialized = false;
63
64 while (!initialized)
65 {
66 /* If Start Button pressed */
67 if (PCA_Read_Pin(1, 2, &pinvalue) == kStatus_Success)
68 {
69 if (pinvalue == 1)
70 {
71 uart2WriteLine("START");
72
73 char response[3] = {0};
74 size_t received = 0;
75 status_t status;
76
77 LOG_INFO("Waiting for Communications with RPI...");
78
79 Show1Liner("Connecting to RPI...");
80
81 status = UART_RTOS_Receive(&UART2_rtos_handle, (uint8_t*)response, sizeof(response),
82 &received, 5000);
83 response[sizeof(response) - 1] = '\0';
84 if (received != sizeof(response))
85 {
86 continue;
87 }
88 else
89 {
90 if (strcmp(response, "OK") == 0)
91 {
92 state = IDLE;
93 initialized = true;
94 break;
95 }
96 else
97 {
98 /* wrong answer, wait for a new response */
99 continue;
100 }
101 }
102 }
103 }
104 else
105 {
106 state = ERROR;
107 }
108 vTaskDelay(pdMS_TO_TICKS(100));
109 }
110
111 LOG_INFO("Communications Initialized");
112 Show1Liner("Idle");
113
114 /* statemachine main loop */
115 while (1)
116 {
117 switch (state)
118 {
119 case IDLE:
120 {
121 char receivedLine[32];
122 uint16_t index = 0;
123 char receivedChar = '\0';
124 size_t received = 0;
125 status_t status;
126
127 while (1)
128 {
129 status = UART_RTOS_Receive(&UART2_rtos_handle, (uint8_t*)&receivedChar, 1,
130 &received, 1000);
131 if (received != 1)
132 {
133 continue;
134 }
135
136 if (receivedChar == '\r')
137 {
138 continue;
139 }
140
141 if (receivedChar == '\n')
142 {
143 receivedLine[index] = '\0';
144 break;
145 }
146
147 if (index < sizeof(receivedLine) - 1)
148 {
149 receivedLine[index++] = receivedChar;
150 }
151 }
152
153 if (ReadCmd(&currentpkg, receivedLine))
154 {
155 state = EXECUTE;
156 char buffer[32];
157 snprintf(buffer, sizeof(buffer), "%s running", CommandToString(currentpkg.cmd));
158
159 Show1Liner(buffer);
160 }
161 else
162 {
163 state = ERROR;
164 }
165 vTaskDelay(pdMS_TO_TICKS(10));
166 break;
167 }
168 case EXECUTE:
169 if (estopFlag == true)
170 {
171 state = ERROR;
172 break;
173 }
174 if (ExecuteCmd(currentpkg))
175 {
176 state = DONE;
177 }
178 else
179 {
180 state = ERROR;
181 }
182 break;
183 case DONE:
184 SendOK(currentpkg.transID);
185 state = IDLE;
186
187 Show1Liner("Idle");
188 break;
189 case ERROR:
190 /* send error and wait for start button */
191 SendError(currentpkg.transID);
192 {
193 char buffer[64];
194 snprintf(buffer, sizeof(buffer), "error at %s, press start to retry",
195 CommandToString(currentpkg.cmd));
196
197 Show1Liner(buffer);
198 }
199
200 while (1)
201 {
202 if (PCA_Read_Pin(1, 2, &pinvalue) == kStatus_Success)
203 {
204 if (pinvalue == 1)
205 {
206 state = IDLE;
207 break;
208 }
209 }
210 vTaskDelay(pdMS_TO_TICKS(100));
211 }
212 break;
213 }
214 }
215}
