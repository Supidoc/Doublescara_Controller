104status_t TMC_init_default(TMC_Handle_t* handle)

126static void process(void)
127{
128 TMC_CommandQueueItem_t queueItem;
129 BaseType_t status;
130 status = xQueueReceive(commandQueue, &queueItem, pdMS_TO_TICKS(10));
131 if (status != pdPASS)
132 {
133 return;
134 }
135
136 switch (queueItem.commandType)
137 {
138 case TMC_CMD_DEFAULT_INIT:
139 default_init(queueItem.handle);
140 break;
141 default:
142 break;
143 }
144}

146static status_t set_microstepping(TMC_Handle_t* handle, TMC_MICROSTEPPING_t microstepping)
147{
148 uint32_t chopConf;
149 read(handle, TMC_CHOPCONF_ADDR, &chopConf);
150 chopConf &= ~(0b1111 << 24);
151 chopConf |= microstepping << 24;
152 char logMsg[60];
153
154 if (kStatus_Success != write(handle, TMC_CHOPCONF_ADDR, &chopConf))
155 {
156 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to write Microstepping to CHOPCONF register",
157 handle->label);
158 LOG_ERROR(logMsg);
159 return kStatus_Fail;
160 }
161 return kStatus_Success;
162}

182static status_t default_init(TMC_Handle_t* handle)
183{
184 static char logMsg[60];
185 snprintf(logMsg, sizeof(logMsg), "[%s] Initializing with default config", handle->label);
186 LOG_INFO(logMsg);
187
188 if (read_transmissioncount(handle) != kStatus_Success)
189 {
190 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to read transmission count", handle->label);
191 LOG_ERROR(logMsg);
192 return kStatus_Fail;
193 }
194
195 // gconf default values
196 uint32_t gconf = 0;
197 gconf |= 0b1; // use external VREF
198 gconf |= 0b0 << 1; // use external sense resistors
199 gconf |= 0b0 << 2; // use StealthChopfp PWM
200 gconf |= 0b0 << 3; // Do not invert Shaft
201 gconf |= 0b0 << 4; // Index Pin shows first microstep position of sequencer
202 gconf |= 0b0 << 5; // Output Index of first microstep
203 gconf |= 0b1 << 6; // PDN_UART input disabled, use uart
204 gconf |= 0b1 << 7; // Microstep resolution set by register
205 gconf |= 0b0 << 8; // No filtering of STEP pulses
206
207 if (kStatus_Success != write(handle, TMC_GCONF_ADDR, &gconf))
208 {
209 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to write GCONF register", handle->label);
210 LOG_ERROR(logMsg);
211 return kStatus_Fail;
212 }
213
214 // reset GSTAT
215 uint32_t gstat = 0b111;
216
217 if (kStatus_Success != write(handle, TMC_GSTAT_ADDR, &gstat))
218 {
219 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to write GSTAT register", handle->label);
220 LOG_ERROR(logMsg);
221 return kStatus_Fail;
222 }
223
224 set_microstepping(handle, TMC_MS_4);
225 set_double_edge_step_pulse(handle, 1);
226
227 snprintf(logMsg, sizeof(logMsg), "[%s] Default init completed successfully", handle->label);
228 LOG_INFO(logMsg);
229 return kStatus_Success;
230}

232static status_t read(TMC_Handle_t* handle, uint8_t regAddr, uint32_t* data)
233{
234 status_t status;
235 TMC_Read_Datagram_t readDatagram;
236
237 create_read_datagram(handle, regAddr, &readDatagram);
238 build_read_package(&readDatagram, transmit_read_package);
239
240 status = UART_RTOS_Send(handle->uartRTOSHandle, transmit_read_package, TMC_READ_PACKAGE_SIZE);
241 dump_rx_buffer(handle, TMC_READ_PACKAGE_SIZE);
242 if (status != kStatus_Success)
243 {
244 char logMsg[60];
245 snprintf(logMsg, sizeof(logMsg), "[%s] UART send failed during read", handle->label);
246 LOG_ERROR(logMsg);
247 return status;
248 }
249
250 status = UART_RTOS_Receive(handle->uartRTOSHandle, transmit_response_package,
251 TMC_WRITE_PACKAGE_SIZE, NULL, TMC_TIMEOUT_MS);
252 if (status != kStatus_Success)
253 {
254 char logMsg[60];
255 snprintf(logMsg, sizeof(logMsg), "[%s] UART receive timeout during read", handle->label);
256 LOG_WARN(logMsg);
257 return status;
258 }
259
260 calc_CRC(transmit_response_package, TMC_WRITE_PACKAGE_SIZE);
261
262 *data = (transmit_response_package[3] << 24) | (transmit_response_package[4] << 16) |
263 (transmit_response_package[5] << 8) | transmit_response_package[6];
264
265 /* Clear response buffer */
266 for (size_t i = 0; i < TMC_WRITE_PACKAGE_SIZE; i++)
267 {
268 transmit_response_package[i] = 0;
269 }
270
271 return kStatus_Success;
272}

274static status_t write(TMC_Handle_t* handle, uint8_t regAddr, uint32_t* data)
275{
276#if TMC_CONFIRM_WRITES == (1)
277 uint8_t expectedTransmissionCount = handle->transmissionCount + 1;
278#endif
279
280 status_t status;
281 TMC_Write_Datagram_t writeDatagram;
282
283 create_write_datagram(handle, regAddr, *data, &writeDatagram);
284 build_write_package(&writeDatagram, transmit_write_package);
285
286 status = UART_RTOS_Send(handle->uartRTOSHandle, transmit_write_package, TMC_WRITE_PACKAGE_SIZE);
287 dump_rx_buffer(handle, TMC_WRITE_PACKAGE_SIZE);
288 if (status != kStatus_Success)
289 {
290 char logMsg[60];
291 snprintf(logMsg, sizeof(logMsg), "[%s] UART send failed during write", handle->label);
292 LOG_ERROR(logMsg);
293 return status;
294 }
295
296#if TMC_CONFIRM_WRITES == (1)
297 status = read_transmissioncount(handle);
298 if (status != kStatus_Success)
299 {
300 char logMsg[70];
301 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to confirm write (transmissioncount)",
302 handle->label);
303 LOG_WARN(logMsg);
304 return status;
305 }
306
307 if (expectedTransmissionCount != handle->transmissionCount)
308 {
309 char logMsg[80];
310 snprintf(logMsg, sizeof(logMsg), "[%s] Write confirmation mismatch: exp=%d got=%d",
311 handle->label, expectedTransmissionCount, handle->transmissionCount);
312 LOG_ERROR(logMsg);
313 return kStatus_Fail;
314 }
315#endif
316
317 return kStatus_Success;
318}

354static inline void calc_CRC(uint8_t* datagram, size_t datagramLength)
355{
356 uint8_t i;
357 uint8_t j;
358 uint8_t* crc = datagram + (datagramLength - 1); // CRC located in last byte of message
359 uint8_t currentByte;
360 *crc = 0;
361 for (i = 0; i < (datagramLength - 1); i++)
362 { // Execute for all bytes of a message
363 currentByte = datagram[i]; // Retrieve a byte to be sent from Array
364 for (j = 0; j < 8; j++)
365 {
366 if ((*crc >> 7) ^ (currentByte & 0x01)) // update CRC based result of XOR operation
367 {
368 *crc = (*crc << 1) ^ 0x07;
369 }
370 else
371 {
372 *crc = (*crc << 1);
373 }
374 currentByte = currentByte >> 1;
375 } // for CRC bit
376 } // for message byte
377}

379static inline void build_read_package(TMC_Read_Datagram_t* datagram, uint8_t* package)
380{
381 package[0] = ((datagram->reserved << 4) & 0xF0) | (datagram->sync & 0x0F);
382 package[1] = datagram->dev_addr & 0xFF;
383 package[2] = datagram->reg_addr & 0xFF;
384 package[3] = datagram->crc & 0xFF;
385 calc_CRC(package, TMC_READ_PACKAGE_SIZE);
386}

388static inline void build_write_package(TMC_Write_Datagram_t* datagram, uint8_t* package)
389{
390 package[0] = ((datagram->reserved << 4) & 0xF0) | (datagram->sync & 0x0F);
391 package[1] = datagram->dev_addr & 0xFF;
392 package[2] = datagram->reg_addr & 0xFF;
393 package[3] = (datagram->data >> 24) & 0xFF;
394 package[4] = (datagram->data >> 16) & 0xFF;
395 package[5] = (datagram->data >> 8) & 0xFF;
396 package[6] = datagram->data & 0xFF;
397 package[7] = datagram->crc & 0xFF;
398 calc_CRC(package, TMC_WRITE_PACKAGE_SIZE);
399}

401static uint8_t dump_rx_buffer(TMC_Handle_t* handle, uint8_t dumpCount)
402{
403 /* Dump received echo */
404 uint8_t dump = 0;
405 size_t n;
406 uint8_t testCount = 0;
407
408 for (uint8_t i = 0; i < dumpCount; i++)
409 {
410 status_t status = UART_RTOS_Receive(handle->uartRTOSHandle, &dump, 1, &n, TMC_TIMEOUT_MS);
411 size_t currentLength = UART_TransferGetRxRingBufferLength(handle->uartHandle);
412 if (status == kStatus_Success && n > 0)
413 {
414 testCount++;
415 }
416 else
417 {
418 LOG_ERROR("Failed to dump RX buffer byte");
419 }
420 }
421
422 return testCount;
