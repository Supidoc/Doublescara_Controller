417status_t TMC_current_to_divider(float desired_current, TMC_RoundingMode_t rounding,
418 uint8_t* divider)
419{
420
421 float reference_current = 1.767766953f; // Reference current for TMC2209 at 100% (1.77A)
422 // calculated from datasheet Rev. 1.09 p. 53
423
424 /* Validate inputs */
425 if (divider == NULL)
426 {
427 return kStatus_Fail;
428 }
429
430 if (desired_current <= 0.0f)
431 {
432 return kStatus_Fail;
433 }
434
435 if (desired_current > reference_current)
436 {
437 return kStatus_Fail;
438 }
439
440 float raw_divider = (desired_current * 32 / reference_current) - 1.0f;
441
442 /* Apply rounding mode */
443 uint8_t rounded_divider;
444 switch (rounding)
445 {
446 case TMC_ROUND_FLOOR:
447 rounded_divider = (uint8_t)floorf(raw_divider);
448 break;
449 case TMC_ROUND_CEIL:
450 rounded_divider = (uint8_t)ceilf(raw_divider);
451 break;
452 case TMC_ROUND_NEAREST:
453 rounded_divider = (uint8_t)roundf(raw_divider);
454 break;
455 default:
456 return kStatus_Fail;
457 }
458
459 /* Validate divider is within valid range [0, 31] for 3-bit TMC2209 register */
460 if (rounded_divider > 31)
461 {
462 return kStatus_Fail;
463 }
464 if (rounded_divider < 0)
465 {
466 rounded_divider = 0;
467 }
468
469 *divider = rounded_divider;
470 return kStatus_Success;
471}

514static void process(void)
515{
516 TMC_CommandQueueItem_t queueItem = {0};
517 BaseType_t status;
518 status = xQueueReceive(cmdQueue, &queueItem, pdMS_TO_TICKS(10));
519 if (status != pdPASS)
520 {
521 return;
522 }
523 status_t cmdStatus = kStatus_Success;
524 switch (queueItem.commandType)
525 {
526 case TMC_CMD_DEFAULT_INIT:
527 cmdStatus = init_handle(queueItem.data.initHandle.config, queueItem.deadline);
528 break;
529 case TMC_CMD_SET_MICROSTEPPING:
530 cmdStatus =
531 set_microstepping(queueItem.handle, queueItem.data.setMicrostepping.microstepping,
532 queueItem.deadline);
533 break;
534 case TMC_CMD_SET_IHOLD_DIVIDER:
535 cmdStatus = set_ihold_divider(queueItem.handle, queueItem.data.setIholdDivider.ihold,
536 queueItem.deadline);
537 if (cmdStatus == kStatus_Success)
538 {
539 queueItem.handle->iholdDivider = queueItem.data.setIholdDivider.ihold;
540 }
541 break;
542 case TMC_CMD_SET_IRUN_DIVIDER:
543 cmdStatus = set_irun_divider(queueItem.handle, queueItem.data.setIrunDivider.irun,
544 queueItem.deadline);
545 if (cmdStatus == kStatus_Success)
546 {
547 queueItem.handle->irunDivider = queueItem.data.setIrunDivider.irun;
548 }
549 break;
550 case TMC_CMD_ENABLE_FREEWHEELING:
551 cmdStatus = set_freewheeling(queueItem.handle, 1, queueItem.deadline);
552 if (cmdStatus == kStatus_Success)
553 {
554 queueItem.handle->freewheeling = 1;
555 }
556 break;
557 case TMC_CMD_DISABLE_FREEWHEELING:
558 cmdStatus = set_freewheeling(queueItem.handle, 0, queueItem.deadline);
559 if (cmdStatus == kStatus_Success)
560 {
561 queueItem.handle->freewheeling = 0;
562 }
563 break;
564 default:
565 cmdStatus = kStatus_Fail;
566 break;
567 }
568
569 if (queueItem.cmdHandle != NULL)
570 {
571 if (queueItem.deadline <= xTaskGetTickCount())
572 {
573 THE_notify_task_timeout(queueItem.cmdHandle);
574 }
575 else if (cmdStatus == kStatus_Success)
576 {
577 THE_notify_task_success(queueItem.cmdHandle);
578 }
579 else
580 {
581 THE_notify_task_failure(queueItem.cmdHandle);
582 }
583
584 THE_remove_cmd_handle_ref(queueItem.cmdHandle);
585 }
586}

588static status_t send_cmd_async(TMC_CommandQueueItem_t* queueItem, TickType_t deadline,
589 THE_CmdHandle_t* cmdHandle)
590{
591 if (queueItem == NULL)
592 {
593 return kStatus_Fail;
594 }
595 THE_CmdHandle_t internaleCmdHandle = NULL;
596
597 status_t allocStatus =
598 THE_get_cmd_handle(&internaleCmdHandle, cmdHandles, TMC_MAX_CMD_HANDLE_COUNT);
599 if (allocStatus != kStatus_Success)
600 {
601 return kStatus_Fail;
602 }
603 if (cmdHandle != NULL)
604 {
605 THE_add_cmd_handle_ref(internaleCmdHandle);
606 }
607
608 queueItem->cmdHandle = internaleCmdHandle;
609
610 status_t sendStatus = THE_send_cmd(cmdQueue, queueItem, deadline, internaleCmdHandle);
611 if (sendStatus != kStatus_Success)
612 {
613 if (cmdHandle != NULL)
614 {
615 THE_remove_cmd_handle_ref(*cmdHandle);
616 *cmdHandle = NULL;
617 }
618 return kStatus_Fail;
619 }
620 if (cmdHandle != NULL)
621 {
622 *cmdHandle = internaleCmdHandle;
623 }
624 return kStatus_Success;
625}

627static status_t set_ihold_divider(TMC_Handle_t handle, uint8_t ihold, TickType_t deadline)
628{
629
630 if (handle == NULL)
631 {
632 return kStatus_Fail;
633 }
634
635 uint32_t ihold_irun;
636 static char logMsg[60];
637
638 if (read(handle, TMC_IHOLD_IRUN_ADDR, &ihold_irun, deadline) != kStatus_Success)
639 {
640 snprintf(logMsg, sizeof(logMsg),
641 "[%s] Failed to read IHOLD register - driver communication error", handle->label);
642 LOG_ERROR(logMsg);
643 return kStatus_Fail;
644 }
645
646 ihold_irun &= ~(0b11111 << 0); // Clear IHOLD bits
647 ihold_irun |= (ihold & 0b11111) << 0; // Set new IHOLD bits
648
649 if (write(handle, TMC_IHOLD_IRUN_ADDR, &ihold_irun, deadline) != kStatus_Success)
650 {
651 snprintf(logMsg, sizeof(logMsg),
652 "[%s] Failed to write IHOLD register - driver communication error", handle->label);
653 LOG_ERROR(logMsg);
654 return kStatus_Fail;
655 }
656
657 snprintf(logMsg, sizeof(logMsg), "[%s] Hold current divider set to %d", handle->label, ihold);
658 LOG_INFO(logMsg);
659
660 return kStatus_Success;
661}

700static status_t set_microstepping(TMC_Handle_t handle, TMC_MICROSTEPPING_t microstepping,
701 TickType_t deadline)
702{
703 if (handle == NULL)
704 {
705 return kStatus_Fail;
706 }
707
708 static char logMsg[60];
709
710 uint32_t chopConf;
711 read(handle, TMC_CHOPCONF_ADDR, &chopConf, deadline);
712 chopConf &= ~(0b1111 << 24);
713 chopConf |= microstepping << 24;
714
715 if (kStatus_Success != write(handle, TMC_CHOPCONF_ADDR, &chopConf, deadline))
716 {
717 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to write Microstepping to CHOPCONF register",
718 handle->label);
719 LOG_ERROR(logMsg);
720
721 return kStatus_Fail;
722 }
723 uint16_t microsteppingInteger;
724 TMC_microstepping_enum_to_uint(microstepping, &microsteppingInteger);
725 snprintf(logMsg, sizeof(logMsg), "[%s] Microstepping set to %d", handle->label,
726 microsteppingInteger);
727 LOG_INFO(logMsg);
728
729 return kStatus_Success;
730}

732static status_t set_double_edge_step_pulse(TMC_Handle_t handle, uint8_t enabled,
733 TickType_t deadline)
734{
735 if (handle == NULL)
736 {
737 return kStatus_Fail;
738 }
739
740 uint32_t chopConf;
741 read(handle, TMC_CHOPCONF_ADDR, &chopConf, deadline);
742 chopConf &= ~(0b1 << 29);
743 chopConf |= (enabled && 0b1) << 29;
744 char logMsg[60];
745
746 if (kStatus_Success != write(handle, TMC_CHOPCONF_ADDR, &chopConf, deadline))
747 {
748 snprintf(logMsg, sizeof(logMsg),
749 "[%s] Failed to write Double Edge Pulse to CHOPCONF register", handle->label);
750 LOG_ERROR(logMsg);

757{

773 }

798 }

804 gconf |= 0b0 << 2; // use StealthChopfp PWM
805 gconf |= 0b0 << 3; // Do not invert Shaft

808 gconf |= 0b1 << 6; // PDN_UART input disabled, use uart

810 gconf |= 0b0 << 8; // No filtering of STEP pulses

818 LOG_FATAL(logMsg);
819
820 return kStatus_Fail;
821 }
822
823 // reset GSTAT
824 uint32_t gstat = 0b111;
825
826 if (kStatus_Success != write(handle, TMC_GSTAT_ADDR, &gstat, deadline))
827 {
828 free_handle(handle);
829
830 snprintf(logMsg, sizeof(logMsg),
831 "[%s] Failed to write GSTAT register - initialization failure", handle->label);
832 LOG_FATAL(logMsg);
833
834 return kStatus_Fail;
835 }
836 if (set_microstepping(handle, config.microstepping, deadline) != kStatus_Success)
837 {
838 free_handle(handle);
839
840 snprintf(logMsg, sizeof(logMsg),
841 "[%s] Failed to set microstepping - initialization failure", handle->label);
842 LOG_FATAL(logMsg);
843
844 return kStatus_Fail;
845 }
846
847 if (set_double_edge_step_pulse(handle, 1, deadline) != kStatus_Success)
848 {
849 free_handle(handle);
850
851 snprintf(logMsg, sizeof(logMsg),
852 "[%s] Failed to set double edge step pulse - initialization failure",
853 handle->label);
854 LOG_FATAL(logMsg);
855
856 return kStatus_Fail;
857 }
858
859 if (set_irun_divider(handle, config.irunDivider, deadline) != kStatus_Success)
860 {
861 free_handle(handle);

863 snprintf(logMsg, sizeof(logMsg),
864 "[%s] Failed to set run current divider - initialization failure", handle->label);
865 LOG_FATAL(logMsg);
866

868 }

875 "[%s] Failed to set hold current divider - initialization failure", handle->label);
876 LOG_FATAL(logMsg);
877
878 return kStatus_Fail;
879 }
880
881 snprintf(logMsg, sizeof(logMsg), "[%s] TMC2209 initialized successfully", handle->label);
882 LOG_INFO(logMsg);

884 return kStatus_Success;

906static status_t read(TMC_Handle_t handle, uint8_t regAddr, uint32_t* data, TickType_t deadline)
907{
908 if (handle == NULL || data == NULL)
909 {
910 return kStatus_Fail;
911 }
912
913 status_t status;
914 TMC_Read_Datagram_t readDatagram;
915 char logMsg[60];
916
917 create_read_datagram(handle, regAddr, &readDatagram);
918 build_read_package(&readDatagram, transmit_read_package);
919 TickType_t currentTick = xTaskGetTickCount();
920 if (deadline <= currentTick)
921 {
922 snprintf(logMsg, sizeof(logMsg), "[%s] Deadline for UART receive has already passed",
923 handle->label);
924 LOG_ERROR(logMsg);
925 return kStatus_Fail;
926 }
927 TickType_t ticksUntilDeadline = deadline - currentTick;
928 PORT_SetPinMux(PORTE, 0, kPORT_MuxAlt3);
929 status = UART_RTOS_Send(handle->uartRTOSHandle, transmit_read_package, TMC_READ_PACKAGE_SIZE);
930 PORT_SetPinMux(PORTE, 0, kPORT_MuxAlt1);
931 dump_rx_buffer(handle, TMC_READ_PACKAGE_SIZE, deadline);
932 if (status != kStatus_Success)
933 {
934 char logMsg[60];
935 snprintf(logMsg, sizeof(logMsg), "[%s] UART send failed during read", handle->label);
936 LOG_ERROR(logMsg);
937 return status;
938 }
939
940 status = UART_RTOS_Receive(handle->uartRTOSHandle, transmit_response_package,
941 TMC_WRITE_PACKAGE_SIZE, NULL, ticksUntilDeadline);
942 if (status != kStatus_Success)
943 {
944 snprintf(logMsg, sizeof(logMsg), "[%s] UART receive timeout during read", handle->label);
945 LOG_WARN(logMsg);
946 return status;
947 }
948
949 calc_CRC(transmit_response_package, TMC_WRITE_PACKAGE_SIZE);
950
951 *data = (transmit_response_package[3] << 24) | (transmit_response_package[4] << 16) |
952 (transmit_response_package[5] << 8) | transmit_response_package[6];
953
954 /* Clear response buffer */
955 for (size_t i = 0; i < TMC_WRITE_PACKAGE_SIZE; i++)
956 {
957 transmit_response_package[i] = 0;
958 }
959
960 return kStatus_Success;
961}

1064static inline void calc_CRC(uint8_t* datagram, size_t datagramLength)
1065{
1066 if (datagram == NULL || datagramLength == 0)
1067 {
1068 return;
1069 }
1070
1071 uint8_t i;
1072 uint8_t j;
1073 uint8_t* crc = datagram + (datagramLength - 1); // CRC located in last byte of message
1074 uint8_t currentByte;
1075 *crc = 0;
1076 for (i = 0; i < (datagramLength - 1); i++)
1077 { // Execute for all bytes of a message
1078 currentByte = datagram[i]; // Retrieve a byte to be sent from Array
1079 for (j = 0; j < 8; j++)
1080 {
1081 if ((*crc >> 7) ^ (currentByte & 0x01)) // update CRC based result of XOR operation
1082 {
1083 *crc = (*crc << 1) ^ 0x07;
1084 }
1085 else
1086 {
1087 *crc = (*crc << 1);
1088 }
1089 currentByte = currentByte >> 1;
1090 } // for CRC bit
1091 } // for message byte
1092}
