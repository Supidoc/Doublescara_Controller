672static inline status_t EnableIRQ(IRQn_Type interrupt)
673{
674 status_t status = kStatus_Success;
675
676 if (NotAvail_IRQn == interrupt)
677 {
678 status = kStatus_Fail;
679 }
680
681#if defined(FSL_FEATURE_NUMBER_OF_LEVEL1_INT_VECTORS) && (FSL_FEATURE_NUMBER_OF_LEVEL1_INT_VECTORS > 0)
682 else if ((int32_t)interrupt >= (int32_t)FSL_FEATURE_NUMBER_OF_LEVEL1_INT_VECTORS)
683 {
684#if defined(FSL_FEATURE_IRQSTEER_EXT_INT_MAX_NUM) && (FSL_FEATURE_IRQSTEER_EXT_INT_MAX_NUM > 0) && defined(FSL_FEATURE_IRQSTEER_IRQ_START_INDEX) && (FSL_FEATURE_IRQSTEER_IRQ_START_INDEX > 0)
685 int32_t irqsteerInstIdx = (int32_t)((interrupt + 1 - FSL_FEATURE_IRQSTEER_IRQ_START_INDEX) / FSL_FEATURE_IRQSTEER_EXT_INT_MAX_NUM);
686
687 IRQSTEER_EnableInterrupt(irqsteerInstIdx, interrupt);
688#else
689 status = kStatus_Fail;
690#endif
691 }
692#endif
693
694 else
695 {
696#if defined(__GIC_PRIO_BITS)
697 GIC_EnableIRQ(interrupt);
698#else
699 NVIC_EnableIRQ(interrupt);
700#endif
701 }
702
703 return status;
704}
