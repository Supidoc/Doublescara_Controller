92
93 STP_init_stepper(config);
94
95 LOG_DEBUG("Configured stepper motor: motor0");
96
97 TMC_Handle_t tmcHandle;
98 tmcHandle.label = "tmc0";
99 tmcHandle.serialAdress = TMC_SERIAL_ADDRESS_0;
100 tmcHandle.transmissionCount = 0;
101 tmcHandle.uartHandle = &UART1_uart_handle;
102 tmcHandle.uartRTOSHandle = &UART1_rtos_handle;
103
104 if (TMC_init_default(&tmcHandle) == kStatus_Success)
105 LOG_DEBUG("Configured TMC2209 driver: tmc0");
106
107 for (;;)
108 {
109 vTaskDelay(1);
110 }

117BaseType_t motor_stop(char* pcWriteBuffer, size_t xWriteBufferLen, const char* pcCommandString)
118{
119 STP_StepperHandle_t* handle;
120 uint32_t stepCount;
121 STP_Direction_t direction;
122 char logMsg[80];
123
124 uint8_t parameterFound;
125 const char* pcParameter = NULL;
126 BaseType_t parameterStringLength;
127 uint8_t doDeceleration = 0;
128
129 CLU_get_parameter_value_string("-m", pcCommandString, &parameterFound, &pcParameter,
130 &parameterStringLength);
131 if (parameterFound == 0 || parameterStringLength == 0)
132 {
133 LOG_WARN("Motor command received without motor ID");
134 strncpy(pcWriteBuffer, "Please specify a motorId", strlen("Please specify a motorId") + 1);
135 return pdFALSE;
136 }
137 else
138 {
139 if (STP_get_handle_by_id(pcParameter, &handle) != kStatus_Success)
140 {
141 snprintf(logMsg, sizeof(logMsg), "Invalid motor ID requested: %.*s",
142 (int)parameterStringLength, pcParameter);
143 LOG_WARN(logMsg);
144 strncpy(pcWriteBuffer, "Invalid motorId", strlen("Invalid motorId") + 1);
145 return pdFALSE;
146 }
147 }
148
149 CLU_get_parameter_value_string("-d", pcCommandString, &parameterFound, &pcParameter,
150 &parameterStringLength);
151 if (parameterFound == 1)
152 {
153 doDeceleration = 1;
154 }
155
156 STP_stop(handle, doDeceleration);
157 return pdFALSE;
158}

160BaseType_t motor_move_command(char* pcWriteBuffer, size_t xWriteBufferLen,
161 const char* pcCommandString)
162{
163
164 STP_StepperHandle_t* handle;
165 uint32_t stepCount;
166 STP_Direction_t direction;
167 char logMsg[80];
168
169 uint8_t parameterFound;
170 const char* pcParameter = NULL;
171 BaseType_t parameterStringLength;
172
173 CLU_get_parameter_value_string("-m", pcCommandString, &parameterFound, &pcParameter,
174 &parameterStringLength);
175 if (parameterFound == 0 || parameterStringLength == 0)
176 {
177 strncpy(pcWriteBuffer, "Please specify a motorId", strlen("Please specify a motorId") + 1);
178 return pdFALSE;
179 }
180 else
181 {
182 if (STP_get_handle_by_id(pcParameter, &handle) != kStatus_Success)
183 {
184 strncpy(pcWriteBuffer, "Invalid motorId", strlen("Invalid motorId") + 1);
185 return pdFALSE;
186 }
187 }
188
189 CLU_get_parameter_value_string("-s", pcCommandString, &parameterFound, &pcParameter,
190 &parameterStringLength);
191 if (parameterFound == 0 || parameterStringLength == 0)
192 {
193 strncpy(pcWriteBuffer, "Please specify a step count",
194 strlen("Please specify a step count") + 1);
195 return pdFALSE;
196 }
197 else
198 {
199 stepCount = strtoul(pcParameter, NULL, 0);
200 }
201
202 CLU_get_parameter_value_string("-d", pcCommandString, &parameterFound, &pcParameter,
203 &parameterStringLength);
204 if (parameterFound == 0 || parameterStringLength == 0)
205 {
206 strncpy(pcWriteBuffer, "Please specify a direction",
207 strlen("Please specify a direction") + 1);
208 return pdFALSE;
209 }
210 else
211 {
212 if (strncmp("cw", pcParameter, 2) == 0)
213 {
214 direction = STP_CLOCKWISE;
215 }
216 else if (strncmp("ccw", pcParameter, 3) == 0)
217 {
218 direction = STP_COUNTERCLOCKWISE;
219 }
220 else
221 {
222 strncpy(pcWriteBuffer, "Invalid direction", strlen("Invalid direction") + 1);
223 return pdFALSE;
224 }
225 }
226
227 STP_move_relative(handle, stepCount, direction);
228 return pdFALSE;
229}
