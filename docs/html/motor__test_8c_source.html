170void init_motor(MTR_MotorConfig_t config)
171{
172 char logMsg[60];
173 TickType_t deadline = SYW_deadline_from_timeout_ms(400);
174 CHD_CmdHandle_t cmdHandle = NULL;
175 if (MTR_init_handle_async(config, deadline, &cmdHandle) == kStatus_Success &&
176 SYW_cmd_wait_result(cmdHandle, deadline, NULL) == kStatus_Success)
177 {
178 CHD_remove_cmd_handle_ref(cmdHandle);
179 snprintf(logMsg, sizeof(logMsg),
180 ""
181 "[%s] Motor init completed",
182 config.label);
183 LOG_INFO(logMsg);
184 }
185 else
186 {
187 CHD_remove_cmd_handle_ref(cmdHandle);
188 snprintf(logMsg, sizeof(logMsg), "[%s] Motor init failed or timed out", config.label);
189 LOG_ERROR(logMsg);
190 }
191}

193static status_t MTT_wait_and_release(CHD_CmdHandle_t cmdHandle, TickType_t deadline)
194{
195 if (cmdHandle == NULL)
196 {
197 return kStatus_Fail;
198 }
199
200 status_t status = SYW_cmd_wait_result(cmdHandle, deadline, NULL);
201 CHD_remove_cmd_handle_ref(cmdHandle);
202 return status;
203}
