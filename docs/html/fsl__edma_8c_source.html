63static uint32_t EDMA_GetInstance(DMA_Type *base)
64{
65 uint32_t instance;
66
67 /* Find the instance index from base address mappings. */
68 for (instance = 0; instance < ARRAY_SIZE(s_edmaBases); instance++)
69 {
70 if (MSDK_REG_SECURE_ADDR(s_edmaBases[instance]) == MSDK_REG_SECURE_ADDR(base))
71 {
72 break;
73 }
74 }
75
76 assert(instance < ARRAY_SIZE(s_edmaBases));
77
78 return instance;
79}

88void EDMA_InstallTCD(DMA_Type *base, uint32_t channel, edma_tcd_t *tcd)
89{
90 assert(channel < (uint32_t)FSL_FEATURE_EDMA_MODULE_CHANNEL);
91 assert(tcd != NULL);
92 assert(((uint32_t)tcd & 0x1FU) == 0U);
93
94 /* Push tcd into hardware TCD register */
95 base->TCD[channel].SADDR = tcd->SADDR;
96 base->TCD[channel].SOFF = tcd->SOFF;
97 base->TCD[channel].ATTR = tcd->ATTR;
98 base->TCD[channel].NBYTES_MLNO = tcd->NBYTES;
99 base->TCD[channel].SLAST = (int32_t)tcd->SLAST;
100 base->TCD[channel].DADDR = tcd->DADDR;
101 base->TCD[channel].DOFF = tcd->DOFF;
102 base->TCD[channel].CITER_ELINKNO = tcd->CITER;
103 base->TCD[channel].DLAST_SGA = (int32_t)tcd->DLAST_SGA;
104 /* Clear DONE bit first, otherwise ESG cannot be set */
105 base->TCD[channel].CSR = 0;
106 base->TCD[channel].CSR = tcd->CSR;
107 base->TCD[channel].BITER_ELINKNO = tcd->BITER;

120void EDMA_Init(DMA_Type *base, const edma_config_t *config)
121{
122 assert(config != NULL);
123
124 uint32_t tmpreg;
125
126#if !(defined(FSL_SDK_DISABLE_DRIVER_CLOCK_CONTROL) && FSL_SDK_DISABLE_DRIVER_CLOCK_CONTROL)
127 /* Ungate EDMA peripheral clock */
128 CLOCK_EnableClock(s_edmaClockName[EDMA_GetInstance(base)]);
129#endif /* FSL_SDK_DISABLE_DRIVER_CLOCK_CONTROL */
130
131 /* clear all the enabled request, status to make sure EDMA status is in normal condition */
132 base->ERQ = 0U;
133 base->INT = 0xFFFFFFFFU;
134 base->ERR = 0xFFFFFFFFU;
135 /* Configure EDMA peripheral according to the configuration structure. */
136 tmpreg = base->CR;
137 tmpreg &= ~(DMA_CR_ERCA_MASK | DMA_CR_HOE_MASK | DMA_CR_CLM_MASK | DMA_CR_EDBG_MASK);
138 tmpreg |= (DMA_CR_ERCA(config->enableRoundRobinArbitration) | DMA_CR_HOE(config->enableHaltOnError) |
139 DMA_CR_CLM(config->enableContinuousLinkMode) | DMA_CR_EDBG(config->enableDebugMode) | DMA_CR_EMLM(1U));
140 base->CR = tmpreg;
141}

387{
388 assert(channel < (uint32_t)FSL_FEATURE_EDMA_MODULE_CHANNEL);
389
390 /* Enable error interrupt */
391 if (0U != (mask & (uint32_t)kEDMA_ErrorInterruptEnable))
392 {
393 base->EEI |= ((uint32_t)0x1U << channel);
394 }
395
396 /* Enable Major interrupt */
397 if (0U != (mask & (uint32_t)kEDMA_MajorInterruptEnable))
398 {
399 base->TCD[channel].CSR |= DMA_CSR_INTMAJOR_MASK;
400 }
401
402 /* Enable Half major interrupt */
403 if (0U != (mask & (uint32_t)kEDMA_HalfInterruptEnable))
404 {
405 base->TCD[channel].CSR |= DMA_CSR_INTHALF_MASK;
406 }
407}

495void EDMA_TcdSetTransferConfig(edma_tcd_t *tcd, const edma_transfer_config_t *config, edma_tcd_t *nextTcd)
496{
497 assert(tcd != NULL);
498 assert(((uint32_t)tcd & 0x1FU) == 0U);
499 assert(config != NULL);
500 assert(((uint32_t)nextTcd & 0x1FU) == 0U);
501 assert((config->srcAddr % (1UL << (uint32_t)config->srcTransferSize)) == 0U);
502 assert((config->destAddr % (1UL << (uint32_t)config->destTransferSize)) == 0U);
503
504 /* source address */
505#if defined FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET && FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET
506 tcd->SADDR = MEMORY_ConvertMemoryMapAddress(config->srcAddr, kMEMORY_Local2DMA);
507 /* destination address */
508 tcd->DADDR = MEMORY_ConvertMemoryMapAddress(config->destAddr, kMEMORY_Local2DMA);
509#else
510 tcd->SADDR = config->srcAddr;
511 /* destination address */
512 tcd->DADDR = config->destAddr;
513#endif /* FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET */
514 /* Source data and destination data transfer size */
515 tcd->ATTR = DMA_ATTR_SSIZE(config->srcTransferSize) | DMA_ATTR_DSIZE(config->destTransferSize);
516 /* Source address signed offset */
517 tcd->SOFF = (uint16_t)config->srcOffset;
518 /* Destination address signed offset */
519 tcd->DOFF = (uint16_t)config->destOffset;
520 /* Minor byte transfer count */
521 tcd->NBYTES = config->minorLoopBytes;
522 /* Current major iteration count */
523 tcd->CITER = (uint16_t)config->majorLoopCounts;
524 /* Starting major iteration count */
525 tcd->BITER = (uint16_t)config->majorLoopCounts;
526 /* Enable scatter/gather processing */
527 if (nextTcd != NULL)
528 {
529#if defined FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET && FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET
530 tcd->DLAST_SGA = MEMORY_ConvertMemoryMapAddress((uint32_t)nextTcd, kMEMORY_Local2DMA);
531#else
532 tcd->DLAST_SGA = (uint32_t)nextTcd;
533#endif /* FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET */
534 /*
535 Before call EDMA_TcdSetTransferConfig or EDMA_SetTransferConfig,
536 user must call EDMA_TcdReset or EDMA_ResetChannel which will set
537 DREQ, so must use "|" or "&" rather than "=".
538
539 Clear the DREQ bit because scatter gather has been enabled, so the
540 previous transfer is not the last transfer, and channel request should
541 be enabled at the next transfer(the next TCD).
542 */
543 tcd->CSR = (tcd->CSR | (uint16_t)DMA_CSR_ESG_MASK) & ~(uint16_t)DMA_CSR_DREQ_MASK;
544 }
545}

604void EDMA_TcdSetChannelLink(edma_tcd_t *tcd, edma_channel_link_type_t linkType, uint32_t linkedChannel)
605{
606 assert(tcd != NULL);
607 assert(((uint32_t)tcd & 0x1FU) == 0U);
608 assert(linkedChannel < (uint32_t)FSL_FEATURE_EDMA_MODULE_CHANNEL);
609
610 if (linkType == kEDMA_MinorLink) /* Minor link config */
611 {
612 uint16_t tmpreg;
613
614 /* Enable minor link */
615 tcd->CITER |= DMA_CITER_ELINKYES_ELINK_MASK;
616 tcd->BITER |= DMA_BITER_ELINKYES_ELINK_MASK;
617 /* Set linked channel */
618 tmpreg = tcd->CITER & (~(uint16_t)DMA_CITER_ELINKYES_LINKCH_MASK);
619 tmpreg |= DMA_CITER_ELINKYES_LINKCH(linkedChannel);
620 tcd->CITER = tmpreg;
621 tmpreg = tcd->BITER & (~(uint16_t)DMA_BITER_ELINKYES_LINKCH_MASK);
622 tmpreg |= DMA_BITER_ELINKYES_LINKCH(linkedChannel);
623 tcd->BITER = tmpreg;
624 }
625 else if (linkType == kEDMA_MajorLink) /* Major link config */
626 {
627 uint16_t tmpreg;
628
629 /* Enable major link */
630 tcd->CSR |= DMA_CSR_MAJORELINK_MASK;
631 /* Set major linked channel */
632 tmpreg = tcd->CSR & (~(uint16_t)DMA_CSR_MAJORLINKCH_MASK);
633 tcd->CSR = tmpreg | DMA_CSR_MAJORLINKCH(linkedChannel);
634 }
635 else /* Link none */
636 {
637 tcd->CITER &= ~(uint16_t)DMA_CITER_ELINKYES_ELINK_MASK;
638 tcd->BITER &= ~(uint16_t)DMA_BITER_ELINKYES_ELINK_MASK;
639 tcd->CSR &= ~(uint16_t)DMA_CSR_MAJORELINK_MASK;
640 }
641}

734uint32_t EDMA_GetRemainingMajorLoopCount(DMA_Type *base, uint32_t channel)
735{
736 assert(channel < (uint32_t)FSL_FEATURE_EDMA_MODULE_CHANNEL);
737
738 uint32_t remainingCount = 0;
739
740 if (0U != (DMA_CSR_DONE_MASK & base->TCD[channel].CSR))
741 {
742 remainingCount = 0;
743 }
744 else
745 {
746 /* Calculate the unfinished bytes */
747 if (0U != (base->TCD[channel].CITER_ELINKNO & DMA_CITER_ELINKNO_ELINK_MASK))
748 {
749 remainingCount = (((uint32_t)base->TCD[channel].CITER_ELINKYES & DMA_CITER_ELINKYES_CITER_MASK) >>
750 DMA_CITER_ELINKYES_CITER_SHIFT);
751 }
752 else
753 {
754 remainingCount = (((uint32_t)base->TCD[channel].CITER_ELINKNO & DMA_CITER_ELINKNO_CITER_MASK) >>
755 DMA_CITER_ELINKNO_CITER_SHIFT);
756 }
757 }
758
759 return remainingCount;
760}

845void EDMA_CreateHandle(edma_handle_t *handle, DMA_Type *base, uint32_t channel)
846{
847 assert(handle != NULL);
848 assert(channel < (uint32_t)FSL_FEATURE_EDMA_MODULE_CHANNEL);
849
850 uint32_t edmaInstance;
851 uint32_t channelIndex;
852 edma_tcd_t *tcdRegs;
853
854 /* Zero the handle */
855 (void)memset(handle, 0, sizeof(*handle));
856
857 handle->base = base;
858 handle->channel = (uint8_t)channel;
859
860 /* Get the DMA instance number */
861 edmaInstance = EDMA_GetInstance(base);
862 channelIndex = (EDMA_GetInstanceOffset(edmaInstance) * (uint32_t)FSL_FEATURE_EDMA_MODULE_CHANNEL) + channel;
863 s_EDMAHandle[channelIndex] = handle;
864
865 /* Enable NVIC interrupt */
866 (void)EnableIRQ(s_edmaIRQNumber[edmaInstance][channel]);
867
868 /*
869 Reset TCD registers to zero. Unlike the EDMA_TcdReset(DREQ will be set),
870 CSR will be 0. Because in order to suit EDMA busy check mechanism in
871 EDMA_SubmitTransfer, CSR must be set 0.
872 */
873 tcdRegs = (edma_tcd_t *)(uint32_t)&handle->base->TCD[handle->channel];
874 tcdRegs->SADDR = 0;
875 tcdRegs->SOFF = 0;
876 tcdRegs->ATTR = 0;
877 tcdRegs->NBYTES = 0;
878 tcdRegs->SLAST = 0;
879 tcdRegs->DADDR = 0;
880 tcdRegs->DOFF = 0;
881 tcdRegs->CITER = 0;
882 tcdRegs->DLAST_SGA = 0;
883 tcdRegs->CSR = 0;
884 tcdRegs->BITER = 0;
885}

904 /* Initialize tcd queue attribute. */
905 /* header should initial as 1, since that it is used to point to the next TCD to be loaded into TCD memory,
906 * In EDMA driver IRQ handler, header will be used to calculate how many tcd has done, for example,
907 * If application submit 4 transfer request, A->B->C->D,
908 * when A finshed, the header is 0, C is the next TCD to be load, since B is already loaded,
909 * according to EDMA driver IRQ handler, tcdDone = C - A - header = 2 - header = 2, but actually only 1 TCD done,
910 * so the issue will be the wrong TCD done count will pass to application in first TCD interrupt.
911 * During first submit, the header should be assigned to 1, since 0 is current one and 1 is next TCD to be loaded,
912 * but software cannot know which submission is the first one, so assign 1 to header here.
913 */
914 handle->header = 1;
915 handle->tcdUsed = 0;
916 handle->tcdSize = (int8_t)tcdSize;
917 handle->flags = 0;
918 handle->tcdPool = tcdPool;
919}

1109status_t EDMA_SubmitTransfer(edma_handle_t *handle, const edma_transfer_config_t *config)
1110{
1111 assert(handle != NULL);
1112 assert(config != NULL);
1113
1114 edma_tcd_t *tcdRegs = (edma_tcd_t *)(uint32_t)&handle->base->TCD[handle->channel];
1115
1116 if (handle->tcdPool == NULL)
1117 {
1118 /*
1119 * Check if EDMA channel is busy:
1120 * 1. if channel active bit is set, it implies that minor loop is executing, then channel is busy
1121 * 2. if channel active bit is not set and BITER not equal to CITER, it implies that major loop is executing,
1122 * then channel is busy
1123 *
1124 * There is one case can not be covered in below condition:
1125 * When transfer request is submitted, but no request from peripheral, that is to say channel sevice doesn't
1126 * begin, if application would like to submit another transfer , then the TCD will be overwritten, since the
1127 * ACTIVE is 0 and BITER = CITER, for such case, it is a scatter gather(link TCD) case actually, so
1128 * application should enabled TCD pool for dynamic scatter gather mode by calling EDMA_InstallTCDMemory.
1129 */
1130 if (((handle->base->TCD[handle->channel].CSR & DMA_CSR_ACTIVE_MASK) != 0U) ||
1131 (((handle->base->TCD[handle->channel].CITER_ELINKNO & DMA_CITER_ELINKNO_CITER_MASK) !=
1132 (handle->base->TCD[handle->channel].BITER_ELINKNO & DMA_BITER_ELINKNO_BITER_MASK))))
1133 {
1134 return kStatus_EDMA_Busy;
1135 }
1136 else
1137 {
1138 EDMA_SetTransferConfig(handle->base, handle->channel, config, NULL);
1139 /* Enable auto disable request feature */
1140 handle->base->TCD[handle->channel].CSR |= DMA_CSR_DREQ_MASK;
1141 /* Enable major interrupt */
1142 handle->base->TCD[handle->channel].CSR |= DMA_CSR_INTMAJOR_MASK;
1143
1144 return kStatus_Success;
1145 }
1146 }
1147 else /* Use the TCD queue. */
1148 {
1149 uint32_t primask;
1150 uint16_t csr;
1151 int8_t currentTcd;
1152 int8_t previousTcd;
1153 int8_t nextTcd;
1154 int8_t tmpTcdUsed;
1155 int8_t tmpTcdSize;
1156
1157 /* Check if tcd pool is full. */
1158 primask = DisableGlobalIRQ();
1159 tmpTcdUsed = handle->tcdUsed;
1160 tmpTcdSize = handle->tcdSize;
1161 if (tmpTcdUsed >= tmpTcdSize)
1162 {
1163 EnableGlobalIRQ(primask);
1164
1165 return kStatus_EDMA_QueueFull;
1166 }
1167 currentTcd = handle->tail;
1168 handle->tcdUsed++;
1169 /* Calculate index of next TCD */
1170 nextTcd = currentTcd + 1;
1171 if (nextTcd == handle->tcdSize)
1172 {
1173 nextTcd = 0;
1174 }
1175 /* Advance queue tail index */
1176 handle->tail = nextTcd;
1177 EnableGlobalIRQ(primask);
1178 /* Calculate index of previous TCD */
1179 previousTcd = currentTcd != 0 ? currentTcd - 1 : (handle->tcdSize - 1);
1180 /* Configure current TCD block. */
1181 EDMA_TcdReset(&handle->tcdPool[currentTcd]);
1182 EDMA_TcdSetTransferConfig(&handle->tcdPool[currentTcd], config, NULL);
1183 /* Enable major interrupt */
1184 handle->tcdPool[currentTcd].CSR |= DMA_CSR_INTMAJOR_MASK;
1185 /* Link current TCD with next TCD for identification of current TCD */
1186#if defined FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET && FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET
1187 handle->tcdPool[currentTcd].DLAST_SGA =
1188 MEMORY_ConvertMemoryMapAddress((uint32_t)&handle->tcdPool[nextTcd], kMEMORY_Local2DMA);
1189#else
1190 handle->tcdPool[currentTcd].DLAST_SGA = (uint32_t)&handle->tcdPool[nextTcd];
1191#endif /* FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET */
1192 /* Chain from previous descriptor unless tcd pool size is 1(this descriptor is its own predecessor). */
1193 if (currentTcd != previousTcd)
1194 {
1195 /* Enable scatter/gather feature in the previous TCD block. */
1196 csr = handle->tcdPool[previousTcd].CSR | ((uint16_t)DMA_CSR_ESG_MASK);
1197 csr &= ~((uint16_t)DMA_CSR_DREQ_MASK);
1198 handle->tcdPool[previousTcd].CSR = csr;
1199 /*
1200 Check if the TCD block in the registers is the previous one (points to current TCD block). It
1201 is used to check if the previous TCD linked has been loaded in TCD register. If so, it need to
1202 link the TCD register in case link the current TCD with the dead chain when TCD loading occurs
1203 before link the previous TCD block.
1204 */
1205#if defined FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET && FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET
1206 if (tcdRegs->DLAST_SGA ==
1207 MEMORY_ConvertMemoryMapAddress((uint32_t)&handle->tcdPool[currentTcd], kMEMORY_Local2DMA))
1208#else
1209 if (tcdRegs->DLAST_SGA == (uint32_t)&handle->tcdPool[currentTcd])
1210#endif /* FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET */
1211 {
1212 /* Clear the DREQ bits for the dynamic scatter gather */
1213 tcdRegs->CSR |= DMA_CSR_DREQ_MASK;
1214 /* Enable scatter/gather also in the TCD registers. */
1215 csr = tcdRegs->CSR | DMA_CSR_ESG_MASK;
1216 /* Must write the CSR register one-time, because the transfer maybe finished anytime. */
1217 tcdRegs->CSR = csr;
1218 /*
1219 It is very important to check the ESG bit!
1220 Because this hardware design: if DONE bit is set, the ESG bit can not be set. So it can
1221 be used to check if the dynamic TCD link operation is successful. If ESG bit is not set
1222 and the DLAST_SGA is not the next TCD address(it means the dynamic TCD link succeed and
1223 the current TCD block has been loaded into TCD registers), it means transfer finished
1224 and TCD link operation fail, so must install TCD content into TCD registers and enable
1225 transfer again. And if ESG is set, it means transfer has not finished, so TCD dynamic
1226 link succeed.
1227 */
1228 if (0U != (tcdRegs->CSR & DMA_CSR_ESG_MASK))
1229 {
1230 tcdRegs->CSR &= ~(uint16_t)DMA_CSR_DREQ_MASK;
1231 return kStatus_Success;
1232 }
1233 /*
1234 Check whether the current TCD block is already loaded in the TCD registers. It is another
1235 condition when ESG bit is not set: it means the dynamic TCD link succeed and the current
1236 TCD block has been loaded into TCD registers.
1237 */
1238#if defined FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET && FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET
1239 if (tcdRegs->DLAST_SGA ==
1240 MEMORY_ConvertMemoryMapAddress((uint32_t)&handle->tcdPool[nextTcd], kMEMORY_Local2DMA))
1241#else
1242 if (tcdRegs->DLAST_SGA == (uint32_t)&handle->tcdPool[nextTcd])
1243#endif /* FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET */
1244 {
1245 return kStatus_Success;
1246 }
1247 /*
1248 If go to this, means the previous transfer finished, and the DONE bit is set.
1249 So shall configure TCD registers.
1250 */
1251 }
1252 else if (tcdRegs->DLAST_SGA != 0UL)
1253 {
1254 /* The current TCD block has been linked successfully. */
1255 return kStatus_Success;
1256 }
1257 else
1258 {
1259 /*
1260 DLAST_SGA is 0 and it means the first submit transfer, so shall configure
1261 TCD registers.
1262 */
1263 }
1264 }
1265 /* There is no live chain, TCD block need to be installed in TCD registers. */
1266 EDMA_InstallTCD(handle->base, handle->channel, &handle->tcdPool[currentTcd]);
1267 /* Enable channel request again. */
1268 if (0U != (handle->flags & EDMA_TRANSFER_ENABLED_MASK))
1269 {
1270 handle->base->SERQ = DMA_SERQ_SERQ(handle->channel);
1271 }
1272
1273 return kStatus_Success;
1274 }
1275}

1285void EDMA_StartTransfer(edma_handle_t *handle)
1286{
1287 assert(handle != NULL);
1288 uint32_t tmpCSR = 0;
1289
1290 if (handle->tcdPool == NULL)
1291 {
1292 handle->base->SERQ = DMA_SERQ_SERQ(handle->channel);
1293 }
1294 else /* Use the TCD queue. */
1295 {
1296 uint32_t primask;
1297 edma_tcd_t *tcdRegs = (edma_tcd_t *)(uint32_t)&handle->base->TCD[handle->channel];
1298
1299 handle->flags |= EDMA_TRANSFER_ENABLED_MASK;
1300
1301 /* Check if there was at least one descriptor submitted since reset (TCD in registers is valid) */
1302 if (tcdRegs->DLAST_SGA != 0U)
1303 {
1304 primask = DisableGlobalIRQ();
1305 /* Check if channel request is actually disable. */
1306 if ((handle->base->ERQ & ((uint32_t)1U << handle->channel)) == 0U)
1307 {
1308 /* Check if transfer is paused. */
1309 tmpCSR = tcdRegs->CSR;
1310 if ((0U == (tmpCSR & DMA_CSR_DONE_MASK)) || (0U != (tmpCSR & DMA_CSR_ESG_MASK)))
1311 {
1312 /*
1313 Re-enable channel request must be as soon as possible, so must put it into
1314 critical section to avoid task switching or interrupt service routine.
1315 */
1316 handle->base->SERQ = DMA_SERQ_SERQ(handle->channel);
1317 }
1318 }
1319 EnableGlobalIRQ(primask);
1320 }
1321 }

1349{
1350 handle->base->CERQ = DMA_CERQ_CERQ(handle->channel);
1351 /*
1352 Clear CSR to release channel. Because if the given channel started transfer,
1353 CSR will be not zero. Because if it is the last transfer, DREQ will be set.
1354 If not, ESG will be set.

1357 /* Cancel all next TCD transfer. */
1358 handle->base->TCD[handle->channel].DLAST_SGA = 0;
1359 /* clear the CITER and BITER to make sure the TCD register in a correct state for next calling of
1360 * EDMA_SubmitTransfer */

1362 handle->base->TCD[handle->channel].BITER_ELINKNO = 0;
1363
1364 /* Handle the tcd */
1365 if (handle->tcdPool != NULL)
1366 {
1367 handle->header = 1;
1368 handle->tail = 0;
1369 handle->tcdUsed = 0;
1370 }

1401void EDMA_HandleIRQ(edma_handle_t *handle)
1402{
1403 assert(handle != NULL);
1404
1405 bool transfer_done;
1406
1407 /* Clear EDMA interrupt flag */
1408 handle->base->CINT = handle->channel;
1409 /* Check if transfer is already finished. */
1410 transfer_done = ((handle->base->TCD[handle->channel].CSR & DMA_CSR_DONE_MASK) != 0U);
1411
1412 if (handle->tcdPool == NULL)
1413 {
1414 if (handle->callback != NULL)
1415 {
1416 (handle->callback)(handle, handle->userData, transfer_done, 0);
1417 }
1418 }
1419 else /* Use the TCD queue. Please refer to the API descriptions in the eDMA header file for detailed information. */
1420 {
1421 uint32_t sga = (uint32_t)handle->base->TCD[handle->channel].DLAST_SGA;
1422 uint32_t sga_index;
1423 int32_t tcds_done;
1424 uint8_t new_header;
1425 bool esg = ((handle->base->TCD[handle->channel].CSR & DMA_CSR_ESG_MASK) != 0U);
1426
1427 /* Get the offset of the next transfer TCD blocks to be loaded into the eDMA engine. */
1428#if defined FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET && FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET
1429 sga -= MEMORY_ConvertMemoryMapAddress((uint32_t)handle->tcdPool, kMEMORY_Local2DMA);
1430#else
1431 sga -= (uint32_t)handle->tcdPool;
1432#endif /* FSL_FEATURE_MEMORY_HAS_ADDRESS_OFFSET */
1433 /* Get the index of the next transfer TCD blocks to be loaded into the eDMA engine. */
1434 sga_index = sga / sizeof(edma_tcd_t);
1435 /* Adjust header positions. */
1436 if (transfer_done)
1437 {
1438 /* New header shall point to the next TCD to be loaded (current one is already finished) */
1439 new_header = (uint8_t)sga_index;
1440 }
1441 else
1442 {
1443 /* New header shall point to this descriptor currently loaded (not finished yet) */
1444 new_header = sga_index != 0U ? (uint8_t)sga_index - 1U : (uint8_t)handle->tcdSize - 1U;
1445 }
1446 /* Calculate the number of finished TCDs */
1447 if (new_header == (uint8_t)handle->header)
1448 {
1449 int8_t tmpTcdUsed = handle->tcdUsed;
1450 int8_t tmpTcdSize = handle->tcdSize;
1451
1452 /* check esg here for the case that application submit only one request, once the request complete:
1453 * new_header(1) = handle->header(1)
1454 * tcdUsed(1) != tcdSize(>1)
1455 * As the application submit only once, so scatter gather must not enabled, then tcds_done should be 1
1456 */
1457 if ((tmpTcdUsed == tmpTcdSize) || (!esg))
1458 {
1459 tcds_done = handle->tcdUsed;
1460 }
1461 else
1462 {
1463 /* No TCD in the memory are going to be loaded or internal error occurs. */
1464 tcds_done = 0;
1465 }
1466 }
1467 else
1468 {
1469 tcds_done = (int32_t)new_header - (int32_t)handle->header;
1470 if (tcds_done < 0)
1471 {
1472 tcds_done += handle->tcdSize;
1473 }
1474 /*
1475 * While code run to here, it means a TCD transfer Done and a new TCD has loaded to the hardware
1476 * so clear DONE here to allow submit scatter gather transfer request in the callback to avoid TCD
1477 * overwritten.
1478 */
1479 if (transfer_done)
1480 {
1481 handle->base->CDNE = handle->channel;
1482 }
1483 }
1484 /* Advance header which points to the TCD to be loaded into the eDMA engine from memory. */
1485 handle->header = (int8_t)new_header;
1486 /* Release TCD blocks. tcdUsed is the TCD number which can be used/loaded in the memory pool. */
1487 handle->tcdUsed -= (int8_t)tcds_done;
1488 /* Invoke callback function. */
1489 if (NULL != handle->callback)
1490 {
1491 (handle->callback)(handle, handle->userData, transfer_done, tcds_done);
1492 }
1493
1494 /*
1495 * 1.clear the DONE bit here is meaningful for below cases:
1496 * A new TCD has been loaded to EDMA already:
1497 * need to clear the DONE bit in the IRQ handler to avoid TCD in EDMA been overwritten
1498 * if peripheral request isn't coming before next transfer request.
1499 * 2. Don't clear DONE bit for below case,
1500 * for the case that transfer request submitted in the privious edma callback, this is a case that doesn't
1501 * need scatter gather, so keep DONE bit during the next transfer request submission will re-install the TCD and
1502 * the DONE bit will be cleared together with TCD re-installation.
1503 */
1504 if (transfer_done)
1505 {
1506 if ((handle->base->TCD[handle->channel].CSR & DMA_CSR_ESG_MASK) != 0U)
1507 {
1508 handle->base->CDNE = handle->channel;
1509 }
1510 }
1511 }
1512}
