740/*
741** ===================================================================
742** Method : DrawLine (component GDisplay)
743**
744** Description :
745** Draws a line using the Bresenham method
746** Parameters :
747** NAME - DESCRIPTION
748** xstart - x start coordinate
749** ystart - y start coordinate
750** xend - x end coordinate
751** yend - y end coordinate
752** color - color to be used for the line
753** Returns : Nothing
754** ===================================================================
755*/
756void McuGDisplaySSD1306_DrawLine(McuGDisplaySSD1306_PixelDim xstart,
757 McuGDisplaySSD1306_PixelDim ystart,
758 McuGDisplaySSD1306_PixelDim xend, McuGDisplaySSD1306_PixelDim yend,
759 McuGDisplaySSD1306_PixelColor color)
760{
761/* Based on Bresenham algorithm and http://de.wikipedia.org/wiki/Bresenham-Algorithmus */
762#define sgn(x) ((x) > 0) ? 1 : ((x) < 0) ? -1 : 0
763 McuGDisplaySSD1306_PixelDim x, y;
764 int t, dx, dy, incx, incy, pdx, pdy, ddx, ddy, es, el, err;
765
766 /* Calculate distance in both dimensions */
767 dx = xend - xstart;
768 dy = yend - ystart;
769 /* Determin sign of increment */
770 incx = sgn(dx);
771 incy = sgn(dy);
772 if (dx < 0)
773 {
774 dx = -dx;
775 }
776 if (dy < 0)
777 {
778 dy = -dy;
779 }
780 /* Check which distance is larger */
781 if (dx > dy)
782 { /* we are faster in x direction */
783 pdx = incx;
784 pdy = 0; /* pd. is parallel step */
785 ddx = incx;
786 ddy = incy; /* dd. is diagonal step */
787 es = dy;
788 el = dx; /* error steps fast (es) and slow (el) */
789 }
790 else
791 { /* faster in y direction */
792 pdx = 0;
793 pdy = incy; /* pd. is parallel step */
794 ddx = incx;
795 ddy = incy; /* dd. is diagonal step */
796 es = dx;
797 el = dy; /* error step fast (es) and slow (el) */
798 }
799 /* Do some initialization first... */
800 x = xstart;
801 y = ystart;
802 err = el / 2;
803 McuGDisplaySSD1306_PutPixel(x, y, color); /* put first pixel */
804 /* calculate pixels */
805 for (t = 0; t < el; ++t)
806 { /* t counts the pixels, el is the number of pixels */
807 err -= es; /* adapt error */
808 if (err < 0)
809 { /* make error positive again */
810 err += el;
811 /* step into the slow direction, diagonal step */
812 x += ddx;
813 y += ddy;
814 }
815 else
816 {
817 /* step into the fast direction, parallel step*/
818 x += pdx;
819 y += pdy;
820 }
821 McuGDisplaySSD1306_PutPixel(x, y, color);
822 }
823}

826** ===================================================================
827** Method : DrawCircle (component GDisplay)

830** Draws a circle using the Bresenham method

839*/

841 McuGDisplaySSD1306_PixelDim radius,
842 McuGDisplaySSD1306_PixelColor color)
843{
844 /* draw a circle using the Bresenham method, see
845 * http://de.wikipedia.org/wiki/Bresenham-Algorithmus */
846 int f = 1 - radius;
847 int ddF_x = 0;
848 int ddF_y = -2 * radius;
849 McuGDisplaySSD1306_PixelDim x = 0;

857 {
858 if (f >= 0)
859 {
860 y--;
861 ddF_y += 2;
862 f += ddF_y;
863 }
864 x++;
865 ddF_x += 2;
866 f += ddF_x + 1;
867 McuGDisplaySSD1306_PutPixel((McuGDisplaySSD1306_PixelDim)(x0 + x),
868 (McuGDisplaySSD1306_PixelDim)(y0 + y), color);

1153 McuGDisplaySSD1306_PixelDim y2, uint8_t* bmp,
1154 bool compressed)
1155{
1156 /* For uncompressed bitmap data:
1157 The pixels are provided in an array of 8bit pixel information (bmp[]).
1158 Each byte describes a pixel in 3-3-2 RBG format
1159 For compressed bitmap data:
1160 If two bytes are the same, this means that the 3rd byte contains how many times the color
1161 has to be repeated. Example: 10 red dots are compressed as: RED RED 8
1162 */
1163
1164#if McuGDisplaySSD1306_CONFIG_USE_WINDOW_CAPABILITY
1165 McuGDisplaySSD1306_PixelCount pixelCount =
1166 (McuGDisplaySSD1306_PixelCount)((x2 - x1 + 1) * (y2 - y1 + 1));

1168#endif

1174 y2); /* set up window as large as the box */
1175 if (compressed)
1176 {
1177 /* compressed pixel information */
1178 uint8_t pixelColor = 0;
1179 uint8_t lastPixel;
1180 McuGDisplaySSD1306_PixelCount pos = 0;
1181 bool first = TRUE; /* if we read the first time, or if we may have a new sequence */
1182 uint8_t repeat;
1183
1184 while (pixelCount != 0)
1185 { /* while there are pixels to write... */
1186 if (first)
1187 { /* first pixel, or first after a compressed sequence */
1188 pixelColor = *bmp; /* read pixel */
1189 bmp++; /* advance pointer to next pixel */
1190 McuGDisplaySSD1306_CONFIG_FCT_NAME_WRITEPIXEL(
1191 c332to565[pixelColor]); /* write pixel */
1192 pixelCount--; /* one more done */
1193 first = FALSE; /* reset flag */
1194 }
1195 else
1196 {
1197 lastPixel = pixelColor; /* remember already written pixel to detect potential
1198 compressed sequence */
1199 pixelColor = *bmp; /* read pixel */
1200 bmp++; /* advance pointer to next pixel */
1201 McuGDisplaySSD1306_CONFIG_FCT_NAME_WRITEPIXEL(
1202 c332to565[pixelColor]); /* write pixel */
1203 pixelCount--; /* one more done */
1204 if (lastPixel == pixelColor)
1205 { /* do we have twice the same pixel? Then it this starts a compressed sequence...
