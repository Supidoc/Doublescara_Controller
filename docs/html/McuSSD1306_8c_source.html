487/*
488** ===================================================================
489** Method : UpdateRegion (component SSD1306)
490**
491** Description :
492** Updates a region of the display. This is only a stub for
493** this display as we are using windowing.
494** Parameters :
495** NAME - DESCRIPTION
496** x - x coordinate
497** y - y coordinate
498** w - width of the region
499** h - Height of the region
500** Returns : Nothing
501** ===================================================================
502*/
503void McuSSD1306_UpdateRegion(McuSSD1306_PixelDim x, McuSSD1306_PixelDim y, McuSSD1306_PixelDim w,
504 McuSSD1306_PixelDim h)
505{
506 int page, pageBeg, pageEnd, colStart;
507 McuSSD1306_PixelDim d_width = McuSSD1306_GetWidth();
508 McuSSD1306_PixelDim d_height = McuSSD1306_GetHeight();
509
510 /* check boundaries */
511 if (x >= d_width)
512 {
513 x = d_width - 1;
514 }
515 if (y >= d_height)
516 {
517 y = d_height - 1;
518 }
519 if (x + w >= d_width)
520 {
521 w = d_width - x;
522 }
523 if (y + h >= d_height)
524 {
525 h = d_height - y;
526 }
527 pageBeg = (y / 8);
528 pageEnd = (y + h - 1) / 8;
529 colStart = x;
530 for (page = pageBeg; page <= pageEnd; page++)
531 {
532 (void)SSD1306_SetPageStartAddr(page);
533 (void)SSD1306_SetColStartAddr(colStart);
534 SSD1306_WriteDataBlock(&McuSSD1306_DisplayBuf[0][0] +
535 (page * McuSSD1306_DISPLAY_HW_NOF_COLUMNS + colStart),
536 w);
537 }
538}

600/*
601** ===================================================================
602** Method : SetDisplayOrientation (component SSD1306)
603**
604** Description :
605** Sets the display orientation. If you enable this method,
606** then the orientation of the display can be changed at
607** runtime. However, this requires additional resources.
608** Parameters :
609** NAME - DESCRIPTION
610** newOrientation - new orientation to
611** be used
612** Returns : Nothing
613** ===================================================================
614*/
615void McuSSD1306_SetDisplayOrientation(McuSSD1306_DisplayOrientation newOrientation)
616{
617#if McuSSD1306_CONFIG_DYNAMIC_DISPLAY_ORIENTATION
618 currentOrientation = newOrientation;
619#endif
620 switch (newOrientation)
621 {
622 case McuSSD1306_ORIENTATION_LANDSCAPE:
623 SSD1306_WriteCommand(SSD1306_MEMORY_MODE); /* set memory mode */
624 SSD1306_WriteCommand(0x00); /* 00b: horizontal addressing mode mode */
625 SSD1306_WriteCommand(SSD1306_SEG_REMAP | 0x01); /* set segment re-map 128 to 0 */

629 SSD1306_WriteCommand(SSD1306_MEMORY_MODE); /* set memory mode */
630 SSD1306_WriteCommand(0x00); /* 00b: horizontal addressing mode mode */
631 SSD1306_WriteCommand(SSD1306_SEG_REMAP); /* set segment re-map 0 to 128 */
632 SSD1306_WriteCommand(SSD1306_COM_SCAN_INC); /* set COM output scan direction 0 to 64 */
633 break;
634 case McuSSD1306_ORIENTATION_PORTRAIT:
635 SSD1306_WriteCommand(SSD1306_MEMORY_MODE); /* set memory mode */
636 SSD1306_WriteCommand(0x01); /* 01b: vertical addressing mode */
637 SSD1306_WriteCommand(SSD1306_SEG_REMAP | 0x01); /* set segment re-map 128 to 0 */
638 SSD1306_WriteCommand(SSD1306_COM_SCAN_INC); /* set COM output scan direction 0 to 64 */
639 break;
640 case McuSSD1306_ORIENTATION_PORTRAIT180:
641 SSD1306_WriteCommand(SSD1306_MEMORY_MODE); /* set memory mode */
642 SSD1306_WriteCommand(0x01); /* 01b: vertical addressing mode */
643 SSD1306_WriteCommand(SSD1306_SEG_REMAP); /* set segment re-map 0 to 128 */
644 SSD1306_WriteCommand(SSD1306_COM_SCAN_DEC); /* set COM output scan direction 64 to 0 */
645 break;
646 default:
647 break;
648 } /* switch */
649}

1088/*
1089** ===================================================================
1090** Method : PutPixel (component SSD1306)
1091**
1092** Description :
1093** Draws a pixel into the display buffer (not on the display).
1094** Parameters :
1095** NAME - DESCRIPTION
1096** x - x position of the pixel
1097** y - y position of the pixel
1098** color - color value of the pixel
1099** Returns : Nothing
1100** ===================================================================
1101*/
1102void McuSSD1306_PutPixel(McuSSD1306_PixelDim x, McuSSD1306_PixelDim y, McuSSD1306_PixelColor color)
1103{
1104 uint8_t val;
1105
1106#if 0 /* done at the the higher level */
1107 if (x>=McuSSD1306_GetWidth() || y>=McuSSD1306_GetHeight()) {
1108 return; /* error case */
1109 }
1110#endif
1111 val = McuSSD1306_DisplayBuf[y / 8][x]; /* get current value */
1112 if (color != 0)
1113 {
1114 val |= (1 << (y % 8)); /* set pixel */
1115 }
1116 else
1117 {
1118 val &= ~(1 << (y % 8)); /* clear pixel */
1119 }
1120 McuSSD1306_DisplayBuf[y / 8][x] = val; /* store value */
1121}
1122
