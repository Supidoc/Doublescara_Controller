54status_t CLI_init(void)

57

60 {
61 return kStatus_Fail;
62 }
63
64 status = CLI_register_command(&xPingCommandDefinition);

67 return kStatus_Fail;

73status_t CLI_register_command(const CLI_Command_Definition_t* const pxCommandToRegister)
74{
75 BaseType_t status;
76
77 if (xTaskGetSchedulerState() != taskSCHEDULER_NOT_STARTED)
78 {
79 if (xSemaphoreTake(cliMutex, portMAX_DELAY) != pdTRUE)
80 {
81 LOG_ERROR("Failed to acquire CLI mutex for command registration");
82 return kStatus_Fail;
83 }
84 }
85
86 status = FreeRTOS_CLIRegisterCommand(pxCommandToRegister);
87 if (status != pdTRUE)
88 {
89 if (xTaskGetSchedulerState() != taskSCHEDULER_NOT_STARTED)
90 {
91 LOG_ERROR("Failed to register CLI command");
92 xSemaphoreGive(cliMutex);
93 }
94 return kStatus_Fail;
95 }
96
97 if (xTaskGetSchedulerState() != taskSCHEDULER_NOT_STARTED)
98 {
99 char logMsg[60];
100 snprintf(logMsg, sizeof(logMsg), "Registered CLI command: %s",
101 pxCommandToRegister->pcCommand);
102 LOG_DEBUG(logMsg);
103 xSemaphoreGive(cliMutex);
104 }
105
106 return kStatus_Success;
107}

123static void process(void)
124{
125
126 uint8_t cRxedChar;
127 static uint8_t cInputIndex = 0;
128 size_t rxReceivedCount;
129 BaseType_t xMoreDataToFollow;
130
131 /* The input and output buffers are declared static to keep them off the stack. */
132 static uint8_t pcOutputString[CLI_MAX_OUTPUT_LENGTH];
133 static uint8_t pcInputString[CLI_MAX_INPUT_LENGTH];
134
135 /* This implementation reads a single character at a time. Wait in the
136
137 Blocked state until a character is received. */
138
139 LPUART_RTOS_Receive(&LPUART0_rtos_handle, &cRxedChar, 1, &rxReceivedCount);
140 if (rxReceivedCount == 0)
141 return;
142 if (cRxedChar == '\n')
143
144 {
145 if (xSemaphoreTake(cliMutex, portMAX_DELAY) != pdTRUE)
146 {
147 LOG_ERROR("Failed to acquire CLI mutex for command processing");
148 return;
149 }
150 /* A newline character was received, so the input command string is
151 complete and can be processed. Transmit a line separator, just to
152 make the output easier to read. */
153
154 LPUART_RTOS_Send(&LPUART0_rtos_handle, (unsigned char*)"\r\n", strlen("\r\n"));
155
156 if (strncmp((char*)pcInputString, "log", strlen("log")) != 0)
157 {
158 snprintf((char*)pcOutputString, sizeof(pcOutputString), "USER: %s", pcInputString);
159#if CLI_SHOW_COMMAND_INPUT
160 LOG_INFO((char*)pcOutputString);
161#else
162 LOG_INFO_SILENT((char*)pcOutputString);
163#endif
164 memset(pcOutputString, 0x00, CLI_MAX_OUTPUT_LENGTH);
165 }
166 /* The command interpreter is called repeatedly until it returns
167 pdFALSE. See the "Implementing a command" documentation for an
168 exaplanation of why this is. */
169 do
170
171 {
172
173 /* Send the command string to the command interpreter. Any
174 output generated by the command interpreter will be placed in the
175 pcOutputString buffer. */
176
177 xMoreDataToFollow = FreeRTOS_CLIProcessCommand(
178 (char*)pcInputString, /* The command string.*/
179 (char*)pcOutputString, /* The output buffer. */
180 CLI_MAX_OUTPUT_LENGTH /* The size of the output buffer. */
181 );
182
183 /* Write the output generated by the command interpreter to the
184 console. */
185 LPUART_RTOS_Send(&LPUART0_rtos_handle, pcOutputString, strlen((char*)pcOutputString));
186 } while (xMoreDataToFollow != pdFALSE);
187
188 /* All the strings generated by the input command have been sent.
189 Processing of the command is complete. Clear the input string ready
190 to receive the next command. */
191
192 cInputIndex = 0;
193 memset(pcInputString, 0x00, CLI_MAX_INPUT_LENGTH);
194
195 xSemaphoreGive(cliMutex);
196 }
197 else
198 {
199 /* The if() clause performs the processing after a newline character
200 is received. This else clause performs the processing if any other
201 character is received. */
202
203 if (cRxedChar == '\r')
204
205 {
206 /* Ignore carriage returns. */
207 }
208 else if (cRxedChar == '\0')
209
210 {
211 /* Ignore null termination. */
212 }
213 else if (cRxedChar == '\b')
214 {
215 /* Backspace was pressed. Erase the last character in the input
216 buffer - if there are any. */
217 if (cInputIndex > 0)
218 {
219 cInputIndex--;
220 pcInputString[cInputIndex] = '\0';
221 }
222 }
223 else
224 {
225 /* A character was entered. It was not a new line, backspace
226 or carriage return, so it is accepted as part of the input and
227 placed into the input buffer. When a n is entered the complete
228 string will be passed to the command interpreter. */
229 if (cInputIndex < CLI_MAX_INPUT_LENGTH)
230 {
231 pcInputString[cInputIndex] = cRxedChar;
232 cInputIndex++;
233 }
234 }
235 }
236}
