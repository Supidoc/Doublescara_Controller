73uint32_t FTM_GetInstance(FTM_Type *base)
74{
75 uint32_t instance;
76 uint32_t ftmArrayCount = (sizeof(s_ftmBases) / sizeof(s_ftmBases[0]));
77
78 /* Find the instance index from base address mappings. */
79 for (instance = 0; instance < ftmArrayCount; instance++)
80 {
81 if (MSDK_REG_SECURE_ADDR(s_ftmBases[instance]) == MSDK_REG_SECURE_ADDR(base))
82 {
83 break;
84 }
85 }
86
87 assert(instance < ftmArrayCount);
88
89 return instance;
90}

92static void FTM_SetPwmSync(FTM_Type *base, uint32_t syncMethod, bool swRstCnt, bool hwRstCnt)
93{
94 uint8_t chnlNumber = 0;
95 uint32_t reg = 0, syncReg = 0;
96
97 /* The CHANNEL_COUNT macro returns -1 if it cannot match the FTM instance */
98 assert(-1 != FSL_FEATURE_FTM_CHANNEL_COUNTn(base));
99
100 syncReg = base->SYNC;
101 /* Enable PWM synchronization of output mask register */
102 syncReg |= FTM_SYNC_SYNCHOM_MASK;
103
104 reg = base->COMBINE;
105 for (chnlNumber = 0; chnlNumber < ((uint8_t)FSL_FEATURE_FTM_CHANNEL_COUNTn(base) / 2U); chnlNumber++)
106 {
107 /* Enable PWM synchronization of registers C(n)V and C(n+1)V */
108 reg |= (1UL << (FTM_COMBINE_SYNCEN0_SHIFT + (FTM_COMBINE_COMBINE1_SHIFT * chnlNumber)));
109 }
110 base->COMBINE = reg;
111
112 reg = base->SYNCONF;
113
114 /* Use enhanced PWM synchronization method. Use PWM sync to update register values */
115 reg |= (FTM_SYNCONF_SYNCMODE_MASK | FTM_SYNCONF_CNTINC_MASK | FTM_SYNCONF_INVC_MASK | FTM_SYNCONF_SWOC_MASK);
116
117 if ((syncMethod & FTM_SYNC_SWSYNC_MASK) != 0U)
118 {
119 /* Enable needed bits for software trigger to update registers with its buffer value */
120 reg |= (FTM_SYNCONF_SWWRBUF_MASK | FTM_SYNCONF_SWINVC_MASK | FTM_SYNCONF_SWSOC_MASK | FTM_SYNCONF_SWOM_MASK);
121 /* Enable software trigger synchronization count. */
122 if (swRstCnt)
123 {
124 reg |= FTM_SYNCONF_SWRSTCNT_MASK;
125 }
126 }
127
128 if ((syncMethod & (FTM_SYNC_TRIG0_MASK | FTM_SYNC_TRIG1_MASK | FTM_SYNC_TRIG2_MASK)) != 0U)
129 {
130 /* Enable needed bits for hardware trigger to update registers with its buffer value */
131 reg |= (FTM_SYNCONF_HWWRBUF_MASK | FTM_SYNCONF_HWINVC_MASK | FTM_SYNCONF_HWSOC_MASK | FTM_SYNCONF_HWOM_MASK);
132 /* Enable hardware trigger synchronization count */
133 if (hwRstCnt)
134 {
135 reg |= FTM_SYNCONF_HWRSTCNT_MASK;
136 }
137
138 /* Enable the appropriate hardware trigger that is used for PWM sync */
139 if ((syncMethod & FTM_SYNC_TRIG0_MASK) != 0U)
140 {
141 syncReg |= FTM_SYNC_TRIG0_MASK;
142 }
143 if ((syncMethod & FTM_SYNC_TRIG1_MASK) != 0U)
144 {
145 syncReg |= FTM_SYNC_TRIG1_MASK;
146 }
147 if ((syncMethod & FTM_SYNC_TRIG2_MASK) != 0U)
148 {
149 syncReg |= FTM_SYNC_TRIG2_MASK;
150 }
151 }
152
153 /* Write back values to the SYNC register */
154 base->SYNC = syncReg;
155
156 /* Write the PWM synch values to the SYNCONF register */
157 base->SYNCONF = reg;
158}

160static void FTM_SetReloadPoints(FTM_Type *base, uint32_t reloadPoints)
161{
162 uint32_t chnlNumber = 0;
163 uint32_t reg = 0;
164 int8_t chnlCount = FSL_FEATURE_FTM_CHANNEL_COUNTn(base);
165
166 /* The CHANNEL_COUNT macro returns -1 if it cannot match the FTM instance */
167 assert(-1 != chnlCount);
168
169 /* Need CNTINC bit to be 1 for CNTIN register to update with its buffer value on reload */
170 base->SYNCONF |= FTM_SYNCONF_CNTINC_MASK;
171
172 reg = base->COMBINE;
173 for (chnlNumber = 0; chnlNumber < ((uint32_t)chnlCount / 2U); chnlNumber++)
174 {
175 /* Need SYNCEN bit to be 1 for CnV reg to update with its buffer value on reload */
176 reg |= (1UL << (FTM_COMBINE_SYNCEN0_SHIFT + (FTM_COMBINE_COMBINE1_SHIFT * chnlNumber)));
177 }
178 base->COMBINE = reg;
179
180 /* Set the reload points */
181 reg = base->PWMLOAD;
182
183 /* Enable the selected channel match reload points */
184 reg &= ~((1UL << (uint32_t)chnlCount) - 1U);
185 reg |= (reloadPoints & ((1UL << (uint32_t)chnlCount) - 1U));
186
187#if defined(FSL_FEATURE_FTM_HAS_HALFCYCLE_RELOAD) && (FSL_FEATURE_FTM_HAS_HALFCYCLE_RELOAD)
188 /* Enable half cycle match as a reload point */
189 if ((reloadPoints & (uint32_t)kFTM_HalfCycMatch) != 0U)
190 {
191 reg |= FTM_PWMLOAD_HCSEL_MASK;
192 }
193 else
194 {
195 reg &= ~FTM_PWMLOAD_HCSEL_MASK;
196 }
197#endif /* FSL_FEATURE_FTM_HAS_HALFCYCLE_RELOAD */
198
199 base->PWMLOAD = reg;
200
201 /* These reload points are used when counter is in up-down counting mode */
202 reg = base->SYNC;
203 if ((reloadPoints & (uint32_t)kFTM_CntMax) != 0U)
204 {
205 /* Reload when counter turns from up to down */
206 reg |= FTM_SYNC_CNTMAX_MASK;
207 }
208 else
209 {
210 reg &= ~FTM_SYNC_CNTMAX_MASK;
211 }
212
213 if ((reloadPoints & (uint32_t)kFTM_CntMin) != 0U)
214 {
215 /* Reload when counter turns from down to up */
216 reg |= FTM_SYNC_CNTMIN_MASK;
217 }
218 else
219 {
220 reg &= ~FTM_SYNC_CNTMIN_MASK;
221 }
222 base->SYNC = reg;
223}

236status_t FTM_Init(FTM_Type *base, const ftm_config_t *config)
237{
238 assert(config);
239#if defined(FSL_FEATURE_FTM_IS_TPM_ONLY_INSTANCE)
240 /* This function does not support the current instance, please use the TPM driver. */
241 assert((FSL_FEATURE_FTM_IS_TPM_ONLY_INSTANCE(base) == 0U));
242#endif /* FSL_FEATURE_FTM_IS_TPM_ONLY_INSTANCE */
243
244 uint32_t reg;
245
246 if ((config->pwmSyncMode & (uint32_t)((uint32_t)FTM_SYNC_TRIG0_MASK | (uint32_t)FTM_SYNC_TRIG1_MASK |
247 (uint32_t)FTM_SYNC_TRIG2_MASK | (uint32_t)FTM_SYNC_SWSYNC_MASK)) == 0U)
248 {
249 /* Invalid PWM sync mode */
250 return kStatus_Fail;
251 }
252
253#if !(defined(FSL_SDK_DISABLE_DRIVER_CLOCK_CONTROL) && FSL_SDK_DISABLE_DRIVER_CLOCK_CONTROL)
254 /* Ungate the FTM clock*/
255 (void)CLOCK_EnableClock(s_ftmClocks[FTM_GetInstance(base)]);
256#endif /* FSL_SDK_DISABLE_DRIVER_CLOCK_CONTROL */
257
258#if defined(FTM_RESETS_ARRAY)
259 RESET_ReleasePeripheralReset(s_ftmResets[FTM_GetInstance(base)]);
260#endif
261
262#if (defined(FSL_FEATURE_FTM_HAS_BASIC_FEATURE_ONLY_INSTANCE) && FSL_FEATURE_FTM_HAS_BASIC_FEATURE_ONLY_INSTANCE)
263 if (0 != FSL_FEATURE_FTM_IS_BASIC_FEATURE_ONLY_INSTANCEn(base))
264 {
265 /* Enable FTM mode and disable write protection */
266 base->MODE = FTM_MODE_FTMEN_MASK | FTM_MODE_WPDIS_MASK;
267 }
268 else
269 {
270 /* Configure the fault mode, enable FTM mode and disable write protection */
271 base->MODE = FTM_MODE_FAULTM(config->faultMode) | FTM_MODE_FTMEN_MASK | FTM_MODE_WPDIS_MASK;
272 }
273#else
274 /* Configure the fault mode, enable FTM mode and disable write protection */
275 base->MODE = FTM_MODE_FAULTM(config->faultMode) | FTM_MODE_FTMEN_MASK | FTM_MODE_WPDIS_MASK;
276#endif
277
278 /* Configure the update mechanism for buffered registers */
279 FTM_SetPwmSync(base, config->pwmSyncMode, config->swTriggerResetCount, config->hwTriggerResetCount);
280
281 /* Setup intermediate register reload points */
282 FTM_SetReloadPoints(base, config->reloadPoints);
283
284 /* Set the clock prescale factor */
285 base->SC = FTM_SC_PS(config->prescale);
286
287#if (defined(FSL_FEATURE_FTM_HAS_FILTER_PRESCALER) && FSL_FEATURE_FTM_HAS_FILTER_PRESCALER)
288 /* Set filter clock prescale factor */
289 base->SC |= FTM_SC_FLTPS(config->filterPrescale);
290#endif /* FSL_FEATURE_FTM_HAS_FILTER_PRESCALER */
291
292 /* Setup the counter operation */
293 base->CONF = (FTM_CONF_BDMMODE(config->bdmMode) | FTM_CONF_GTBEEN(config->useGlobalTimeBase));
294
295 /* Initial state of channel output */
296 base->OUTINIT = config->chnlInitState;
297
298 /* Channel polarity */
299 base->POL = config->chnlPolarity;
300
301 /* Set the external trigger sources */
302 base->EXTTRIG = config->extTriggers;
303#if defined(FSL_FEATURE_FTM_HAS_RELOAD_INITIALIZATION_TRIGGER) && (FSL_FEATURE_FTM_HAS_RELOAD_INITIALIZATION_TRIGGER)
304 if ((config->extTriggers & (uint32_t)kFTM_ReloadInitTrigger) != 0U)
305 {
306 base->CONF |= FTM_CONF_ITRIGR_MASK;
307 }
308 else
309 {
310 base->CONF &= ~FTM_CONF_ITRIGR_MASK;
311 }
312#endif /* FSL_FEATURE_FTM_HAS_RELOAD_INITIALIZATION_TRIGGER */
313
314 /* FTM deadtime insertion control */
315 base->DEADTIME = (0u |
316#if defined(FSL_FEATURE_FTM_HAS_EXTENDED_DEADTIME_VALUE) && (FSL_FEATURE_FTM_HAS_EXTENDED_DEADTIME_VALUE)
317 /* Has extended deadtime value register) */
318 FTM_DEADTIME_DTVALEX(config->deadTimeValue >> 6) |
319#endif /* FSL_FEATURE_FTM_HAS_EXTENDED_DEADTIME_VALUE */
320 FTM_DEADTIME_DTPS(config->deadTimePrescale) | FTM_DEADTIME_DTVAL(config->deadTimeValue));
321
322#if (defined(FSL_FEATURE_FTM_HAS_BASIC_FEATURE_ONLY_INSTANCE) && FSL_FEATURE_FTM_HAS_BASIC_FEATURE_ONLY_INSTANCE)
323 if (0 == FSL_FEATURE_FTM_IS_BASIC_FEATURE_ONLY_INSTANCEn(base))
324 {
325 /* FTM fault filter value */
326 reg = base->FLTCTRL;
327 reg &= ~FTM_FLTCTRL_FFVAL_MASK;
328 reg |= FTM_FLTCTRL_FFVAL(config->faultFilterValue);
329#if (defined(FSL_FEATURE_FTM_HAS_FAULT_OUTPUT_STATE) && FSL_FEATURE_FTM_HAS_FAULT_OUTPUT_STATE)
330 reg &= ~FTM_FLTCTRL_FSTATE_MASK;
331 reg |= FTM_FLTCTRL_FSTATE(config->faultOutputState);
332#endif /* FSL_FEATURE_FTM_HAS_FAULT_OUTPUT_STATE */
333 base->FLTCTRL = reg;
334 }
335#else
336 /* FTM fault filter value */
337 reg = base->FLTCTRL;
338 reg &= ~FTM_FLTCTRL_FFVAL_MASK;
339 reg |= FTM_FLTCTRL_FFVAL(config->faultFilterValue);
340#if (defined(FSL_FEATURE_FTM_HAS_FAULT_OUTPUT_STATE) && FSL_FEATURE_FTM_HAS_FAULT_OUTPUT_STATE)
341 /* Set fault output state */
342 reg &= ~FTM_FLTCTRL_FSTATE_MASK;
343 reg |= FTM_FLTCTRL_FSTATE(config->faultOutputState);
344#endif /* FSL_FEATURE_FTM_HAS_FAULT_OUTPUT_STATE */
345 base->FLTCTRL = reg;
346#endif
347
348 return kStatus_Success;
349}

389void FTM_GetDefaultConfig(ftm_config_t *config)
390{
391 assert(config != NULL);
392
393 /* Initializes the configure structure to zero. */
394 (void)memset(config, 0, sizeof(*config));
395
396 /* Divide FTM clock by 1 */
397 config->prescale = kFTM_Prescale_Divide_1;
398#if (defined(FSL_FEATURE_FTM_HAS_FILTER_PRESCALER) && FSL_FEATURE_FTM_HAS_FILTER_PRESCALER)
399 /* Divide FTM clock by 1 */
400 config->filterPrescale = kFTM_Filter_Prescale_Divide_1;
401#endif /* FSL_FEATURE_FTM_HAS_FILTER_PRESCALER */
402 /* FTM behavior in BDM mode */
403 config->bdmMode = kFTM_BdmMode_0;
404 /* Software trigger will be used to update registers */
405 config->pwmSyncMode = (uint32_t)kFTM_SoftwareTrigger;
406 /* No intermediate register load */
407 config->reloadPoints = 0;
408 /* Fault control disabled for all channels */
409 config->faultMode = kFTM_Fault_Disable;
410 /* Disable the fault filter */
411 config->faultFilterValue = 0;
412#if (defined(FSL_FEATURE_FTM_HAS_FAULT_OUTPUT_STATE) && FSL_FEATURE_FTM_HAS_FAULT_OUTPUT_STATE)
413 /* Configure fault output state. */
414 config->faultOutputState = kFTM_FaultOutput_PreDefined;
415#endif /* FSL_FEATURE_FTM_HAS_FAULT_OUTPUT_STATE */
416 /* Divide the system clock by 1 */
417 config->deadTimePrescale = kFTM_Deadtime_Prescale_1;
418 /* No counts are inserted */
419 config->deadTimeValue = 0;
420 /* No external trigger */
421 config->extTriggers = 0;
422 /* Initialization value is 0 for all channels */
423 config->chnlInitState = 0;
424 /* Active high polarity for all channels */
425 config->chnlPolarity = 0;
426 /* Use internal FTM counter as timebase */
427 config->useGlobalTimeBase = false;
428 /* Set hardware trigger activation counter sync to false */
429 config->hwTriggerResetCount = false;
430 /* Set software trigger activation counter sync to true */
431 config->swTriggerResetCount = true;
432}

450status_t FTM_SetupPwm(FTM_Type *base,
451 const ftm_chnl_pwm_signal_param_t *chnlParams,
452 uint8_t numOfChnls,
453 ftm_pwm_mode_t mode,
454 uint32_t pwmFreq_Hz,
455 uint32_t srcClock_Hz)
456{
457 assert(NULL != chnlParams);
458
459 uint32_t mod, reg;
460 uint32_t ftmClock = (srcClock_Hz / (1UL << (base->SC & FTM_SC_PS_MASK)));
461 uint32_t cnv, cnvFirstEdge;
462 uint8_t i;
463
464 if ((0U == pwmFreq_Hz) || (0U == srcClock_Hz) || (0U == numOfChnls))
465 {
466 return kStatus_InvalidArgument;
467 }
468
469 if (mode == kFTM_CenterAlignedPwm)
470 {
471 base->SC |= FTM_SC_CPWMS_MASK;
472 mod = ftmClock / (pwmFreq_Hz * 2U);
473 }
474 else
475 {
476 base->SC &= ~FTM_SC_CPWMS_MASK;
477 mod = (ftmClock / pwmFreq_Hz) - 1U;
478 }
479
480 /* Return an error in case we overflow the registers, probably would require changing
481 * clock source to get the desired frequency */
482 if (mod > 65535U)
483 {
484 return kStatus_OutOfRange;
485 }
486 /* Set the PWM period */
487 base->CNTIN = 0U;
488 base->MOD = mod;
489
490 /* Setup each FTM channel */
491 for (i = 0; i < numOfChnls; i++)
492 {
493 /* Return error if requested chnlNumber is greater than the max allowed */
494 if (((uint8_t)chnlParams->chnlNumber >= (uint8_t)FSL_FEATURE_FTM_CHANNEL_COUNTn(base)) ||
495 (-1 == (int8_t)FSL_FEATURE_FTM_CHANNEL_COUNTn(base)))
496 {
497 return kStatus_InvalidArgument;
498 }
499 /* Return error if requested dutycycle is greater than the max allowed */
500 if (chnlParams->dutyCyclePercent > 100U)
501 {
502 return kStatus_OutOfRange;
503 }
504
505 if (chnlParams->dutyCyclePercent == 100U)
506 {
507 /* For 100% duty cycle */
508 cnv = mod + 1U;
509 }
510 else
511 {
512 cnv = (mod * chnlParams->dutyCyclePercent) / 100U;
513 }
514
515 if ((mode == kFTM_EdgeAlignedPwm) || (mode == kFTM_CenterAlignedPwm))
516 {
517 /* Clear the current mode and edge level bits */
518 reg = base->CONTROLS[chnlParams->chnlNumber].CnSC;
519 reg &= ~(FTM_CnSC_MSA_MASK | FTM_CnSC_MSB_MASK | FTM_CnSC_ELSA_MASK | FTM_CnSC_ELSB_MASK);
520
521 /* Setup the active level */
522 reg |= (uint32_t)chnlParams->level << FTM_CnSC_ELSA_SHIFT;
523
524 /* Edge-aligned mode needs MSB to be 1, don't care for Center-aligned mode */
525 reg |= FTM_CnSC_MSB(1U);
526
527 /* Update the mode and edge level */
528 base->CONTROLS[chnlParams->chnlNumber].CnSC = reg;
529
530 base->CONTROLS[chnlParams->chnlNumber].CnV = cnv;
531#if defined(FSL_FEATURE_FTM_HAS_ENABLE_PWM_OUTPUT) && (FSL_FEATURE_FTM_HAS_ENABLE_PWM_OUTPUT)
532 /* Set to output mode */
533 FTM_SetPwmOutputEnable(base, chnlParams->chnlNumber, true);
534#endif
535 }
536 else
537 {
538 /* This check is added for combined mode as the channel number should be the pair number */
539 if (((uint32_t)chnlParams->chnlNumber) >= ((uint32_t)FSL_FEATURE_FTM_CHANNEL_COUNTn(base) / 2U))
540 {
541 return kStatus_InvalidArgument;
542 }
543
544 if (mode == kFTM_EdgeAlignedCombinedPwm)
545 {
546 cnvFirstEdge = 0;
547 }
548 else if (mode == kFTM_CenterAlignedCombinedPwm)
549 {
550 cnvFirstEdge = (mod > cnv) ? ((mod - cnv) / 2U) : 0U;
551 }
552 else
553 {
554 /* Return error if requested value is greater than the max allowed */
555 if ((chnlParams->firstEdgeDelayPercent + chnlParams->dutyCyclePercent) > 100U)
556 {
557 return kStatus_OutOfRange;
558 }
559 /* Configure delay of the first edge */
560 cnvFirstEdge = (mod * chnlParams->firstEdgeDelayPercent) / 100U;
561 }
562
563 /* Clear the current mode and edge level bits for channel n */
564 reg = base->CONTROLS[((uint32_t)chnlParams->chnlNumber) * 2U].CnSC;
565 reg &= ~(FTM_CnSC_MSA_MASK | FTM_CnSC_MSB_MASK | FTM_CnSC_ELSA_MASK | FTM_CnSC_ELSB_MASK);
566
567 /* Setup the active level for channel n */
568 reg |= (uint32_t)chnlParams->level << FTM_CnSC_ELSA_SHIFT;
569
570 /* Update the mode and edge level for channel n */
571 base->CONTROLS[((uint32_t)chnlParams->chnlNumber) * 2U].CnSC = reg;
572
573 /* Clear the current mode and edge level bits for channel n + 1 */
574 reg = base->CONTROLS[(((uint32_t)chnlParams->chnlNumber) * 2U) + 1U].CnSC;
575 reg &= ~(FTM_CnSC_MSA_MASK | FTM_CnSC_MSB_MASK | FTM_CnSC_ELSA_MASK | FTM_CnSC_ELSB_MASK);
576
577 /* Setup the active level for channel n + 1 */
578 reg |= (uint32_t)chnlParams->level << FTM_CnSC_ELSA_SHIFT;
579
580 /* Update the mode and edge level for channel n + 1*/
581 base->CONTROLS[(((uint32_t)chnlParams->chnlNumber) * 2U) + 1U].CnSC = reg;
582
583 /* Set the combine bit for the channel pair */
584 base->COMBINE |=
585 (1UL << (FTM_COMBINE_COMBINE0_SHIFT + (FTM_COMBINE_COMBINE1_SHIFT * (uint32_t)chnlParams->chnlNumber)));
586
587 /* Set the channel pair values */
588 base->CONTROLS[((uint32_t)chnlParams->chnlNumber) * 2U].CnV = cnvFirstEdge;
589 base->CONTROLS[(((uint32_t)chnlParams->chnlNumber) * 2U) + 1U].CnV = cnvFirstEdge + cnv;
590
591#if defined(FSL_FEATURE_FTM_HAS_ENABLE_PWM_OUTPUT) && (FSL_FEATURE_FTM_HAS_ENABLE_PWM_OUTPUT)
592 /* Set to output mode */
593 FTM_SetPwmOutputEnable(base, (ftm_chnl_t)(uint8_t)((uint8_t)chnlParams->chnlNumber * 2U), true);
594 FTM_SetPwmOutputEnable(base, (ftm_chnl_t)(uint8_t)((uint8_t)chnlParams->chnlNumber * 2U + 1U), true);
595#endif /* FSL_FEATURE_FTM_HAS_ENABLE_PWM_OUTPUT */
596
597 /* Enable/Disable complementary output on the channel pair */
598 FTM_SetComplementaryEnable(base, chnlParams->chnlNumber, chnlParams->enableComplementary);
599 /* Enable/Disable Deadtime insertion on the channel pair */
600 FTM_SetDeadTimeEnable(base, chnlParams->chnlNumber, chnlParams->enableDeadtime);
601 }
602 chnlParams++;
603 }
604
605 return kStatus_Success;
606}

621status_t FTM_UpdatePwmDutycycle(FTM_Type *base,
622 ftm_chnl_t chnlNumber,
623 ftm_pwm_mode_t currentPwmMode,
624 uint8_t dutyCyclePercent)
625{
626 uint32_t cnv, cnvFirstEdge = 0, mod;
627
628 /* Return error if requested chnlNumber is greater than the max allowed */
629 if (((uint8_t)chnlNumber >= (uint8_t)FSL_FEATURE_FTM_CHANNEL_COUNTn(base)) ||
630 (-1 == (int8_t)FSL_FEATURE_FTM_CHANNEL_COUNTn(base)))
631 {
632 return kStatus_InvalidArgument;
633 }
634
635 mod = base->MOD;
636
637 if (dutyCyclePercent > 100U)
638 {
639 return kStatus_OutOfRange;
640 }
641 if (dutyCyclePercent == 100U)
642 {
643 /* For 100% duty cycle */
644 cnv = mod + 1U;
645 }
646 else
647 {
648 cnv = (mod * dutyCyclePercent) / 100U;
649 }
650
651 if ((currentPwmMode == kFTM_EdgeAlignedPwm) || (currentPwmMode == kFTM_CenterAlignedPwm))
652 {
653 base->CONTROLS[chnlNumber].CnV = cnv;
654 }
655 else
656 {
657 /* This check is added for combined mode as the channel number should be the pair number */
658 if ((uint32_t)chnlNumber >= ((uint32_t)FSL_FEATURE_FTM_CHANNEL_COUNTn(base) / 2U))
659 {
660 return kStatus_InvalidArgument;
661 }
662
663 if (currentPwmMode == kFTM_CenterAlignedCombinedPwm)
664 {
665 cnvFirstEdge = (mod > cnv) ? ((mod - cnv) / 2U) : 0U;
666 }
667 else
668 {
669 cnvFirstEdge = base->CONTROLS[((uint32_t)chnlNumber) * 2U].CnV;
670 if (((cnvFirstEdge != 0U) && (cnv == (mod + 1U))) || (((cnvFirstEdge + cnv) > mod) && (cnv < mod)))
671 {
672 /* Return error if firstEdgeDelayPercent + dutyCyclePercent > 100 */
673 return kStatus_OutOfRange;
674 }
675 }

717status_t FTM_SetupPwmMode(FTM_Type *base,
718 const ftm_chnl_pwm_config_param_t *chnlParams,
719 uint8_t numOfChnls,
720 ftm_pwm_mode_t mode)
721{
722 assert(chnlParams != NULL);
723 assert(numOfChnls != 0U);
724 /* The CHANNEL_COUNT macro returns -1 if it cannot match the FTM instance */
725 assert(-1 != FSL_FEATURE_FTM_CHANNEL_COUNTn(base));
726
727 uint32_t reg;
728 uint32_t mod, cnvFirstEdge;
729 uint8_t i;
730
731 switch (mode)
732 {
733 case kFTM_EdgeAlignedPwm:
734 case kFTM_EdgeAlignedCombinedPwm:
735 case kFTM_CenterAlignedCombinedPwm:
736 case kFTM_AsymmetricalCombinedPwm:
737 base->SC &= ~FTM_SC_CPWMS_MASK;
738 break;
739 case kFTM_CenterAlignedPwm:
740 base->SC |= FTM_SC_CPWMS_MASK;
741 break;
742 default:
743 assert(false);
744 break;
745 }
746
747 /* Get percent PWM period */
748 mod = base->MOD;
749 /* Setup each FTM channel */
750 for (i = 0; i < numOfChnls; i++)
751 {
752 if ((mode == kFTM_EdgeAlignedPwm) || (mode == kFTM_CenterAlignedPwm))
753 {
754 /* Clear the current mode and edge level bits */
755 reg = base->CONTROLS[chnlParams->chnlNumber].CnSC;
756 reg &= ~(FTM_CnSC_MSA_MASK | FTM_CnSC_MSB_MASK | FTM_CnSC_ELSA_MASK | FTM_CnSC_ELSB_MASK);
757
758 /* Setup the active level */
759 reg |= (uint32_t)chnlParams->level << FTM_CnSC_ELSA_SHIFT;
760
761 /* Edge-aligned mode needs MSB to be 1, don't care for Center-aligned mode */
762 reg |= FTM_CnSC_MSB(1U);
763
764 /* Update the mode and edge level */
765 base->CONTROLS[chnlParams->chnlNumber].CnSC = reg;
766
767 base->CONTROLS[chnlParams->chnlNumber].CnV = chnlParams->dutyValue;
768#if defined(FSL_FEATURE_FTM_HAS_ENABLE_PWM_OUTPUT) && (FSL_FEATURE_FTM_HAS_ENABLE_PWM_OUTPUT)
769 /* Set to output mode */
770 FTM_SetPwmOutputEnable(base, chnlParams->chnlNumber, true);
771#endif
772 }
773 else
774 {
775 /* This check is added for combined mode as the channel number should be the pair number */
776 if (((uint32_t)chnlParams->chnlNumber) >= (((uint32_t)FSL_FEATURE_FTM_CHANNEL_COUNTn(base)) / 2U))
777 {
778 return kStatus_Fail;
779 }
780
781 if (mode == kFTM_EdgeAlignedCombinedPwm)
782 {
783 cnvFirstEdge = 0;
784 }
785 else if (mode == kFTM_CenterAlignedCombinedPwm)
786 {
787 cnvFirstEdge = (mod - chnlParams->dutyValue) / 2U;
788 }
789 else
790 {
791 /* Return error if requested value is greater than the mod */
792 if (chnlParams->firstEdgeValue > mod)
793 {
794 return kStatus_Fail;
795 }
796 cnvFirstEdge = chnlParams->firstEdgeValue;
797 }
798
799 /* Re-configure first edge when 0% duty cycle */
800 if (chnlParams->dutyValue == 0U)
801 {
802 /* Signal stays low */
803 cnvFirstEdge = 0;
804 }
805
806 /* Clear the current mode and edge level bits for channel n */
807 reg = base->CONTROLS[((uint32_t)chnlParams->chnlNumber) * 2U].CnSC;
808 reg &= ~(FTM_CnSC_MSA_MASK | FTM_CnSC_MSB_MASK | FTM_CnSC_ELSA_MASK | FTM_CnSC_ELSB_MASK);
809
810 /* Setup the active level for channel n */
811 reg |= (uint32_t)chnlParams->level << FTM_CnSC_ELSA_SHIFT;
812
813 /* Update the mode and edge level for channel n */
814 base->CONTROLS[((uint32_t)chnlParams->chnlNumber) * 2U].CnSC = reg;
815
816 /* Clear the current mode and edge level bits for channel n + 1 */
817 reg = base->CONTROLS[(((uint32_t)chnlParams->chnlNumber) * 2U) + 1U].CnSC;
818 reg &= ~(FTM_CnSC_MSA_MASK | FTM_CnSC_MSB_MASK | FTM_CnSC_ELSA_MASK | FTM_CnSC_ELSB_MASK);
819
820 /* Setup the active level for channel n + 1 */
821 reg |= (uint32_t)chnlParams->level << FTM_CnSC_ELSA_SHIFT;
822
823 /* Update the mode and edge level for channel n + 1*/
824 base->CONTROLS[(((uint32_t)chnlParams->chnlNumber) * 2U) + 1U].CnSC = reg;
825
826 /* Set the combine bit for the channel pair */
827 base->COMBINE |=
828 (1UL << (FTM_COMBINE_COMBINE0_SHIFT + (FTM_COMBINE_COMBINE1_SHIFT * (uint32_t)chnlParams->chnlNumber)));
829
830 /* Set the channel pair values */
831 base->CONTROLS[((uint32_t)chnlParams->chnlNumber) * 2U].CnV = cnvFirstEdge;
832 base->CONTROLS[(((uint32_t)chnlParams->chnlNumber) * 2U) + 1U].CnV = cnvFirstEdge + chnlParams->dutyValue;
833
834#if defined(FSL_FEATURE_FTM_HAS_ENABLE_PWM_OUTPUT) && (FSL_FEATURE_FTM_HAS_ENABLE_PWM_OUTPUT)
835 /* Set to output mode */
836 FTM_SetPwmOutputEnable(base, (ftm_chnl_t)(uint8_t)((uint8_t)chnlParams->chnlNumber * 2U), true);
837 FTM_SetPwmOutputEnable(base, (ftm_chnl_t)(uint8_t)((uint8_t)chnlParams->chnlNumber * 2U + 1U), true);
838#endif /* FSL_FEATURE_FTM_HAS_ENABLE_PWM_OUTPUT */
839
840 /* Enable/Disable complementary output on the channel pair */
841 FTM_SetComplementaryEnable(base, chnlParams->chnlNumber, chnlParams->enableComplementary);
842 /* Enable/Disable Deadtime insertion on the channel pair */
843 FTM_SetDeadTimeEnable(base, chnlParams->chnlNumber, chnlParams->enableDeadtime);
844 }
845 chnlParams++;
846 }
847
848 return kStatus_Success;
849}
