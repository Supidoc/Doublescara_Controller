18uint32_t InstallIRQHandler(IRQn_Type irq, uint32_t irqHandler)
19{
20#ifdef __VECTOR_TABLE
21#undef __VECTOR_TABLE
22#endif
23
24 if (((int32_t)irq + 16) < 0)
25 {
26 return MSDK_INVALID_IRQ_HANDLER;
27 }
28
29/* Addresses for VECTOR_TABLE and VECTOR_RAM come from the linker file */
30#if defined(__CC_ARM) || defined(__ARMCC_VERSION)
31 extern uint32_t Image$$VECTOR_ROM$$Base[];
32 extern uint32_t Image$$VECTOR_RAM$$Base[];
33 extern uint32_t Image$$VECTOR_RAM$$ZI$$Limit[];
34
35#define __VECTOR_TABLE Image$$VECTOR_ROM$$Base
36#define __VECTOR_RAM Image$$VECTOR_RAM$$Base
37#define __RAM_VECTOR_TABLE_SIZE (((uint32_t)Image$$VECTOR_RAM$$ZI$$Limit - (uint32_t)Image$$VECTOR_RAM$$Base))
38#elif defined(__ICCARM__)
39 extern uint32_t __RAM_VECTOR_TABLE_SIZE[];
40 extern uint32_t __VECTOR_TABLE[];
41 extern uint32_t __VECTOR_RAM[];
42#elif defined(__GNUC__)
43 extern uint32_t __VECTOR_TABLE[];
44 extern uint32_t __VECTOR_RAM[];
45 extern uint32_t __RAM_VECTOR_TABLE_SIZE_BYTES[];
46 uint32_t __RAM_VECTOR_TABLE_SIZE = (uint32_t)(__RAM_VECTOR_TABLE_SIZE_BYTES);
47#endif /* defined(__CC_ARM) || defined(__ARMCC_VERSION) */
48 uint32_t n;
49 uint32_t ret;
50 uint32_t irqMaskValue;
51
52 irqMaskValue = DisableGlobalIRQ();
53 if (SCB->VTOR != (uint32_t)__VECTOR_RAM)
54 {
55 /* Copy the vector table from ROM to RAM */
56 for (n = 0; n < ((uint32_t)__RAM_VECTOR_TABLE_SIZE) / sizeof(uint32_t); n++)
57 {
58 __VECTOR_RAM[n] = __VECTOR_TABLE[n];
59 }
60 /* Point the VTOR to the position of vector table */
61 SCB->VTOR = (uint32_t)__VECTOR_RAM;
62 }
63
64 ret = __VECTOR_RAM[(int32_t)irq + 16];
65 /* make sure the __VECTOR_RAM is noncachable */
66 __VECTOR_RAM[(int32_t)irq + 16] = irqHandler;
67
68 EnableGlobalIRQ(irqMaskValue);
69
70 return ret;
71}
72#endif /* ENABLE_RAM_VECTOR_TABLE. */
73#endif /* __GIC_PRIO_BITS. */

89void EnableDeepSleepIRQ(IRQn_Type interrupt)
90{
91 uint32_t intNumber = (uint32_t)interrupt;
92
93 uint32_t index = 0;
94
95 while (intNumber >= 32u)
96 {
97 index++;
98 intNumber -= 32u;
99 }
100
101 SYSCON->STARTERSET[index] = 1UL << intNumber;
102 (void)EnableIRQ(interrupt); /* also enable interrupt at NVIC */
103}

167 "loop%=: \n"

210void SDK_DelayAtLeastUs(uint32_t delayTime_us, uint32_t coreClock_Hz)
211{
212 uint64_t count;
213
214 if (delayTime_us > 0U)
215 {
216 count = USEC_TO_COUNT(delayTime_us, coreClock_Hz);
217
218 assert(count <= UINT32_MAX);
219
220#if defined(SDK_DELAY_USE_DWT) && defined(DWT) /* Use DWT for better accuracy */
221
222 MSDK_EnableCpuCycleCounter();
223 /* Calculate the count ticks. */
224 count += MSDK_GetCpuCycleCount();
225
226 if (count > UINT32_MAX)
227 {
228 count -= UINT32_MAX;
229 /* Wait for cyccnt overflow. */
230 while (count < MSDK_GetCpuCycleCount())
231 {
232 }
233 }
234
235 /* Wait for cyccnt reach count value. */
236 while (count > MSDK_GetCpuCycleCount())
237 {
238 }
239#else
240#if defined(__CORTEX_Axx) && ((__CORTEX_Axx == 53) || (__CORTEX_Axx == 55))
241 /*
242 * Cortex-A53/A55 execution throughput:
243 * - SUB/CMP: 2 instructions per cycle
244 * - BNE: 1 instruction per cycle
245 * So, each loop takes 2 CPU cycles.
246 */
247 count = count / 2U;
248#elif (__CORTEX_M == 7)
249 /* Divide value may be different in various environment to ensure delay is precise.
250 * Every loop count includes three instructions, due to Cortex-M7 sometimes executes
251 * two instructions in one period, through test here set divide 1.5. Other M cores use
252 * divide 4. By the way, divide 1.5 or 4 could let the count lose precision, but it does
253 * not matter because other instructions outside while loop is enough to fill the time.
254 */
255 count = count / 3U * 2U;
256#else
257 count = count / 4U;
258#endif
259 DelayLoop((uint32_t)count);
260#endif /* defined(SDK_DELAY_USE_DWT) && defined(DWT) */
261 }
262}

312uint32_t DisableGlobalIRQEx(uint32_t id)
313{
314 uint32_t primask = DisableGlobalIRQ();
315 if (primask != 0U)
316 {
317#ifdef FSL_FEATURE_MEASURE_CRITICAL_SECTION_DEBUG
318 /* Check for the critical section id. */
319 assert(id != FSL_FEATURE_CRITICAL_SECTION_INVALID_ID);
320 assert(id < FSL_FEATURE_CRITICAL_SECTION_MAX_ID);
321 assert(s_critical_section_measurement_context.id == FSL_FEATURE_CRITICAL_SECTION_INVALID_ID);
322#endif
323 if (s_critical_section_measurement_context.getTimestamp != NULL)
324 {
325 s_critical_section_measurement_context.id = id;
326 s_critical_section_measurement_context.startTime = s_critical_section_measurement_context.getTimestamp();
327 }
328 }
329 return primask;
330}

342void EnableGlobalIRQEx(uint32_t primask)
343{
344 if (primask != 0U)
345 {
346#ifdef FSL_FEATURE_MEASURE_CRITICAL_SECTION_DEBUG
347 /* Check for the critical section id. */
348 assert(s_critical_section_measurement_context.id != FSL_FEATURE_CRITICAL_SECTION_INVALID_ID);
349 assert(s_critical_section_measurement_context.id < FSL_FEATURE_CRITICAL_SECTION_MAX_ID);
350#endif
351 if (s_critical_section_measurement_context.getTimestamp != NULL)
352 {
353 /* Calculate the critical section duration. */
354 uint32_t dur = s_critical_section_measurement_context.getTimestamp() - s_critical_section_measurement_context.startTime;
355 if (dur > s_critical_section_measurement_context.dur_max[s_critical_section_measurement_context.id])
356 {
357 s_critical_section_measurement_context.dur_max[s_critical_section_measurement_context.id] = dur;
358 }
359 s_critical_section_measurement_context.execution_times[s_critical_section_measurement_context.id]++;
360 }
361#ifdef FSL_FEATURE_MEASURE_CRITICAL_SECTION_DEBUG
362 /* Exit the critical section, set the id to invalid. In this case when entering critical
363 section again DisableGlobalIRQEx has to be called first to avoid assertion. */
364 s_critical_section_measurement_context.id = FSL_FEATURE_CRITICAL_SECTION_INVALID_ID;
365#endif
366 }
367 EnableGlobalIRQ(primask);
368}
