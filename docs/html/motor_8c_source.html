259status_t MTR_init(void)
260{
261
262 for (uint8_t i = 0; i < MAX_MOTORS; i++)
263 {
264 motorHandles[i].used = 0;
265 }
266
267 cmdQueue = xQueueCreate(CMD_QUEUE_SIZE, sizeof(MTR_CmdQueueItem_t));
268 if (cmdQueue == NULL)
269 {
270 return kStatus_Fail;
271 }
272 vQueueAddToRegistry(cmdQueue, "MTR Command Queue");
273
274 THE_init_cmd_handles(cmdHandles, MTR_MAX_CMD_HANDLE_COUNT);
275
276 emergencyStopFlag = 0;
277 return kStatus_Success;
278}

603
604static void process(void)
605{
606 MTR_CmdQueueItem_t queueItem;
607 BaseType_t queueStatus;
608 uint8_t activeTaskCount = 0;
609
610 status_t taskStatus = kStatus_Fail;
611 for (uint8_t i = 0; i < MAX_PARALLEL_TASKS; i++)
612 {
613 if (parallelTasks[i].used)
614 {
615 activeTaskCount++;
616 taskStatus =
617 THE_cmd_check_all(parallelTasks[i].cmdHandles, parallelTasks[i].count, NULL);
618 if (taskStatus == kStatus_Busy)
619 {
620 continue;
621 }
622 else if (taskStatus == kStatus_Success)
623 {
624 THE_notify_task_success(parallelTasks[i].returnHandle);
625 }
626 else if (taskStatus == kStatus_Timeout)
627 {
628 THE_notify_task_timeout(parallelTasks[i].returnHandle);
629 }
630 else
631 {
632 THE_notify_task_failure(parallelTasks[i].returnHandle);
633 }
634 THE_remove_cmd_handle_ref(parallelTasks[i].returnHandle);
635 parallelTasks[i].used = 0;
636 }
637 }
638
639 TickType_t delay = activeTaskCount ? pdMS_TO_TICKS(2) : portMAX_DELAY;
640
641 queueStatus = xQueueReceive(cmdQueue, &queueItem, delay);
642 if (queueStatus != pdPASS)
643 {
644 return;
645 }
646
647 // Check emergency stop flag - if set, don't execute any commands
648 if (emergencyStopFlag)
649 {
650 LOG_DEBUG("Command ignored due to active emergency stop");
651 THE_notify_task_failure(queueItem.cmdHandle);
652 THE_remove_cmd_handle_ref(queueItem.cmdHandle);
653 return;
654 }
655
656 status_t cmdStatus = kStatus_Fail;
657 MTR_ParallelTaskItem* taskItem = NULL;
658 for (uint8_t i = 0; i < MAX_PARALLEL_TASKS; i++)
659 {
660 if (parallelTasks[i].used == 0)
661 {
662 parallelTasks[i].used = 1;
663 taskItem = &parallelTasks[i];
664 break;
665 }
666 }
667 if (taskItem != NULL)
668 {
669 taskItem->count = 0;
670 taskItem->returnHandle = queueItem.cmdHandle;
671
672 // Execute command based on type
673 switch (queueItem.type)
674 {
675 case MTR_CMD_MOVE_ANGLE:
676 cmdStatus = move_angle(queueItem.handle, queueItem.data.moveAngle.angle,
677 queueItem.deadline, taskItem);
678 break;
679
680 case MTR_CMD_MOVE_ABSOLUTE_ANGLE:
681 cmdStatus = move_absolute_angle(queueItem.handle, queueItem.data.moveAngle.angle,
682 queueItem.deadline, taskItem);
683 break;
684
685 case MTR_CMD_MOVE_REVOLUTIONS:
686 cmdStatus =
687 move_revolutions(queueItem.handle, queueItem.data.moveRevolutions.revolutions,
688 queueItem.deadline, taskItem);
689 break;
690
691 case MTR_CMD_SET_VELOCITY:
692 cmdStatus =
693 set_velocity(queueItem.handle, queueItem.data.setVelocity.velocity_deg_per_sec,
694 queueItem.deadline, taskItem);
695 break;
696
697 case MTR_CMD_SET_ACCELERATION:
698 cmdStatus = set_acceleration(
699 queueItem.handle, queueItem.data.setAcceleration.acceleration_deg_per_sec2,
700 queueItem.deadline, taskItem);
701 break;
702
703 case MTR_CMD_STOP:
704 cmdStatus = stop_motor(queueItem.handle, queueItem.data.stop.decelerate,
705 queueItem.deadline, taskItem);
706 break;
707
708 case MTR_CMD_GET_CURRENT_ANGLE:
709 cmdStatus = get_current_angle(
710 queueItem.handle, queueItem.data.getCurrentAngle.angle, queueItem.deadline);
711 break;
712
713 case MTR_CMD_GET_MOVEMENT_STATE:
714 cmdStatus =
715 get_movement_state(queueItem.handle, queueItem.data.getMovementState.state);
716 break;
717
718 case MTR_CMD_SET_HOME_POSITION:
719 cmdStatus = set_home_position(queueItem.handle);
720 break;
721
722 case MTR_CMD_SYNCHRONIZED_MOVE:
723 cmdStatus = synchronized_move(

727
728 case MTR_CMD_SET_RUN_CURRENT:

732
733 case MTR_CMD_SET_HOLD_CURRENT:
734 cmdStatus = set_hold_current(

737 case MTR_CMD_INIT_MOTOR:

740 case MTR_CMD_ENABLE_FREEWHEELING:

742 break;
743 case MTR_CMD_DISABLE_FREEWHEELING:
744 cmdStatus = disable_freewheeling(queueItem.handle, queueItem.deadline);
745 break;
746 default:
747 {
748 static char errMsg[100];
749 snprintf(errMsg, sizeof(errMsg), "Unknown motor command type: %d", queueItem.type);
750 LOG_ERROR(errMsg);
751 break;
752 }
753 }
754
755 if (cmdStatus == kStatus_Success && taskItem->count == 0)
756 {
757 THE_notify_task_success(queueItem.cmdHandle);
758 taskItem->used = 0;
759 THE_remove_cmd_handle_ref(queueItem.cmdHandle);
760 }

762 {
763 THE_notify_task_timeout(queueItem.cmdHandle);
764 taskItem->used = 0;
765 THE_remove_cmd_handle_ref(queueItem.cmdHandle);
766 }
767 else if (cmdStatus == kStatus_Fail)
768 {
769 THE_notify_task_failure(queueItem.cmdHandle);
770 taskItem->used = 0;
771 THE_remove_cmd_handle_ref(queueItem.cmdHandle);
772 }

983static int32_t angle_to_steps(MTR_MotorHandle_t handle, double angle, MTR_roundingMethod_t method)
984{
985 if (handle == NULL)
986 {
987 return 0;
988 }
989
990 if (handle->stepAngle <= 0.0 || handle->microstep == 0 || handle->reductionFactor <= 0.0)
991 {
992 return 0;
993 }
994
995 double steps = angle * handle->reductionFactor;
996 steps = steps * (double)handle->microstep;
997 steps = steps / handle->stepAngle;
998
999 // Clamp to int32_t range
1000 if (steps < (double)INT32_MIN)
1001 {
1002 steps = (double)INT32_MIN;
1003 }
1004 if (steps > (double)INT32_MAX)
1005 {
1006 steps = (double)INT32_MAX;
1007 }
1008
1009 int32_t result;
1010 switch (method)
1011 {
1012 case ROUND_DOWN:
1013 result = (int32_t)floor(steps);
1014 break;
1015 case ROUND_UP:
1016 result = (int32_t)ceil(steps);
1017 break;
1018 case ROUND_NEAREST:
1019 result = (int32_t)round(steps);
1020 break;
1021 default:
1022 result = (int32_t)round(steps);
1023 break;
1024 }
1025 return result;
1026}

1271static inline status_t calculate_sync_motion_parameters(MTR_MotorHandle_t* handles, double* angles,
1272 uint8_t count, int32_t* stepCounts,
1273 double* originalVelocities,
1274 double* maxDuration)
1275{
1276 *maxDuration = 0.0;
1277
1278 for (uint8_t i = 0; i < count; i++)
1279 {
1280 if (handles[i] == NULL)
1281 {
1282 return kStatus_Fail;
1283 }
1284
1285 stepCounts[i] = angle_to_steps(handles[i], angles[i], ROUND_NEAREST);
1286 if (STP_get_end_velocity(handles[i]->stepperHandle, &originalVelocities[i]) !=
1287 kStatus_Success)
1288 {
1289 return kStatus_Fail;
1290 }
1291
1292 if (originalVelocities[i] <= 0)
1293 {
1294 return kStatus_Fail;
1295 }
1296
1297 // Use absolute value of steps for duration calculation
1298 uint32_t absSteps = (stepCounts[i] >= 0) ? stepCounts[i] : -stepCounts[i];
1299 double moveDuration = (double)absSteps / originalVelocities[i];
1300 if (moveDuration > *maxDuration)
1301 {
1302 *maxDuration = moveDuration;
1303 }
1304 }
1305
1306 return kStatus_Success;
1307}
