70static bool RTC_CheckDatetimeFormat(const rtc_datetime_t *datetime)
71{
72 assert(NULL != datetime);
73
74 /* Table of days in a month for a non leap year. First entry in the table is not used,
75 * valid months start from 1
76 */
77 uint8_t daysPerMonth[] = {0U, 31U, 28U, 31U, 30U, 31U, 30U, 31U, 31U, 30U, 31U, 30U, 31U};
78
79 /* Check year, month, hour, minute, seconds */
80 if ((datetime->year < YEAR_RANGE_START) || (datetime->year > YEAR_RANGE_END) || (datetime->month > 12U) ||
81 (datetime->month < 1U) || (datetime->hour >= 24U) || (datetime->minute >= 60U) || (datetime->second >= 60U))
82 {
83 /* If not correct then error*/
84 return false;
85 }
86
87 /* Adjust the days in February for a leap year */
88 if ((((datetime->year & 3U) == 0U) && (datetime->year % 100U != 0U)) || (datetime->year % 400U == 0U))
89 {
90 daysPerMonth[2] = 29U;
91 }
92
93 /* Check the validity of the day */
94 if ((datetime->day > daysPerMonth[datetime->month]) || (datetime->day < 1U))
95 {
96 return false;
97 }
98
99 return true;
100}

102static uint32_t RTC_ConvertDatetimeToSeconds(const rtc_datetime_t *datetime)
103{
104 assert(NULL != datetime);
105
106 /* Number of days from begin of the non Leap-year*/
107 /* Number of days from begin of the non Leap-year*/
108 uint16_t monthDays[] = {0U, 0U, 31U, 59U, 90U, 120U, 151U, 181U, 212U, 243U, 273U, 304U, 334U};
109 uint32_t seconds;
110
111 /* Compute number of days from 1970 till given year*/
112 seconds = ((uint32_t)datetime->year - 1970U) * DAYS_IN_A_YEAR;
113 /* Add leap year days */
114 seconds += (((uint32_t)datetime->year / 4U) - (1970U / 4U));
115 /* Add number of days till given month*/
116 seconds += monthDays[datetime->month];
117 /* Add days in given month. We subtract the current day as it is
118 * represented in the hours, minutes and seconds field*/
119 seconds += ((uint32_t)datetime->day - 1U);
120 /* For leap year if month less than or equal to Febraury, decrement day counter*/
121 if ((0U == (datetime->year & 3U)) && (datetime->month <= 2U) && 0U != seconds)
122 {
123 seconds--;
124 }
125
126 assert(seconds < UINT32_MAX / SECONDS_IN_A_DAY);
127 seconds = (seconds * SECONDS_IN_A_DAY) + ((uint32_t)datetime->hour * SECONDS_IN_A_HOUR) +
128 ((uint32_t)datetime->minute * SECONDS_IN_A_MINUTE) + datetime->second;
129
130 return seconds;
131}

133static void RTC_ConvertSecondsToDatetime(uint32_t seconds, rtc_datetime_t *datetime)
134{
135 assert(NULL != datetime);
136
137 uint32_t x;
138 uint32_t secondsRemaining, days;
139 uint16_t daysInYear;
140 /* Table of days in a month for a non leap year. First entry in the table is not used,
141 * valid months start from 1
142 */
143 uint8_t daysPerMonth[] = {0U, 31U, 28U, 31U, 30U, 31U, 30U, 31U, 31U, 30U, 31U, 30U, 31U};
144
145 /* Start with the seconds value that is passed in to be converted to date time format */
146 secondsRemaining = seconds;
147
148 /* Calcuate the number of days, we add 1 for the current day which is represented in the
149 * hours and seconds field
150 */
151 days = secondsRemaining / SECONDS_IN_A_DAY + 1U;
152
153 /* Update seconds left*/
154 secondsRemaining = secondsRemaining % SECONDS_IN_A_DAY;
155
156 /* Calculate the datetime hour, minute and second fields */
157 datetime->hour = (uint8_t)(secondsRemaining / SECONDS_IN_A_HOUR);
158 secondsRemaining = secondsRemaining % SECONDS_IN_A_HOUR;
159 datetime->minute = (uint8_t)(secondsRemaining / 60U);
160 datetime->second = (uint8_t)(secondsRemaining % SECONDS_IN_A_MINUTE);
161
162 /* Calculate year */
163 daysInYear = DAYS_IN_A_YEAR;
164 datetime->year = YEAR_RANGE_START;
165 while (days > daysInYear)
166 {
167 /* Decrease day count by a year and increment year by 1 */
168 days -= daysInYear;
169 datetime->year++;
170
171 /* Adjust the number of days for a leap year */
172 if (0U != (datetime->year & 3U))
173 {
174 daysInYear = DAYS_IN_A_YEAR;
175 }
176 else
177 {
178 daysInYear = DAYS_IN_A_YEAR + 1U;
179 }
180 }
181
182 /* Adjust the days in February for a leap year */
183 if (0U == (datetime->year & 3U))
184 {
185 daysPerMonth[2] = 29U;
186 }
187
188 /*
189 * $Branch Coverage Justification$
190 * (x > 12U) not covered. $ref rtc_c_ref_1$.
191 */
192 for (x = 1U; x <= 12U; x++)
193 {
194 if (days <= daysPerMonth[x])
195 {
196 datetime->month = (uint8_t)x;
197 break;
198 }
199 else
200 {
201 days -= daysPerMonth[x];
202 }
203 }
204
205 datetime->day = (uint8_t)days;
206}

218void RTC_Init(RTC_Type *base, const rtc_config_t *config)
219{
220 assert(NULL != config);
221
222 uint32_t reg;
223
224#if defined(RTC_CLOCKS)
225#if !(defined(FSL_SDK_DISABLE_DRIVER_CLOCK_CONTROL) && FSL_SDK_DISABLE_DRIVER_CLOCK_CONTROL)
226 CLOCK_EnableClock(kCLOCK_Rtc0);
227#endif /* FSL_SDK_DISABLE_DRIVER_CLOCK_CONTROL */
228#endif /* RTC_CLOCKS */
229
230 /* Issue a software reset if timer is invalid */
231 if ((uint32_t)kRTC_TimeInvalidFlag == (RTC_GetStatusFlags(base) & (uint32_t)kRTC_TimeInvalidFlag))
232 {
233 RTC_Reset(base);
234 }
235
236 reg = base->CR;
237/* Setup the update mode and supervisor access mode */
238#if !(defined(FSL_FEATURE_RTC_HAS_NO_CR_OSCE) && FSL_FEATURE_RTC_HAS_NO_CR_OSCE)
239 reg &= ~(RTC_CR_UM_MASK | RTC_CR_SUP_MASK);
240 reg |= RTC_CR_UM(config->updateMode ? 1U : 0U) | RTC_CR_SUP(config->supervisorAccess ? 1U : 0U);
241#else
242 reg &= ~RTC_CR_UM_MASK;
243 reg |= RTC_CR_UM(config->updateMode ? 1U : 0U);
244#endif
245
246#if defined(FSL_FEATURE_RTC_HAS_WAKEUP_PIN_SELECTION) && FSL_FEATURE_RTC_HAS_WAKEUP_PIN_SELECTION
247 /* Setup the wakeup pin select */
248 reg &= ~(RTC_CR_WPS_MASK);
249 reg |= RTC_CR_WPS(config->wakeupSelect ? 1U : 0U);
250#endif /* FSL_FEATURE_RTC_HAS_WAKEUP_PIN */
251 base->CR = reg;
252
253 /* Configure the RTC time compensation register */
254 base->TCR = (RTC_TCR_CIR(config->compensationInterval) | RTC_TCR_TCR(config->compensationTime));
255
256#if defined(FSL_FEATURE_RTC_HAS_TSIC) && FSL_FEATURE_RTC_HAS_TSIC
257 /* Configure RTC timer seconds interrupt to be generated once per second */
258 base->IER &= ~(RTC_IER_TSIC_MASK | RTC_IER_TSIE_MASK);
259#endif
260}

275void RTC_GetDefaultConfig(rtc_config_t *config)
276{
277 assert(NULL != config);
278
279 /* Initializes the configure structure to zero. */
280 (void)memset(config, 0, sizeof(*config));
281
282 /* Wakeup pin will assert if the RTC interrupt asserts or if the wakeup pin is turned on */
283 config->wakeupSelect = false;
284 /* Registers cannot be written when locked */
285 config->updateMode = false;
286 /* Non-supervisor mode write accesses are not supported and will generate a bus error */
287 config->supervisorAccess = false;
288 /* Compensation interval used by the crystal compensation logic */
289 config->compensationInterval = 0;
290 /* Compensation time used by the crystal compensation logic */
291 config->compensationTime = 0;
292}

351status_t RTC_SetAlarm(RTC_Type *base, const rtc_datetime_t *alarmTime)
352{
353 uint32_t alarmSeconds = 0;
354 uint32_t currSeconds = 0;
355
356#if defined(FSL_FEATURE_RTC_HAS_ERRATA_010716) && (FSL_FEATURE_RTC_HAS_ERRATA_010716)
357 bool restartCounter = false;
358#endif /* FSL_FEATURE_RTC_HAS_ERRATA_010716 */
359
360 /* Return error if the alarm time provided is not valid */
361 if (!(RTC_CheckDatetimeFormat(alarmTime)))
362 {
363 return kStatus_InvalidArgument;
364 }
365
366 alarmSeconds = RTC_ConvertDatetimeToSeconds(alarmTime);
367
368 /* Get the current time */
369 currSeconds = base->TSR;
370
371 /* Return error if the alarm time has passed */
372 if (alarmSeconds < currSeconds)
373 {
374 return kStatus_Fail;
375 }
376
377#if defined(FSL_FEATURE_RTC_HAS_ERRATA_010716) && (FSL_FEATURE_RTC_HAS_ERRATA_010716)
378 /* Save current TCE state */
379 restartCounter = ((base->SR & RTC_SR_TCE_MASK) != 0U);
380 /* Disable time counter */
381 base->SR &= ~RTC_SR_TCE_MASK;
382#endif /* FSL_FEATURE_RTC_HAS_ERRATA_010716 */
383
384 /* Set alarm in seconds*/
385 base->TAR = alarmSeconds;
386
387#if defined(FSL_FEATURE_RTC_HAS_ERRATA_010716) && (FSL_FEATURE_RTC_HAS_ERRATA_010716)
388 /* Restore TCE if it was enabled */
389 if (restartCounter)
390 {
391 base->SR |= RTC_SR_TCE_MASK;
392 }
393#endif /* FSL_FEATURE_RTC_HAS_ERRATA_010716 */
394
395 return kStatus_Success;
396}
