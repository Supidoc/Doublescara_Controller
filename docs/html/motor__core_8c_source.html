53void MTR_task(void* pvParameters)
54{
55 (void)pvParameters;
56 LOG_INFO("Started Motor Task");
57 mtrTaskHandle = xTaskGetCurrentTaskHandle();
58 for (;;)
59 {
60 MTRi_process();
61 vTaskDelay(pdMS_TO_TICKS(10));
62 }
63}

65status_t MTR_init(void)
66{
67 for (uint8_t i = 0; i < MTR_MAX_MOTORS; i++)
68 {
69 motorHandles[i].used = 0;
70 }
71
72 mtrCmdQueue = xQueueCreate(MTR_CMD_QUEUE_SIZE, sizeof(MTR_CmdQueueItem_t));
73 if (mtrCmdQueue == NULL)
74 {
75 return kStatus_Fail;
76 }
77 vQueueAddToRegistry(mtrCmdQueue, "MTR Command Queue");
78
79 CHD_init_cmd_handles(mtrCmdHandles, MTR_MAX_CMD_HANDLE_COUNT);
80 emergencyStopFlag = 0;
81 return kStatus_Success;
82}

101void MTR_get_motor_by_label(const char* label, MTR_MotorHandle_t* handle)
102{
103 if (label == NULL || handle == NULL)
104 {
105 return;
106 }
107
108 for (uint8_t i = 0; i < MTR_MAX_MOTORS; i++)
109 {
110 if (motorHandles[i].used)
111 {
112 if (strncmp(label, motorHandles[i].handle.label,
113 strlen(motorHandles[i].handle.label)) == 0)
114 {
115 *handle = &motorHandles[i].handle;
116 return;
117 }
118 }
119 }
120
121 *handle = NULL;
122}

358{
