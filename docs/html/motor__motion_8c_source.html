42status_t MTRi_move_angle(MTR_MotorHandle_t handle, double angle, TickType_t deadline,
43 MTR_ParallelTaskItem* taskItem)
44{
45 static char logMsg[100];
46
47 if (handle == NULL || taskItem == NULL)
48 {
49 return kStatus_Fail;
50 }
51 snprintf(logMsg, sizeof(logMsg), "[%s] Moving by %.2f degrees", handle->label, angle);
52 LOG_DEBUG(logMsg);
53
54 int32_t steps = MTR_angle_to_steps(handle, angle, ROUND_NEAREST);
55
56 CHD_CmdHandle_t cmdHandle = NULL;
57 if (STP_move_relative_async(handle->stepperHandle, steps, deadline, &cmdHandle) !=
58 kStatus_Success)
59 {
60 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to queue relative move command",
61 handle->label);
62 LOG_ERROR(logMsg);
63 return kStatus_Fail;
64 }
65 taskItem->cmdHandles[0] = cmdHandle;
66 taskItem->count = 1;
67 return kStatus_Success;
68}

70status_t MTRi_move_absolute_angle(MTR_MotorHandle_t handle, double angle, TickType_t deadline,
71 MTR_ParallelTaskItem* taskItem)
72{
73 static char logMsg[100];
74
75 if (handle == NULL || taskItem == NULL)
76 {
77 return kStatus_Fail;
78 }
79 snprintf(logMsg, sizeof(logMsg), "[%s] Moving to absolute angle %.2f degrees", handle->label,
80 angle);
81 LOG_DEBUG(logMsg);
82
83 int32_t currentSteps;
84 if (STP_get_absolute_steps(handle->stepperHandle, &currentSteps) != kStatus_Success)
85 {
86 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to read current position", handle->label);
87 LOG_ERROR(logMsg);
88 return kStatus_Fail;
89 }
90 int32_t targetSteps = MTR_angle_to_steps(handle, angle, ROUND_NEAREST);
91 int32_t stepsToMove = targetSteps - currentSteps;
92
93 CHD_CmdHandle_t cmdHandle = NULL;
94 if (STP_move_relative_async(handle->stepperHandle, stepsToMove, deadline, &cmdHandle) !=
95 kStatus_Success)
96 {
97 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to queue absolute move command",
98 handle->label);
99 LOG_ERROR(logMsg);
100 return kStatus_Fail;
101 }
102
103 taskItem->cmdHandles[0] = cmdHandle;
104 taskItem->count = 1;
105 return kStatus_Success;
106}
