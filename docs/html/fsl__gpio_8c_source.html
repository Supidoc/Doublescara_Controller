106static uint32_t GPIO_GetInstance(GPIO_Type *base)
107{
108 uint32_t instance;
109
110 /* Find the instance index from base address mappings. */
111 /*
112 * $Branch Coverage Justification$
113 * (instance >= ARRAY_SIZE(s_gpioBases)) not covered.
114 * $ref gpio_c_ref_1$.
115 */
116 for (instance = 0; instance < ARRAY_SIZE(s_gpioBases); instance++)
117 {
118 if (MSDK_REG_SECURE_ADDR(s_gpioBases[instance]) == MSDK_REG_SECURE_ADDR(base))
119 {
120 break;
121 }
122 }
123
124 assert(instance < ARRAY_SIZE(s_gpioBases));
125
126 return instance;
127}

204void GPIO_PinInit(GPIO_Type *base, uint32_t pin, const gpio_pin_config_t *config)
205{
206 assert(NULL != config);
207
208 GPIO_PortClockEnable(base, true);
209
210#if defined(GPIO_RESETS_ARRAY)
211 RESET_ReleasePeripheralReset(s_gpioResets[GPIO_GetInstance(base)]);
212#endif
213
214 if (config->pinDirection == kGPIO_DigitalInput)
215 {
216 base->PDDR &= GPIO_FIT_REG(~(1UL << pin));
217 }
218 else
219 {
220 GPIO_PinWrite(base, pin, config->outputLogic);
221 base->PDDR |= GPIO_FIT_REG((1UL << pin));
222 }
223}
