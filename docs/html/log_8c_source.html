125status_t LOG_init(void)
126{
127 logQueue = xQueueCreate(LOG_QUEUE_SIZE, sizeof(LOG_QueueItem_t));
128 if (logQueue == NULL)
129 {
130 return kStatus_Fail;
131 }
132 vQueueAddToRegistry(logQueue, "LOG Queue");
133
134 logFileMutex = xSemaphoreCreateMutex();
135 if (logFileMutex == NULL)
136 {
137 return kStatus_Fail;
138 }
139
140 logQueueEmptyEvent = xEventGroupCreate();
141 if (logQueueEmptyEvent == NULL)
142 {
143 return kStatus_Fail;
144 }
145
146 if (set_session_id() != kStatus_Success)
147 {
148 return kStatus_Fail;
149 }
150
151 // set set file path and name
152 sprintf(LogFilePath, DISK_SD_PATH "%lu", sessionId);
153
154 // create log file
155 FRESULT res;
156 res = f_open(&logFile, LogFilePath, FA_WRITE | FA_OPEN_ALWAYS);
157 if (res != FR_OK)
158 {
159 return kStatus_Fail;
160 }
161
162 LOG_LogHeader_t logHeader;
163 logHeader.magic = LOG_MAGIC;
164 logHeader.headerSize = sizeof(LOG_LogHeader_t);
165 logHeader.version = 1;
166 logHeader.sessionId = sessionId;
167 logHeader.tickFreqHz = configTICK_RATE_HZ;
168
169 if (append_to_file((uint8_t*)&logHeader, sizeof(LOG_LogHeader_t)) != kStatus_Success)
170 {
171 return kStatus_Fail;
172 }
173
174 res = f_sync(&logFile);
175 if (res != FR_OK)
176 {
177 return kStatus_Fail;
178 }
179
180 res = f_close(&logFile);
181 if (res != FR_OK)
182 {
183 return kStatus_Fail;
184 }
185
186 return kStatus_Success;
187}

189status_t LOG_send_log_message(LOG_Level_t level, char* message, uint8_t silent)
190{
191 if (message == NULL)
192 {
193 return kStatus_Fail;
194 }
195 LOG_QueueItem_t queueItem;
196
197 queueItem.level = level;
198 queueItem.silent = silent && 0b1;
199 strcpy(queueItem.taskName, pcTaskGetName(NULL));
200 queueItem.tick = xTaskGetTickCount();
201 strncpy(queueItem.msg, message, LOG_MAX_MESSAGE_SIZE - 1);
202 queueItem.msg[LOG_MAX_MESSAGE_SIZE - 1] = '\0';
203
204 if (xQueueSend(logQueue, (void*)&queueItem, portMAX_DELAY) != pdTRUE)
205 {
206 return kStatus_Fail;
207 }
208
209 // Clear the queue empty event bit since we just added a message
210 if (logQueueEmptyEvent != NULL)
211 {
212 xEventGroupClearBits(logQueueEmptyEvent, LOG_QUEUE_EMPTY_EVENT);
213 }
214
215 return kStatus_Success;
216}

230status_t LOG_wait_for_queue_empty(TickType_t xTicksToWait)
231{
232 if (logQueueEmptyEvent == NULL)
233 {
234 return kStatus_Fail;
235 }
236
237 // Don't clear the bit on return - it stays set as long as queue is empty
238 EventBits_t bits = xEventGroupWaitBits(logQueueEmptyEvent, LOG_QUEUE_EMPTY_EVENT, pdFALSE,
239 pdFALSE, xTicksToWait);
240
241 if (bits & LOG_QUEUE_EMPTY_EVENT)
242 {
243 return kStatus_Success;
244 }
245 return kStatus_Fail;
246}

252static BaseType_t log_command(char* pcWriteBuffer, size_t xWriteBufferLen,
253 const char* pcCommandString)
254{
255 static uint32_t fileSessionId;
256 FRESULT res;
257 LOG_LogHeaderPrefix_t prefix;
258 static LOG_LogHeader_t header;
259 UINT br;
260 static char localLogFilePath[11];
261 static FIL localLogFile;
262
263 if (logFileOpened == 0)
264 {
265 if (xSemaphoreTake(logFileMutex, portMAX_DELAY) != pdTRUE)
266 {
267 return kStatus_Fail;
268 }
269
270 res = f_open(&logFile, LogFilePath, FA_WRITE | FA_OPEN_ALWAYS);
271 if (res != FR_OK)
272 {
273 return kStatus_Fail;
274 }
275 logFileOpened = 1;
276
277 uint8_t parameterFound;
278 const char* pcParameter = NULL;
279 BaseType_t parameterStringLength;
280 CLU_get_parameter_value_string("-s", pcCommandString, &parameterFound, &pcParameter,
281 &parameterStringLength);
282 if (parameterFound == 0 || parameterStringLength == 0)
283 {
284 fileSessionId = sessionId;
285 }
286 else
287 {
288 fileSessionId = strtoul(pcParameter, NULL, 0);
289 }
290 sprintf(localLogFilePath, DISK_SD_PATH "%lu", fileSessionId);
291
292 res = f_open(&localLogFile, localLogFilePath, FA_READ | FA_OPEN_ALWAYS);
293 if (res != FR_OK)
294 {
295 return kStatus_Fail;
296 }
297
298 res = f_read(&localLogFile, &prefix, sizeof(prefix), &br);
299 if (res != FR_OK || br != sizeof(prefix))
300 {
301 strncpy(pcWriteBuffer, "Invalid file format", strlen("Invalid file format") + 1);
302 return pdFALSE;
303 }
304
305 if (prefix.magic != LOG_MAGIC)
306 {
307 strncpy(pcWriteBuffer, "Invalid file format", strlen("Invalid file format") + 1);
308 return pdFALSE;
309 }
310
311 f_rewind(&localLogFile);
312 res = f_read(&localLogFile, &header, sizeof(header), &br);
313 if (res != FR_OK || br != sizeof(header))
314 {
315 strncpy(pcWriteBuffer, "Invalid file format", strlen("Invalid file format") + 1);
316 return pdFALSE;
317 }
318
319 assemble_header_string(pcWriteBuffer, xWriteBufferLen, header);
320
321 logFileOpened = 1;
322 }
323 else
324 {
325
326 LOG_QueueItem_t queueItem;
327 res = f_read(&localLogFile, &queueItem, sizeof(LOG_QueueItem_t), &br);
328
329 if (br == sizeof(LOG_QueueItem_t))
330 {
331 assemble_log_string(pcWriteBuffer, xWriteBufferLen, queueItem);
332 }
333 }
334 if (f_eof(&localLogFile) == 0)
335 {
336 return pdTRUE;
337 }
338
339 res = f_sync(&logFile);
340 if (res != FR_OK)
341 {
342 return kStatus_Fail;
343 }
344
345 res = f_close(&logFile);
346 if (res != FR_OK)
347 {
348 return kStatus_Fail;
349 }
350 if (xSemaphoreGive(logFileMutex) != pdTRUE)
351 {
352 return kStatus_Fail;
353 }
354
355 f_close(&localLogFile);
356 logFileOpened = 0;
357 return pdFALSE;
358}

417static void assemble_log_string(char* dest, size_t destSize, LOG_QueueItem_t queueItem)
418{
419 const char* levelString;
420 switch (queueItem.level)
421 {
422 case LOG_LEVEL_DEBUG:
423 levelString = "DEBUG";
424 break;
425 case LOG_LEVEL_INFO:
426 levelString = "INFO";
427 break;
428 case LOG_LEVEL_WARN:
429 levelString = "WARN";
430 break;
431 case LOG_LEVEL_ERROR:
432 levelString = "ERROR";
433 break;
434 case LOG_LEVEL_FATAL:
435 levelString = "FATAL";
436 break;
437 default:
438 levelString = "UNKNOWN";
439 break;
440 }
441
442 uint32_t timestampMs = pdTICKS_TO_MS(queueItem.tick);
443 snprintf(dest, destSize, "[%s]:[%s]:[t+%lu]:%s \r\n", levelString, queueItem.taskName,
444 (uint32_t)timestampMs, queueItem.msg);

472static void process(void)
473{
474 LOG_QueueItem_t queueItem;
475 BaseType_t status;
476 FRESULT res;
477 status = xQueueReceive(logQueue, &queueItem, portMAX_DELAY);
478 if (status != pdPASS)
479 {
480 return;
481 }
482
483 if (enableLogToConsole && queueItem.level > consoleLogLevel && queueItem.silent == 0)
484 {
485 const char* levelString;
486 switch (queueItem.level)
487 {
488 case LOG_LEVEL_DEBUG:
489 levelString = "DEBUG";
490 break;
491 case LOG_LEVEL_INFO:
492 levelString = "INFO";
493 break;
494 case LOG_LEVEL_WARN:
495 levelString = "WARN";
496 break;
497 case LOG_LEVEL_ERROR:
498 levelString = "ERROR";
499 break;
500 case LOG_LEVEL_FATAL:
501 levelString = "FATAL";
502 break;
503 default:
504 levelString = "UNKNOWN";
505 break;
506 }
507 size_t len = snprintf(consoleOutputString, CLI_MAX_OUTPUT_LENGTH,
508 "[%s]:[%s]:[t+%lu]:%s \r\n", levelString, queueItem.taskName,
509 (uint32_t)pdTICKS_TO_MS(queueItem.tick), queueItem.msg);
510#if CLI_USE_RTT
511 SEGGER_RTT_Write(0, (const uint8_t*)consoleOutputString, len);
512#else
513 LPUART_RTOS_Send(&LPUART0_rtos_handle, (uint8_t*)consoleOutputString, len);
514#endif
515 }
516
517 if (logFileOpened == 0)
518 {
519
520 if (xSemaphoreTake(logFileMutex, portMAX_DELAY) != pdTRUE)
521 {
522 return;
523 }
524
525 res = f_open(&logFile, LogFilePath, FA_WRITE | FA_OPEN_ALWAYS);
526 if (res != FR_OK)
527 {
528 return;
529 }
530 logFileOpened = 1;
531 }
532
533 append_to_file((uint8_t*)&queueItem, sizeof(queueItem));
534
535 if (xQueuePeek(logQueue, &queueItem, 0) != pdPASS)
536 {
537 res = f_sync(&logFile);
538 if (res != FR_OK)
539 {
540 return;
541 }
542
543 res = f_close(&logFile);
544 if (res != FR_OK)
545 {
546 return;
547 }
548
549 logFileOpened = 0;
550
551 if (xSemaphoreGive(logFileMutex) != pdTRUE)
552 {
553 return;
554 }
555
556 // Signal all waiting tasks that queue is empty
557 xEventGroupSetBits(logQueueEmptyEvent, LOG_QUEUE_EMPTY_EVENT);
558 }
559}

562{
563 FRESULT res;
564 FIL file;
565 res = f_open(&file, DISK_SD_PATH "session.id", FA_READ | FA_WRITE | FA_OPEN_ALWAYS);

567 {
568
569 return kStatus_Fail;
570 }
571
572 UINT br;
573 f_read(&file, &sessionId, sizeof(sessionId), &br);
574 f_rewind(&file);
575 if (br != sizeof(sessionId))

577 sessionId = 1;
578 }
579 else
580 {
581 sessionId++;
582 }
583 UINT bw;
584 res = f_write(&file, &sessionId, sizeof(sessionId), &bw);
585 if (res != FR_OK)
586 {
587 return kStatus_Fail;
588 }
589 f_sync(&file);
590
591 f_close(&file);
592 return kStatus_Success;
