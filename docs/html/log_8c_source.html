115status_t LOG_init(void)
116{
117 logQueue = xQueueCreate(LOG_QUEUE_SIZE, sizeof(LOG_QueueItem_t));
118 if (logQueue == NULL)
119 return kStatus_Fail;
120
121 if (set_session_id() != kStatus_Success)
122 return kStatus_Fail;
123
124 // set set file path and name
125 sprintf(LogFilePath, DISK_SD_PATH "%lu", sessionId);
126
127 // create log file
128 FRESULT res;
129 res = f_open(&logFile, LogFilePath, FA_WRITE | FA_OPEN_ALWAYS);
130 if (res != FR_OK)
131 return kStatus_Fail;
132
133 f_close(&logFile);
134
135 LOG_LogHeader_t logHeader;
136 logHeader.magic = LOG_MAGIC;
137 logHeader.headerSize = sizeof(LOG_LogHeader_t);
138 logHeader.version = 1;
139 logHeader.sessionId = sessionId;
140 logHeader.tickFreqHz = configTICK_RATE_HZ;
141
142 if (append_to_file((uint8_t*)&logHeader, sizeof(LOG_LogHeader_t)) != kStatus_Success)
143 return kStatus_Fail;
144
145 return kStatus_Success;
146}

148status_t LOG_send_log_message(LOG_Level_t level, char* message, uint8_t silent)
149{
150 if (message == NULL)
151 return kStatus_Fail;
152 LOG_QueueItem_t queueItem;
153
154 queueItem.level = level;
155 queueItem.silent = silent && 0b1;
156 strcpy(queueItem.taskName, pcTaskGetName(NULL));
157 queueItem.tick = xTaskGetTickCount();
158 strncpy(queueItem.msg, message, LOG_MAX_MESSAGE_SIZE - 1);
159 queueItem.msg[LOG_MAX_MESSAGE_SIZE - 1] = '\0';
160
161 if (xQueueSend(logQueue, (void*)&queueItem, portMAX_DELAY) != pdTRUE)
162 return kStatus_Fail;
163 return kStatus_Success;
164}

183static BaseType_t log_command(char* pcWriteBuffer, size_t xWriteBufferLen,
184 const char* pcCommandString)
185{
186 static uint8_t fileOpened = 0;
187 static uint32_t fileSessionId;
188 FRESULT res;
189 LOG_LogHeaderPrefix_t prefix;
190 static LOG_LogHeader_t header;
191 UINT br;
192 static char localLogFilePath[11];
193 static FIL localLogFile;
194
195 if (fileOpened != 1u)
196 {
197 uint8_t parameterFound;
198 const char* pcParameter = NULL;
199 BaseType_t parameterStringLength;
200 CLU_get_parameter_value_string("-s", pcCommandString, &parameterFound, &pcParameter,
201 &parameterStringLength);
202 if (parameterFound == 0 || parameterStringLength == 0)
203 {
204 fileSessionId = sessionId;
205 }
206 else
207 {
208 fileSessionId = strtoul(pcParameter, NULL, 0);
209 }
210 sprintf(localLogFilePath, DISK_SD_PATH "%lu", fileSessionId);
211
212 res = f_open(&localLogFile, localLogFilePath, FA_READ | FA_OPEN_ALWAYS);
213 if (res != FR_OK)
214 return kStatus_Fail;
215
216 res = f_read(&localLogFile, &prefix, sizeof(prefix), &br);
217 if (res != FR_OK || br != sizeof(prefix))
218 {
219 strncpy(pcWriteBuffer, "Invalid file format", strlen("Invalid file format") + 1);
220 return pdFALSE;
221 }
222
223 if (prefix.magic != LOG_MAGIC)
224 {
225 strncpy(pcWriteBuffer, "Invalid file format", strlen("Invalid file format") + 1);
226 return pdFALSE;
227 }
228
229 f_rewind(&localLogFile);
230 res = f_read(&localLogFile, &header, sizeof(header), &br);
231 if (res != FR_OK || br != sizeof(header))
232 {
233 strncpy(pcWriteBuffer, "Invalid file format", strlen("Invalid file format") + 1);
234 return pdFALSE;
235 }
236
237 assemble_header_string(pcWriteBuffer, header);
238
239 fileOpened = 1u;
240 }
241 else
242 {
243
244 LOG_QueueItem_t queueItem;
245 res = f_read(&localLogFile, &queueItem, sizeof(LOG_QueueItem_t), &br);
246
247 if (br == sizeof(LOG_QueueItem_t))
248 {
249 assemble_log_string(pcWriteBuffer, queueItem);
250 }
251 }
252 if (f_eof(&localLogFile) == 0)
253 {
254 return pdTRUE;
255 }
256
257 f_close(&localLogFile);
258 fileOpened = 0u;
259 return pdFALSE;
260}

319static void assemble_log_string(char* dest, LOG_QueueItem_t queueItem)
320{
321
322 char* levelString;
323 switch (queueItem.level)
324 {
325 case LOG_LEVEL_DEBUG:
326 levelString = "DEBUG";
327 break;
328 case LOG_LEVEL_INFO:
329 levelString = "INFO";
330 break;
331 case LOG_LEVEL_WARN:
332 levelString = "WARN";
333 break;
334 case LOG_LEVEL_ERROR:
335 levelString = "ERROR";
336 break;
337 case LOG_LEVEL_FATAL:
338 levelString = "FATAL";
339 break;
340 };
341
342 strcpy(dest, "[");
343
344 strcat(dest, levelString);
345
346 strcat(dest, "]:[");
347
348 strcat(dest, queueItem.taskName);
349
350 strcat(dest, "]:[t+");
351
352 uint32_t timestampMs = pdTICKS_TO_MS(queueItem.tick);
353 char timestampString[21];
354 sprintf(timestampString, "%lu", (uint32_t)timestampMs);
355 strcat(dest, timestampString);
356
357 strcat(dest, "]:");
358
359 strcat(dest, queueItem.msg);
360
361 strcat(dest, " \r\n\0");
362}

382static status_t append_to_file(uint8_t* buffer, size_t bufferLength)
383{
384 FRESULT res;
385 res = f_open(&logFile, LogFilePath, FA_WRITE | FA_OPEN_ALWAYS);
386 if (res != FR_OK)
387 return kStatus_Fail;
388
389 res = f_lseek(&logFile, f_size(&logFile));
390 if (res != FR_OK)
391 return kStatus_Fail;
392
393 UINT bw;
394 res = f_write(&logFile, buffer, bufferLength, &bw);
395 if (res != FR_OK || bw != bufferLength)
396 return kStatus_Fail;
397
398 res = f_sync(&logFile);
399 if (res != FR_OK)
400 return kStatus_Fail;
401
402 res = f_close(&logFile);
403 if (res != FR_OK)
404 return kStatus_Fail;
405
406 return kStatus_Success;
407}

409static void process(void)
410{
411 LOG_QueueItem_t queueItem;
412 BaseType_t status;
413 status = xQueueReceive(logQueue, &queueItem, portMAX_DELAY);
414 if (status != pdPASS)
415 return;
416
417 if (enableLogToConsole && queueItem.level > consoleLogLevel && queueItem.silent == 0)
418 {
419 assemble_log_string(consoleOutputString, queueItem);
420 size_t len = strlen((char*)consoleOutputString);
421 LPUART_RTOS_Send(&LPUART0_rtos_handle, (uint8_t*)consoleOutputString, len);
422 }
423 append_to_file((uint8_t*)&queueItem, sizeof(queueItem));
424}

426static status_t set_session_id(void)
427{
428 FRESULT res;
429 FIL file;
430 res = f_open(&file, DISK_SD_PATH "session.id", FA_READ | FA_WRITE | FA_OPEN_ALWAYS);
431 if (res != FR_OK)
432
433 return kStatus_Fail;
434
435 UINT br;
436 f_read(&file, &sessionId, sizeof(sessionId), &br);
437 f_rewind(&file);
438 if (br != sizeof(sessionId))
439 {
440 sessionId = 1;
441 }
442 else
443 {
444 sessionId++;
445 }
446 UINT bw;
447 res = f_write(&file, &sessionId, sizeof(sessionId), &bw);
448 if (res != FR_OK)
449 return kStatus_Fail;
450 f_sync(&file);
451
452 f_close(&file);
453 return kStatus_Success;
454}
