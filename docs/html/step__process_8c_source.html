48status_t STPi_send_cmd_async(STP_CmdQueueItem_t* queueItem, TickType_t deadline,
49 CHD_CmdHandle_t* cmdHandle)
50{
51 if (queueItem == NULL)
52 {
53 return kStatus_Fail;
54 }
55 CHD_CmdHandle_t internaleCmdHandle = NULL;
56
57 status_t allocStatus =
58 CHD_get_cmd_handle(&internaleCmdHandle, cmdHandles, STP_MAX_CMD_HANDLE_COUNT);
59 if (allocStatus != kStatus_Success)
60 {
61 return kStatus_Fail;
62 }
63 if (cmdHandle != NULL)
64 {
65 CHD_add_cmd_handle_ref(internaleCmdHandle);
66 }
67
68 queueItem->cmdHandle = internaleCmdHandle;
69
70 status_t sendStatus = CDP_send_cmd(stpCmdQueue, queueItem, deadline, internaleCmdHandle);
71 if (sendStatus != kStatus_Success)
72 {
73 if (cmdHandle != NULL)
74 {
75 CHD_remove_cmd_handle_ref(*cmdHandle);
76 *cmdHandle = NULL;
77 }
78 return kStatus_Fail;
79 }
80 if (cmdHandle != NULL)
81 {
82 *cmdHandle = internaleCmdHandle;
83 }
84 return kStatus_Success;
85}

87void STPi_process(void)
88{
89 uint8_t nonIdleMotors = STPi_check_movement_completion();
90
91 STP_CmdQueueItem_t queueItem = {0};
92 BaseType_t status;
93 TickType_t delay = nonIdleMotors ? pdMS_TO_TICKS(2) : portMAX_DELAY;
94 status = xQueueReceive(stpCmdQueue, &queueItem, delay);
95 if (status != pdPASS)
96 {
97 return;
98 }
99 STPi_process_cmd(queueItem);
100}

102void STPi_process_cmd(STP_CmdQueueItem_t queueItem)
103{
104 static char logMsg[100];
105 switch (queueItem.type)
106 {
107 case STP_CMD_MOVE:
108 {
109 LOG_DEBUG("Processing movement command");
110 int32_t steps = queueItem.data.move.steps;
111 queueItem.handle->movementHandle.totalSteps = (steps >= 0) ? steps : -steps;
112 queueItem.handle->movementHandle.direction =
113 (steps >= 0) ? STP_COUNTERCLOCKWISE : STP_CLOCKWISE;
114 queueItem.handle->movementHandle.waitForStart = 0;
115 if (STPi_plan_motion_profile(queueItem.handle) == kStatus_Success)
116 {
117 if (start_steps(queueItem.handle, queueItem.cmdHandle) != kStatus_Success)
118 {
119 CDP_notify_task_failure(queueItem.cmdHandle);
120 CHD_remove_cmd_handle_ref(queueItem.cmdHandle);
121 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to start stepper execution",
122 queueItem.handle->label);
123 LOG_ERROR(logMsg);
124 }
125 }
126 else
127 {
128 CDP_notify_task_failure(queueItem.cmdHandle);
129 CHD_remove_cmd_handle_ref(queueItem.cmdHandle);
130 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to plan motion profile for %lu steps",
131 queueItem.handle->label, queueItem.handle->movementHandle.totalSteps);
132 LOG_ERROR(logMsg);
133 }
134
135 break;
136 }
137 case STP_CMD_MOVE_PREPARE:
138 {
139 LOG_DEBUG("Processing prepare movement command");
140 int32_t steps = queueItem.data.move.steps;
141 queueItem.handle->movementHandle.totalSteps = (steps >= 0) ? steps : -steps;
142 queueItem.handle->movementHandle.direction =
143 (steps >= 0) ? STP_COUNTERCLOCKWISE : STP_CLOCKWISE;
144 queueItem.handle->movementHandle.waitForStart = 1;
145 if (STPi_plan_motion_profile(queueItem.handle) == kStatus_Success)
146 {
147 CDP_notify_task_success(queueItem.cmdHandle);
148 }
149 else
150 {
151 CDP_notify_task_failure(queueItem.cmdHandle);
152 }
153 CHD_remove_cmd_handle_ref(queueItem.cmdHandle);
154 break;
155 }
156 case STP_CMD_TRIGGER_START:
157 {
158 LOG_DEBUG("Triggering synchronized start");
159
160 uint8_t anyStarted = 0;
161 uint8_t ownerSet = 0;
162
163 portENTER_CRITICAL();
164
165 for (size_t i = 0; i < STP_MAX_HANDLE_COUNT; i++)
166 {
167 if (handles[i].used == 1 &&
168 handles[i].handle.movementHandle.state == STP_MOVEMENT_PLANNED &&
169 handles[i].handle.movementHandle.waitForStart == 1)
170 {
171 handles[i].handle.movementHandle.waitForStart = 0;
172
173 CHD_CmdHandle_t h = ownerSet ? NULL : queueItem.cmdHandle;
174 if (start_steps(&handles[i].handle, h) == kStatus_Success)
175 {
176 anyStarted = 1;
177 if (!ownerSet)
178 {
179 ownerSet =
180 1; // only first successfully started motor owns trigger handle
181 }
182 }
183 }
184 }
185
186 portEXIT_CRITICAL();
187 if (!anyStarted)
188 {
189 CDP_notify_task_failure(queueItem.cmdHandle);
190 CHD_remove_cmd_handle_ref(queueItem.cmdHandle);
191 }
192 break;
193 }
194 case STP_CMD_STOP:
195 {
196 LOG_DEBUG("Processing stop command");
197 if (STPi_stop_steps(queueItem.handle, queueItem.data.stop.doDeceleration,
198 queueItem.cmdHandle) != kStatus_Success)
199 {
200 CDP_notify_task_failure(queueItem.cmdHandle);
201 CHD_remove_cmd_handle_ref(queueItem.cmdHandle);
202 LOG_ERROR("Failed to stop steps");
203 }
204 break;
205 }
206 case STP_CMD_INIT_HANDLE:
207 {
208 LOG_DEBUG("Processing init handle command");
209 if (TSU_init_handle(queueItem.data.config.config, queueItem.deadline) ==
210 kStatus_Success)
211 {
212 CDP_notify_task_success(queueItem.cmdHandle);
213 }
214 else
215 {
216 CDP_notify_task_failure(queueItem.cmdHandle);
217 }
218 CHD_remove_cmd_handle_ref(queueItem.cmdHandle);
219 break;
220 }
221 case STP_CMD_SET_ACCELERATION:
222 {
223 LOG_DEBUG("Processing set acceleration command");
224 queueItem.handle->acceleration = queueItem.data.setAcceleration.acceleration;
225 CDP_notify_task_success(queueItem.cmdHandle);
226 CHD_remove_cmd_handle_ref(queueItem.cmdHandle);
227 break;
228 }
229 case STP_CMD_SET_END_VELOCITY:
230 {
231 LOG_DEBUG("Processing set end velocity command");
232 queueItem.handle->endVelocity = queueItem.data.setEndVelocity.endVelocity;
233 CDP_notify_task_success(queueItem.cmdHandle);
234 CHD_remove_cmd_handle_ref(queueItem.cmdHandle);
235 break;
236 }
237 default:
238 {
239 CDP_notify_task_failure(queueItem.cmdHandle);
240 CHD_remove_cmd_handle_ref(queueItem.cmdHandle);
241 LOG_WARN("Received unknown command type");
242 break;
243 }
244 }
245}

247status_t STPi_stop_steps(STP_Handle_t handle, uint8_t doDeceleration, CHD_CmdHandle_t cmdHandle)
248{
249 if (handle == NULL)
250 {
251 return kStatus_Fail;
252 }
253
254 CHD_CmdHandle_t oldCmdHandle = NULL;
255
256 STP_MovementState_t st = handle->movementHandle.state;
257 if (st == STP_MOVEMENT_IDLE || st == STP_MOVEMENT_FINISHED || st == STP_MOVEMENT_STOPPED)
258 {
259 taskEXIT_CRITICAL();
260 return kStatus_Fail;
261 }
262
263 oldCmdHandle = handle->movementHandle.cmdHandle;
264
265 if (doDeceleration == 0)
266 {
267 handle->ftmBase->CONTROLS[handle->ftmChannel].CnSC &=
268 ~(FTM_CnSC_CHIE_MASK | FTM_CnSC_ELSA_MASK | FTM_CnSC_ELSB_MASK);
269 handle->ftmBase->CONTROLS[handle->ftmChannel].CnSC &= ~(FTM_CnSC_CHF_MASK);
270 handle->movementHandle.state = STP_MOVEMENT_STOPPED;
271 }
272 else
273 {
274 if (st == STP_MOVEMENT_PLANNED)
275 {
276 // nothing running yet -> stop immediately
277 handle->movementHandle.state = STP_MOVEMENT_STOPPED;
278 }
279 else if (st == STP_MOVEMENT_ACCELERATING)
280 {
281 handle->movementHandle.isTrapezoidal = 0;
282 handle->movementHandle.endVelocitySteps = 0;
283 handle->movementHandle.accelSteps = handle->movementHandle.currStepCount;
284 handle->movementHandle.decelSteps = handle->movementHandle.currStepCount;
285 handle->movementHandle.totalSteps = handle->movementHandle.currStepCount * 2;
286 }
287 else if (st == STP_MOVEMENT_CONST_VELOCITY)
288 {
289 handle->movementHandle.endVelocitySteps =
290 handle->movementHandle.currStepCount - handle->movementHandle.accelSteps;
291 handle->movementHandle.totalSteps = handle->movementHandle.accelSteps +
292 handle->movementHandle.endVelocitySteps +
293 handle->movementHandle.decelSteps;
294 }
295 // if already decelerating: keep as-is
296 }
297
298 handle->movementHandle.cmdHandle = cmdHandle;
299
300 if (oldCmdHandle != NULL && oldCmdHandle != cmdHandle)
301 {
302 CDP_notify_task_failure(oldCmdHandle);
303 CHD_remove_cmd_handle_ref(oldCmdHandle);
304 }
305
306 return kStatus_Success;
307}

313static status_t TSU_init_handle(const STP_StepperConfig_t config, TickType_t deadline)
314{
315 static char logMsg[90];
316
317 STP_Handle_t handle = NULL;
318 for (uint8_t i = 0; i < STP_MAX_HANDLE_COUNT; i++)
319 {
320 if (handles[i].used == 0)
321 {
322 handles[i].used = 1;
323 handle = &handles[i].handle;
324 break;
325 }
326 }
327
328 if (handle == NULL)
329 {
330 LOG_ERROR("Failed to init stepper handle. Too many handles.");
331
332 return kStatus_Fail;
333 }
334
335 handle->ftmBase = config.ftmBase;
336 handle->ftmChannel = config.ftmChannel;
337
338 handle->stepPort = config.stepPort;
339 handle->stepGPIO = config.stepGPIO;
340 handle->stepPin = config.stepPin;
341
342 handle->dirPort = config.dirPort;
343 handle->dirPin = config.dirPin;
344 handle->dirLogicHighClockwise = config.dirLogicHighClockwise;
345
346 handle->acceleration = config.acceleration;
347 handle->endVelocity = config.endVelocity;
348 handle->label = config.label;
349 handle->absolutePosition = 0;
350
351 if (STPi_reset_movement_handle(&handle->movementHandle) != kStatus_Success)
352 {
353 return kStatus_Fail;
354 }
355
356 if (init_step_pin(handle) != kStatus_Success)
357 {
358 return kStatus_Fail;
359 }
360
361 if (init_dir_pin(handle) != kStatus_Success)
362 {
363 return kStatus_Fail;
364 }
365
366 STPi_update_isr_handle_cache();
367
368 snprintf(logMsg, sizeof(logMsg),
369 "[%s] Stepper initialized successfully (accel=%.1f step/s², vel=%.1f step/s)",
370 handle->label, handle->acceleration, handle->endVelocity);
371 LOG_INFO(logMsg);
372
373 return kStatus_Success;
374}

376static status_t start_steps(STP_Handle_t handle, CHD_CmdHandle_t cmdHandle)
377{
378 SEGGER_SYSVIEW_Print("Starting Steps");
379 static char logMsg[100];
380
381 if (handle == NULL)
382 {
383 return kStatus_Fail;
384 }
385
386 if (handle->movementHandle.state != STP_MOVEMENT_PLANNED)
387 {
388 snprintf(logMsg, sizeof(logMsg),
389 "[%s] Cannot start movement - not in PLANNED state (current: %d)", handle->label,
390 handle->movementHandle.state);
391 LOG_ERROR(logMsg);
392 return kStatus_Fail;
393 }
394
395 handle->ftmBase->CONTROLS[handle->ftmChannel].CnSC &= ~FTM_CnSC_CHF_MASK;
396
397 // Use first value from lookup table
398 uint16_t initialDelay;
399 int8_t poolIndex = handle->movementHandle.accelTablePoolIndex;
400
401 if (poolIndex >= 0 && poolIndex < STP_ACCEL_TABLE_POOL_SIZE &&
402 accelTablePool[poolIndex].tableSize > 0)
403 {
404 initialDelay = accelTablePool[poolIndex].table[0];
405 }
406 else
407 {
408 snprintf(logMsg, sizeof(logMsg),
409 "[%s] Lookup table retrieve failed for %lu steps at index %d (pool %d)",
410 handle->label, handle->movementHandle.totalSteps, 0, poolIndex);
411 LOG_FATAL(logMsg);
412 return kStatus_Fail;
413 }
414 handle->ftmBase->CONTROLS[handle->ftmChannel].CnV = handle->ftmBase->CNT + initialDelay;
415 handle->ftmBase->CONTROLS[handle->ftmChannel].CnSC |=
416 FTM_CnSC_MSA_MASK | FTM_CnSC_ELSA_MASK | FTM_CnSC_CHIE_MASK;
417
418 handle->movementHandle.state = STP_MOVEMENT_STARTED;
419 handle->movementHandle.cmdHandle = cmdHandle;
420
421 snprintf(logMsg, sizeof(logMsg), "[%s] Movement started: %lu steps", handle->label,
422 handle->movementHandle.totalSteps);
423 LOG_DEBUG(logMsg);
424 return kStatus_Success;
425}
