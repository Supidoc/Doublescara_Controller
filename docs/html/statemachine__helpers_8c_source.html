55bool ExecuteCmd(CommandPackage pkg)
56{
57 switch (pkg.cmd)
58 {
59 case INIT:
60 return true;
61 case LED_ON:
62 LED_on();
63 return true;
64 case LED_OFF:
65 LED_off();
66 return true;
67 case GOTO_XY_ROTATE:
68 {
69 CHD_CmdHandle_t cmdHandle = NULL;
70 SK_Point_t targetPoint = {.x = pkg.x, .y = pkg.y};
71 if (SK_move_to_xy_async(targetPoint, pkg.rotate, true, portMAX_DELAY, &cmdHandle) !=
72 kStatus_Success)
73 {
74 return false;
75 }
76 if (SYW_cmd_wait_result(cmdHandle, portMAX_DELAY, NULL) != kStatus_Success)
77 {
78 return false;
79 }
80 return true;
81 }
82 case MAGNET_UP:
83 if (GRB_magnet_up() != kStatus_Success)
84 {
85 return false;
86 }
87 return true;
88 case MAGNET_DOWN:
89 if (GRB_magnet_down() != kStatus_Success)
90 {
91 return false;
92 }
93 return true;
94 case Z_UP:
95 if (GRB_platform_up() != kStatus_Success)
96 {
97 return false;
98 }
99 return true;
100 case Z_DOWN:
101 if (GRB_platform_down() != kStatus_Success)
102 {
103 return false;
104 }
105 return true;
106 case GRAB:
107 if (GRB_grab() != kStatus_Success)
108 {
109 return false;
110 }
111 return true;
112 case RELEASE:
113 if (GRB_release() != kStatus_Success)
114 {
115 return false;
116 }
117 return true;
118 case ERR:
119 return false;
120 case FINISH:
121 /* no-op */
122 break;
123 default:
124 return false;
125 }
126 return true;
127}

129bool ReadCmd(CommandPackage* pkg, const char* receivedMessage)
130{
131 size_t length = strlen(receivedMessage);
132
133 /* convert into 5 fields: cmd,x,y,rotate,transid */
134 char* fields[5];
135 int fieldCount = 0;
136 char* token = strtok((char*)receivedMessage, ",");
137
138 while (token != NULL && fieldCount < 5)
139 {
140 fields[fieldCount++] = token;
141 token = strtok(NULL, ",");
142 }
143
144 if (fieldCount != 5)
145 {
146 return false;
147 }
148
149 /* parse fields */
150 pkg->cmd = (uint8_t)atoi(fields[0]);
151 pkg->x = atof(fields[1]);
152 pkg->y = atof(fields[2]);
153 pkg->rotate = atof(fields[3]);
154 pkg->transID = (uint8_t)atoi(fields[4]);
155
156 return uart2IsValidCommand(pkg->cmd);
157}
