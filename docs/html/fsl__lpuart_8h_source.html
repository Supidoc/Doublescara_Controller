808static inline uint8_t LPUART_ReadByte(LPUART_Type *base)
809{
810#if defined(FSL_FEATURE_LPUART_HAS_7BIT_DATA_SUPPORT) && FSL_FEATURE_LPUART_HAS_7BIT_DATA_SUPPORT
811 uint32_t ctrl = base->CTRL;
812 uint8_t result;
813 /*
814 * $Branch Coverage Justification$
815 * (ctrl & LPUART_CTRL_M7_MASK) == 0U) false is not covered.
816 * If ctrl & LPUART_CTRL_M7_MASK is 0, it can't be !0 in next judge.
817 */
818 bool isSevenDataBits = (((ctrl & LPUART_CTRL_M7_MASK) != 0U) ||
819 (((ctrl & LPUART_CTRL_M7_MASK) == 0U) && ((ctrl & LPUART_CTRL_M_MASK) == 0U) &&
820 ((ctrl & LPUART_CTRL_PE_MASK) != 0U)));
821
822 if (isSevenDataBits)
823 {
824 result = (uint8_t)(base->DATA & 0x7FU);
825 }
826 else
827 {
828 result = (uint8_t)base->DATA;
829 }
830
831 return result;
832#else
833 return (uint8_t)(base->DATA);
834#endif
835}
