465static inline void RTC_Reset(RTC_Type *base)
466{
467 base->CR |= RTC_CR_SWR_MASK;
468 base->CR &= ~RTC_CR_SWR_MASK;
469
470 /*
471 * According to ERR008068, after POR, time invalid flag of RTC is set, that is, RTC_SR[TIF]=1.
472 * If this flag is not cleared, MCU fails to enter low power mode.
473 * Set TSR register to 0x1 to avoid the timer invalid (TIF) bit being set in the SR register.
474 */
475 base->TSR = 1U;
476}
