34status_t scara_inverse_kinematics(double x, double y, SK_Configuration_t configuration,
35 double* theta1, double* theta2)
36{
37
38 d1E = sqrt(pow(x, 2) + pow(y, 2));
39 d2E = sqrt(pow(x - c, 2) + pow(y, 2));
40 alpha1 = acos((pow(d1E, 2) + pow(a1, 2) - pow(a3, 2)) / (2 * a1 * d1E));
41 alpha2 = acos((pow(d2E, 2) + pow(a2, 2) - pow(a4, 2)) / (2 * a2 * d2E));
42 beta1 = atan2(y, x);
43 beta2 = atan2(y, x - c);
44
45 if (isnan(alpha1) || isnan(alpha2) || isinf(alpha1) || isinf(alpha2))
46 {
47 return kStatus_Fail; // Position is unreachable
48 }
49
50 if (isnan(beta1) || isnan(beta2) || isinf(beta1) || isinf(beta2))
51 {
52 return kStatus_Fail; // Position is unreachable
53 }
54
55 switch (configuration)
56 {
57 case SK_BOTH_ELBOW_OUT:
58 theta1Temp = beta1 - alpha1;
59 theta2Temp = beta2 - alpha2;
60 break;
61 case SK_L_ELBOW_OUT_R_ELBOW_IN:
62 theta1Temp = beta1 - alpha1;
63 theta2Temp = beta2 + alpha2;
64 break;
65 case SK_L_ELBOW_IN_R_ELBOW_OUT:
66 theta1Temp = beta1 + alpha1;
67 theta2Temp = beta2 - alpha2;
68 break;
69 case SK_BOTH_ELBOW_IN:
70 theta1Temp = beta1 + alpha1;
71 theta2Temp = beta2 + alpha2;
72 break;
73 }
74
75 *theta1 = theta1Temp;
76 *theta2 = theta2Temp;
77
78 return kStatus_Success;
79}
