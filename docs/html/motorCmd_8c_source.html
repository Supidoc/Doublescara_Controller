224static status_t wait_for_mtr_cmd(THE_CmdHandle_t cmdHandle, TickType_t deadline)
225{
226 if (cmdHandle == NULL)
227 {
228 return kStatus_Fail;
229 }
230
231 status_t status = THE_cmd_wait_result(cmdHandle, deadline, NULL);
232 THE_remove_cmd_handle_ref(cmdHandle);
233 return status;
234}

236static BaseType_t cmd_motor_move_angle(char* pcWriteBuffer, size_t xWriteBufferLen,
237 const char* pcCommandString)
238{
239 MTR_MotorHandle_t handle = NULL;
240 uint8_t parameterFound;
241 const char* pcParameter;
242 BaseType_t parameterStringLength;
243 double angle;
244
245 // Get motor ID
246 CLU_get_parameter_value_string("-m", pcCommandString, &parameterFound, &pcParameter,
247 &parameterStringLength);
248 if (!parameterFound || parameterStringLength == 0)
249 {
250 snprintf(pcWriteBuffer, xWriteBufferLen, "Error: Motor ID required (-m)\r\n");
251 return pdFALSE;
252 }
253
254 MTR_get_motor_by_label(pcParameter, &handle);
255 if (handle == NULL)
256 {
257 snprintf(pcWriteBuffer, xWriteBufferLen, "Error: Motor not found\r\n");
258 return pdFALSE;
259 }
260
261 // Get angle
262 CLU_get_parameter_value_string("-a", pcCommandString, &parameterFound, &pcParameter,
263 &parameterStringLength);
264 if (!parameterFound || parameterStringLength == 0)
265 {
266 snprintf(pcWriteBuffer, xWriteBufferLen, "Error: Angle required (-a)\r\n");
267 return pdFALSE;
268 }
269 angle = strtod(pcParameter, NULL);
270
271 if (MTR_move_angle_async(handle, angle, portMAX_DELAY, NULL) == kStatus_Success)
272 {
273 int32_t stepCount = MTR_angle_to_steps(handle, angle, ROUND_NEAREST);
274 double realAngle = MTR_steps_to_angle(handle, stepCount);
275 snprintf(pcWriteBuffer, xWriteBufferLen, "Moving %.2f degrees\r\n", realAngle);
276 }
277 else
278 {
279 snprintf(pcWriteBuffer, xWriteBufferLen, "Error: Move failed\r\n");
280 }
281
282 return pdFALSE;
283}
