245static status_t wait_for_mtr_cmd(CHD_CmdHandle_t cmdHandle, TickType_t deadline)
246{
247 if (cmdHandle == NULL)
248 {
249 return kStatus_Fail;
250 }
251
252 status_t status = SYW_cmd_wait_result(cmdHandle, deadline, NULL);
253 CHD_remove_cmd_handle_ref(cmdHandle);
254 return status;
255}

257static BaseType_t cmd_motor_move_angle(char* pcWriteBuffer, size_t xWriteBufferLen,
258 const char* pcCommandString)
259{
260 MTR_MotorHandle_t handle = NULL;
261 uint8_t parameterFound;
262 const char* pcParameter;
263 BaseType_t parameterStringLength;
264 double angle;
265
266 // Get motor ID
267 CLU_get_parameter_value_string("-m", pcCommandString, &parameterFound, &pcParameter,
268 &parameterStringLength);
269 if (!parameterFound || parameterStringLength == 0)
270 {
271 snprintf(pcWriteBuffer, xWriteBufferLen, "Error: Motor ID required (-m)\r\n");
272 return pdFALSE;
273 }
274
275 MTR_get_motor_by_label(pcParameter, &handle);
276 if (handle == NULL)
277 {
278 snprintf(pcWriteBuffer, xWriteBufferLen, "Error: Motor not found\r\n");
279 return pdFALSE;
280 }
281
282 // Get angle
283 CLU_get_parameter_value_string("-a", pcCommandString, &parameterFound, &pcParameter,
284 &parameterStringLength);
285 if (!parameterFound || parameterStringLength == 0)
286 {
287 snprintf(pcWriteBuffer, xWriteBufferLen, "Error: Angle required (-a)\r\n");
288 return pdFALSE;
289 }
290 angle = strtod(pcParameter, NULL);
291
292 if (MTR_move_angle_async(handle, angle, portMAX_DELAY, NULL) == kStatus_Success)
293 {
294 int32_t stepCount = MTR_angle_to_steps(handle, angle, ROUND_NEAREST);
295 double realAngle = MTR_steps_to_angle(handle, stepCount);
296 snprintf(pcWriteBuffer, xWriteBufferLen, "Moving %.2f degrees\r\n", realAngle);
297 }
298 else
299 {
300 snprintf(pcWriteBuffer, xWriteBufferLen, "Error: Move failed\r\n");
301 }
302
303 return pdFALSE;
304}

552 }
