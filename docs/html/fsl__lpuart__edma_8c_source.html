85static void LPUART_SendEDMACallback(edma_handle_t *handle, void *param, bool transferDone, uint32_t tcds)
86{
87 assert(NULL != param);
88
89 lpuart_edma_private_handle_t *lpuartPrivateHandle = (lpuart_edma_private_handle_t *)param;
90
91 /* Avoid the warning for unused variables. */
92 handle = handle;
93 tcds = tcds;
94 /*
95 * $Branch Coverage Justification$
96 * $ref fsl_lpuart_edma_c_ref_1$
97 */
98 if (transferDone)
99 {
100 /* Disable LPUART TX EDMA. */
101 LPUART_EnableTxDMA(lpuartPrivateHandle->base, false);
102
103 /* Stop transfer. */
104 EDMA_AbortTransfer(handle);
105
106 /* Enable tx complete interrupt */
107 LPUART_EnableInterrupts(lpuartPrivateHandle->base, (uint32_t)kLPUART_TransmissionCompleteInterruptEnable);
108 }
109}

111static void LPUART_ReceiveEDMACallback(edma_handle_t *handle, void *param, bool transferDone, uint32_t tcds)
112{
113 assert(NULL != param);
114
115 lpuart_edma_private_handle_t *lpuartPrivateHandle = (lpuart_edma_private_handle_t *)param;
116
117 /* Avoid warning for unused parameters. */
118 handle = handle;
119 tcds = tcds;
120 /*
121 * $Branch Coverage Justification$
122 * $ref fsl_lpuart_edma_c_ref_1$
123 */
124 if (transferDone)
125 {
126 /* Disable transfer. */
127 LPUART_TransferAbortReceiveEDMA(lpuartPrivateHandle->base, lpuartPrivateHandle->handle);
128
129 if (NULL != lpuartPrivateHandle->handle->callback)
130 {
131 lpuartPrivateHandle->handle->callback(lpuartPrivateHandle->base, lpuartPrivateHandle->handle,
132 kStatus_LPUART_RxIdle, lpuartPrivateHandle->handle->userData);
133 }
134 }

149void LPUART_TransferCreateHandleEDMA(LPUART_Type *base,
150 lpuart_edma_handle_t *handle,
151 lpuart_edma_transfer_callback_t callback,
152 void *userData,
153 edma_handle_t *txEdmaHandle,
154 edma_handle_t *rxEdmaHandle)
155{
156 assert(NULL != handle);
157
158 uint32_t instance = LPUART_GetInstance(base);
159
160 s_lpuartEdmaPrivateHandle[instance].base = base;
161 s_lpuartEdmaPrivateHandle[instance].handle = handle;
162
163 (void)memset(handle, 0, sizeof(*handle));
164
165 handle->rxState = (uint8_t)kLPUART_RxIdle;
166 handle->txState = (uint8_t)kLPUART_TxIdle;
167
168 handle->rxEdmaHandle = rxEdmaHandle;
169 handle->txEdmaHandle = txEdmaHandle;
170
171 handle->callback = callback;
172 handle->userData = userData;
173
174#if defined(FSL_FEATURE_LPUART_HAS_FIFO) && FSL_FEATURE_LPUART_HAS_FIFO
175 /* Note:
176 Take care of the RX FIFO, EDMA request only assert when received bytes
177 equal or more than RX water mark, there is potential issue if RX water
178 mark larger than 1.
179 For example, if RX FIFO water mark is 2, upper layer needs 5 bytes and
180 5 bytes are received. the last byte will be saved in FIFO but not trigger
181 EDMA transfer because the water mark is 2.
182 */
183 if (NULL != rxEdmaHandle)
184 {
185 base->WATER &= (~LPUART_WATER_RXWATER_MASK);
186 }
187#endif
188
189 /* Save the handle in global variables to support the double weak mechanism. */
190 s_lpuartHandle[instance] = handle;
191 /* Set LPUART_TransferEdmaHandleIRQ as DMA IRQ handler */
192 s_lpuartIsr[instance] = LPUART_TransferEdmaHandleIRQ;
193 /* Disable all LPUART internal interrupts */
194 LPUART_DisableInterrupts(base, (uint32_t)kLPUART_AllInterruptEnable);
195 /* Enable interrupt in NVIC. */
196#if defined(FSL_FEATURE_LPUART_HAS_SEPARATE_RX_TX_IRQ) && FSL_FEATURE_LPUART_HAS_SEPARATE_RX_TX_IRQ
197 (void)EnableIRQ(s_lpuartTxIRQ[instance]);
198#else
199 (void)EnableIRQ(s_lpuartIRQ[instance]);
200#endif
201
202 /* Configure TX. */
203 if (NULL != txEdmaHandle)
204 {
205 EDMA_SetCallback(handle->txEdmaHandle, LPUART_SendEDMACallback, &s_lpuartEdmaPrivateHandle[instance]);
206 }
207
208 /* Configure RX. */
209 if (NULL != rxEdmaHandle)
210 {
211 EDMA_SetCallback(handle->rxEdmaHandle, LPUART_ReceiveEDMACallback, &s_lpuartEdmaPrivateHandle[instance]);
212 }
213}

228status_t LPUART_SendEDMA(LPUART_Type *base, lpuart_edma_handle_t *handle, lpuart_transfer_t *xfer)
229{
230 assert(NULL != handle);
231 assert(NULL != handle->txEdmaHandle);
232 assert(NULL != xfer);
233 assert(NULL != xfer->data);
234 assert(0U != xfer->dataSize);
235
236 edma_transfer_config_t xferConfig;
237 status_t status;
238
239 /* If previous TX not finished. */
240 if ((uint8_t)kLPUART_TxBusy == handle->txState)
241 {
242 status = kStatus_LPUART_TxBusy;
243 }
244 else
245 {
246 handle->txState = (uint8_t)kLPUART_TxBusy;
247 handle->txDataSizeAll = xfer->dataSize;
248
249 /* Prepare transfer. */
250 EDMA_PrepareTransfer(&xferConfig, xfer->data, sizeof(uint8_t),
251 (void *)(uint32_t *)LPUART_GetDataRegisterAddress(base), sizeof(uint8_t), sizeof(uint8_t),
252 xfer->dataSize, kEDMA_MemoryToPeripheral);
253
254 /* Store the initially configured eDMA minor byte transfer count into the LPUART handle */
255 handle->nbytes = (uint8_t)sizeof(uint8_t);
256
257 /* Submit transfer. */
258 if (kStatus_Success !=
259 EDMA_SubmitTransfer(handle->txEdmaHandle, (const edma_transfer_config_t *)(uint32_t)&xferConfig))
260 {
261 return kStatus_Fail;
262 }
263 EDMA_StartTransfer(handle->txEdmaHandle);
264
265 /* Enable LPUART TX EDMA. */
266 LPUART_EnableTxDMA(base, true);
267
268 status = kStatus_Success;
269 }
270
271 return status;
272}

287status_t LPUART_ReceiveEDMA(LPUART_Type *base, lpuart_edma_handle_t *handle, lpuart_transfer_t *xfer)
288{
289 assert(NULL != handle);
290 assert(NULL != handle->rxEdmaHandle);
291 assert(NULL != xfer);
292 assert(NULL != xfer->data);
293 assert(0U != xfer->dataSize);
294
295 edma_transfer_config_t xferConfig;
296 status_t status;
297
298 /* If previous RX not finished. */
299 if ((uint8_t)kLPUART_RxBusy == handle->rxState)
300 {
301 status = kStatus_LPUART_RxBusy;
302 }
303 else
304 {
305 handle->rxState = (uint8_t)kLPUART_RxBusy;
306 handle->rxDataSizeAll = xfer->dataSize;
307
308 /* Prepare transfer. */
309 EDMA_PrepareTransfer(&xferConfig, (void *)(uint32_t *)LPUART_GetDataRegisterAddress(base), sizeof(uint8_t),
310 xfer->data, sizeof(uint8_t), sizeof(uint8_t), xfer->dataSize, kEDMA_PeripheralToMemory);
311
312 /* Store the initially configured eDMA minor byte transfer count into the LPUART handle */
313 handle->nbytes = (uint8_t)sizeof(uint8_t);
314
315 /* Submit transfer. */
316 if (kStatus_Success !=
317 EDMA_SubmitTransfer(handle->rxEdmaHandle, (const edma_transfer_config_t *)(uint32_t)&xferConfig))
318 {
319 return kStatus_Fail;
320 }
321 EDMA_StartTransfer(handle->rxEdmaHandle);
322
323 /* Enable LPUART RX EDMA. */
324 LPUART_EnableRxDMA(base, true);
325
326 status = kStatus_Success;
327 }
328
329 return status;

388status_t LPUART_TransferGetReceiveCountEDMA(LPUART_Type *base, lpuart_edma_handle_t *handle, uint32_t *count)
389{
390 assert(NULL != handle);
391 assert(NULL != handle->rxEdmaHandle);
392 assert(NULL != count);
393
394 if ((uint8_t)kLPUART_RxIdle == handle->rxState)

396 return kStatus_NoTransferInProgress;

448void LPUART_TransferEdmaHandleIRQ(LPUART_Type *base, void *lpuartEdmaHandle)
449{
450 assert(lpuartEdmaHandle != NULL);
451 /*
452 * $Branch Coverage Justification$
453 * $ref fsl_lpuart_edma_c_ref_2$
454 */
455 if (((uint32_t)kLPUART_TransmissionCompleteFlag & LPUART_GetStatusFlags(base)) != 0U)
456 {
457 lpuart_edma_handle_t *handle = (lpuart_edma_handle_t *)lpuartEdmaHandle;
458
459 /* Disable tx complete interrupt */
460 LPUART_DisableInterrupts(base, (uint32_t)kLPUART_TransmissionCompleteInterruptEnable);
461
462 handle->txState = (uint8_t)kLPUART_TxIdle;
463
464 if (handle->callback != NULL)
465 {
466 handle->callback(base, handle, kStatus_LPUART_TxIdle, handle->userData);
467 }
468 }
