51status_t STP_init()
52{
53 // Initialize acceleration table pool
54 for (int8_t i = 0; i < STP_ACCEL_TABLE_POOL_SIZE; i++)
55 {
56 accelTablePool[i].used = 0;
57 accelTablePool[i].tableSize = 0;
58 }
59
60 stpCmdQueue = xQueueCreate(STP_CMD_QUEUE_SIZE, sizeof(STP_CmdQueueItem_t));
61 if (stpCmdQueue == NULL)
62 {
63 return kStatus_Fail;
64 }
65
66 vQueueAddToRegistry(stpCmdQueue, "STP Command Queue");
67
68 SIM->SCGC6 |= SIM_SCGC6_FTM3_MASK;
69
70 NVIC_SetPriority(FTM3_IRQn, 2);
71 NVIC_EnableIRQ(FTM3_IRQn);
72
73 FTM3->MOD = 0xFFFF;
74 FTM3->SC = FTM_SC_CLKS(2) | FTM_SC_PS(0);
75
76 CHD_init_cmd_handles(stpCmdHandles, STP_MAX_CMD_HANDLE_COUNT);
77
78 return kStatus_Success;
79}

140status_t STP_get_handle_by_label(const char* label, STP_Handle_t* handle)
141{
142 if (label == NULL || handle == NULL)
143 {
144 return kStatus_Fail;
145 }
146 for (uint8_t i = 0; i < STP_MAX_HANDLE_COUNT; i++)
147 {
148 if (stpHandles[i].used)
149 {
150 uint32_t handleLength = strlen(stpHandles[i].handle.label);
151
152 if (strncmp(label, stpHandles[i].handle.label, handleLength) == 0)
153 {
154 *handle = &stpHandles[i].handle;
155 return kStatus_Success;
156 }
157 }
158 }
159 return kStatus_Fail;
160}
