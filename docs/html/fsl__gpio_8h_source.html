345static inline void GPIO_PinWrite(GPIO_Type *base, uint32_t pin, uint8_t output)
346{
347#if !(defined(FSL_FEATURE_GPIO_HAS_NO_INDEP_OUTPUT_CONTROL) && FSL_FEATURE_GPIO_HAS_NO_INDEP_OUTPUT_CONTROL)
348 if (output == 0U)
349 {
350 base->PCOR = GPIO_FIT_REG(1UL << pin);
351 }
352 else
353 {
354 base->PSOR = GPIO_FIT_REG(1UL << pin);
355 }
356#else
357 if (output == 0U)
358 {
359 base->PDOR |= GPIO_FIT_REG(1UL << pin);
360 }
361 else
362 {
363 base->PDOR &= ~GPIO_FIT_REG(1UL << pin);
364 }
365#endif
366}
