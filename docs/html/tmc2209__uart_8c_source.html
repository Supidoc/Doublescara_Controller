64status_t TMCi_read(TMC_Handle_t handle, uint8_t regAddr, uint32_t* data, TickType_t deadline)
65{
66 if (handle == NULL || data == NULL)
67 {
68 return kStatus_Fail;
69 }
70
71 status_t status;
72 TMC_Read_Datagram_t readDatagram;
73 char logMsg[60];
74
75 create_read_datagram(handle, regAddr, &readDatagram);
76 build_read_package(&readDatagram, transmit_read_package);
77 TickType_t currentTick = xTaskGetTickCount();
78 if (deadline <= currentTick)
79 {
80 snprintf(logMsg, sizeof(logMsg), "[%s] Deadline for UART receive has already passed",
81 handle->label);
82 LOG_ERROR(logMsg);
83 return kStatus_Fail;
84 }
85 TickType_t ticksUntilDeadline = deadline - currentTick;
86 PORT_SetPinMux(PORTE, 0, kPORT_MuxAlt3);
87 status = UART_RTOS_Send(handle->uartRTOSHandle, transmit_read_package, TMC_READ_PACKAGE_SIZE);
88 PORT_SetPinMux(PORTE, 0, kPORT_MuxAlt1);
89 dump_rx_buffer(handle, TMC_READ_PACKAGE_SIZE, deadline);
90 if (status != kStatus_Success)
91 {
92 char logMsg[60];
93 snprintf(logMsg, sizeof(logMsg), "[%s] UART send failed during read", handle->label);
94 LOG_ERROR(logMsg);
95 return status;
96 }
97
98 status = UART_RTOS_Receive(handle->uartRTOSHandle, transmit_response_package,
99 TMC_WRITE_PACKAGE_SIZE, NULL, ticksUntilDeadline);
100 if (status != kStatus_Success)
101 {
102 snprintf(logMsg, sizeof(logMsg), "[%s] UART receive timeout during read", handle->label);
103 LOG_WARN(logMsg);
104 return status;
105 }
106
107 calc_CRC(transmit_response_package, TMC_WRITE_PACKAGE_SIZE);
108
109 *data = (transmit_response_package[3] << 24) | (transmit_response_package[4] << 16) |
110 (transmit_response_package[5] << 8) | transmit_response_package[6];
111
112 /* Clear response buffer */
113 for (size_t i = 0; i < TMC_WRITE_PACKAGE_SIZE; i++)
114 {
115 transmit_response_package[i] = 0;
116 }
117
118 return kStatus_Success;
119}

121status_t TMCi_write(TMC_Handle_t handle, uint8_t regAddr, uint32_t* data, TickType_t deadline)
122{
123 if (handle == NULL || data == NULL)
124 {
125 return kStatus_Fail;
126 }
127
128#if TMC_CONFIRM_WRITES == (1)
129 uint8_t expectedTransmissionCount = handle->transmissionCount + 1;
130#endif
131
132 status_t status;
133 TMC_Write_Datagram_t writeDatagram;
134
135 create_write_datagram(handle, regAddr, *data, &writeDatagram);
136 build_write_package(&writeDatagram, transmit_write_package);
137 PORT_SetPinMux(PORTE, 0, kPORT_MuxAlt3);
138 status = UART_RTOS_Send(handle->uartRTOSHandle, transmit_write_package, TMC_WRITE_PACKAGE_SIZE);
139 PORT_SetPinMux(PORTE, 0, kPORT_MuxAlt1);
140 dump_rx_buffer(handle, TMC_WRITE_PACKAGE_SIZE, deadline);
141 if (status != kStatus_Success)
142 {
143 char logMsg[60];
144 snprintf(logMsg, sizeof(logMsg), "[%s] UART send failed during write", handle->label);
145 LOG_ERROR(logMsg);
146 return status;
147 }
148
149#if TMC_CONFIRM_WRITES == (1)
150 status = TMCi_read_transmissioncount(handle, deadline);
151 if (status != kStatus_Success)
152 {
153 char logMsg[70];
154 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to confirm write (transmissioncount)",
155 handle->label);
156 LOG_WARN(logMsg);
157 return status;
158 }
159
160 if (expectedTransmissionCount != handle->transmissionCount)
161 {
162 char logMsg[80];
163 snprintf(logMsg, sizeof(logMsg), "[%s] Write confirmation mismatch: exp=%d got=%d",
164 handle->label, expectedTransmissionCount, handle->transmissionCount);
165 LOG_ERROR(logMsg);
166 return kStatus_Fail;
167 }
168#endif
169
170 return kStatus_Success;
171}

198 TMC_Write_Datagram_t* datagram)
199{
200 if (handle == NULL || datagram == NULL)
201 {
202 return;
203 }
204
205 datagram->sync = 0b0101;
206 datagram->reserved = 0b0000;
207 datagram->dev_addr = handle->serialAdress & 0xFF;
208 datagram->reg_addr = (reg_addr | 0x80) & 0xFF;
209 datagram->data = data & 0xFFFFFFFF;

226static inline void calc_CRC(uint8_t* datagram, size_t datagramLength)
227{
228 if (datagram == NULL || datagramLength == 0)
229 {
230 return;
231 }
232
233 uint8_t i;
234 uint8_t j;
235 uint8_t* crc = datagram + (datagramLength - 1); // CRC located in last byte of message
236 uint8_t currentByte;
237 *crc = 0;
238 for (i = 0; i < (datagramLength - 1); i++)
239 { // Execute for all bytes of a message
240 currentByte = datagram[i]; // Retrieve a byte to be sent from Array
241 for (j = 0; j < 8; j++)
242 {
243 if ((*crc >> 7) ^ (currentByte & 0x01)) // update CRC based result of XOR operation
244 {
245 *crc = (*crc << 1) ^ 0x07;
246 }
247 else
248 {
249 *crc = (*crc << 1);
250 }
251 currentByte = currentByte >> 1;
252 } // for CRC bit
253 } // for message byte
254}

256static inline void build_read_package(TMC_Read_Datagram_t* datagram, uint8_t* package)
257{
258 if (datagram == NULL || package == NULL)
259 {
260 return;
261 }
262
263 package[0] = ((datagram->reserved << 4) & 0xF0) | (datagram->sync & 0x0F);
264 package[1] = datagram->dev_addr & 0xFF;
265 package[2] = datagram->reg_addr & 0xFF;
266 package[3] = datagram->crc & 0xFF;
267 calc_CRC(package, TMC_READ_PACKAGE_SIZE);
268}

270static inline void build_write_package(TMC_Write_Datagram_t* datagram, uint8_t* package)
271{
272 if (datagram == NULL || package == NULL)
273 {
274 return;
275 }
276
277 package[0] = ((datagram->reserved << 4) & 0xF0) | (datagram->sync & 0x0F);
278 package[1] = datagram->dev_addr & 0xFF;
279 package[2] = datagram->reg_addr & 0xFF;
280 package[3] = (datagram->data >> 24) & 0xFF;
281 package[4] = (datagram->data >> 16) & 0xFF;
282 package[5] = (datagram->data >> 8) & 0xFF;
283 package[6] = datagram->data & 0xFF;
284 package[7] = datagram->crc & 0xFF;
285 calc_CRC(package, TMC_WRITE_PACKAGE_SIZE);

312
