110void MHM_homing_arm_interrupt_handler(PORT_Type* port, uint32_t pin)
111{
112 MTR_MotorHandle_t currentHomingHandle = (MTR_MotorHandle_t)homingHandle;
113
114 if (currentHomingHandle == NULL)
115 {
116 return;
117 }
118
119 if ((port == l_arm_home_port && pin == l_arm_home_pin) && currentHomingHandle == l_arm_handle)
120 {
121 if (STP_homing_stop(currentHomingHandle->stepperHandle) != kStatus_Success)
122 {
123 return;
124 }
125 }
126 else if ((port == r_arm_home_port && pin == r_arm_home_pin) &&
127 currentHomingHandle == r_arm_handle)
128 {
129 if (STP_homing_stop(currentHomingHandle->stepperHandle) != kStatus_Success)
130 {
131 return;
132 }
133 }
134 else
135 {
136 return;
137 }
138
139 homingSuccessful = 1;
140 homingHandle = NULL;
141}

162static status_t auto_homing(MTR_MotorHandle_t handle, STP_Direction_t direction, uint16_t maxAngle,
163 TickType_t deadline)
164{
165 status_t status = kStatus_Fail;
166 BaseType_t mutexLocked = pdFALSE;
167 static char logMsg[100];
168
169 if (emergencyStopFlag)
170 {
171 return kStatus_Fail;
172 }
173 if (handle == NULL)
174 {
175 return kStatus_Fail;
176 }
177
178 if (homingMutex == NULL && init_homing() != kStatus_Success)
179 {
180 return kStatus_Fail;
181 }
182
183 if (xSemaphoreTake(homingMutex, portMAX_DELAY) != pdTRUE)
184 {
185 return kStatus_Fail;
186 }
187
188 mutexLocked = pdTRUE;
189
190 snprintf(logMsg, sizeof(logMsg), "[%s] Starting auto-homing procedure", handle->label);
191 LOG_INFO(logMsg);
192
193 homingSuccessful = 0;
194 homingHandle = handle;
195
196 do
197 {
198 int32_t steps = -MTR_angle_to_steps(
199 handle, (direction == STP_COUNTERCLOCKWISE ? maxAngle : -maxAngle), ROUND_UP);
200
201 CHD_CmdHandle_t cmdHandle = NULL;
202 if (STP_move_relative_async(handle->stepperHandle, steps, deadline, &cmdHandle) !=
203 kStatus_Success)
204 {
205 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to queue auto-homing move command",
206 handle->label);
207 LOG_ERROR(logMsg);
208 break;
209 }
210
211 if (wait_for_cmd_handle(cmdHandle, deadline) == kStatus_Success)
212 {
213 snprintf(logMsg, sizeof(logMsg), "[%s] Auto-homing reached max angle", handle->label);
214 LOG_WARN(logMsg);
215 break;
216 }
217
218 if (!homingSuccessful)
219 {
220 snprintf(logMsg, sizeof(logMsg), "[%s] Auto-homing failed", handle->label);
221 LOG_ERROR(logMsg);
222 break;
223 }
224
225 if (MTR_set_home_position(handle) != kStatus_Success)
226 {
227 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to set home position after auto-homing",
228 handle->label);
229 LOG_ERROR(logMsg);
230 break;
231 }
232
233 status = kStatus_Success;
234 } while (0);
235
236 homingHandle = NULL;
237
238 if (mutexLocked == pdTRUE)
239 {
240 xSemaphoreGive(homingMutex);
241 }
242
243 if (status == kStatus_Success)
244 {
245 snprintf(logMsg, sizeof(logMsg), "[%s] Auto-homing procedure completed successfully",
246 handle->label);
247 LOG_INFO(logMsg);
248 }
249
250 return status;
251}
