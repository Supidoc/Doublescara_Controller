38status_t CDP_send_cmd(QueueHandle_t xQueue, const void* const pvItemToQueue, TickType_t deadline,
39 CHD_CmdHandle_t cmdHandle)
40{
41 TickType_t currentTick = xTaskGetTickCount();
42
43 CHD_add_cmd_handle_ref(cmdHandle); // Increment reference count for the receiving task
44
45 if (deadline <= currentTick)
46 {
47 LOG_WARN("Command deadline has already passed");
48 CDP_notify_task_timeout(cmdHandle);
49 CHD_remove_cmd_handle_ref(cmdHandle);
50 return kStatus_Fail;
51 }
52
53 TickType_t ticksUntilDeadline = deadline - currentTick;
54
55 if (xQueueSend(xQueue, pvItemToQueue, ticksUntilDeadline) != pdTRUE)
56 {
57 LOG_ERROR("Failed to queue command - queue may be full");
58 CDP_notify_task_timeout(cmdHandle);
59 CHD_remove_cmd_handle_ref(cmdHandle);
60 return kStatus_Fail;
61 }
62 return kStatus_Success;
63}
