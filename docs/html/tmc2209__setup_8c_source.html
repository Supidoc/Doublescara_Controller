49status_t TMCi_init_handle(TMC_Config_t config, TickType_t deadline)
50{
51 TMC_Handle_t handle = NULL;
52
53 for (size_t i = 0; i < TMC_MAX_HANDLE_COUNT; i++)
54 {
55 if (tmcHandleArray[i].used == 0)
56 {
57 tmcHandleArray[i].used = 1;
58 handle = &tmcHandleArray[i].handle;
59 break;
60 }
61 }
62
63 if (handle == NULL)
64 {
65 return kStatus_Fail;
66 }
67
68 handle->uartRTOSHandle = config.uartRTOSHandle;
69 handle->uartHandle = config.uartHandle;
70 handle->serialAdress = config.serialAdress;
71 handle->microstepping = config.microstepping;
72 handle->iholdDivider = config.iholdDivider;
73 handle->irunDivider = config.irunDivider;
74 handle->label = config.label;
75 handle->freewheeling = 0;
76
77 static char logMsg[100];
78 snprintf(logMsg, sizeof(logMsg), "[%s] Initializing handle with config", handle->label);
79 LOG_DEBUG(logMsg);
80
81 if (TMCi_read_transmissioncount(handle, deadline) != kStatus_Success)
82 {
83 TMCi_free_handle(handle);
84
85 snprintf(logMsg, sizeof(logMsg),
86 "[%s] Failed to read transmission count - cannot communicate with driver",
87 handle->label);
88 LOG_FATAL(logMsg);
89
90 return kStatus_Fail;
91 }
92
93 // gconf default values
94 uint32_t gconf = 0;
95 gconf |= 0b1; // use external VREF
96 gconf |= 0b0 << 1; // use external sense resistors
97 gconf |= 0b0 << 2; // use StealthChopfp PWM
98 gconf |= 0b0 << 3; // Do not invert Shaft
99 gconf |= 0b0 << 4; // Index Pin shows first microstep position of sequencer
100 gconf |= 0b0 << 5; // Output Index of first microstep
101 gconf |= 0b1 << 6; // PDN_UART input disabled, use uart
102 gconf |= 0b1 << 7; // Microstep resolution set by register
103 gconf |= 0b1 << 8; // No filtering of STEP pulses
104
105 if (kStatus_Success != TMCi_write(handle, TMC_GCONF_ADDR, &gconf, deadline))
106 {
107 TMCi_free_handle(handle);
108
109 snprintf(logMsg, sizeof(logMsg),
110 "[%s] Failed to write GCONF register - initialization failure", handle->label);
111 LOG_FATAL(logMsg);
112
113 return kStatus_Fail;
114 }
115
116 // reset GSTAT
117 uint32_t gstat = 0b111;
118
119 if (kStatus_Success != TMCi_write(handle, TMC_GSTAT_ADDR, &gstat, deadline))
120 {
121 TMCi_free_handle(handle);
122
123 snprintf(logMsg, sizeof(logMsg),
124 "[%s] Failed to write GSTAT register - initialization failure", handle->label);
125 LOG_FATAL(logMsg);
126
127 return kStatus_Fail;
128 }
129 if (TMCi_set_microstepping(handle, config.microstepping, deadline) != kStatus_Success)
130 {
131 TMCi_free_handle(handle);
132
133 snprintf(logMsg, sizeof(logMsg),
134 "[%s] Failed to set microstepping - initialization failure", handle->label);
135 LOG_FATAL(logMsg);
136
137 return kStatus_Fail;
138 }
139
140 if (TMCi_set_double_edge_step_pulse(handle, 1, deadline) != kStatus_Success)
141 {
142 TMCi_free_handle(handle);
143
144 snprintf(logMsg, sizeof(logMsg),
145 "[%s] Failed to set double edge step pulse - initialization failure",
146 handle->label);
147 LOG_FATAL(logMsg);
148
149 return kStatus_Fail;
150 }
151
152 if (TMCi_set_irun_divider(handle, config.irunDivider, deadline) != kStatus_Success)
153 {
154 TMCi_free_handle(handle);
155
156 snprintf(logMsg, sizeof(logMsg),
157 "[%s] Failed to set run current divider - initialization failure", handle->label);
158 LOG_FATAL(logMsg);
159
160 return kStatus_Fail;
161 }
162
163 if (TMCi_set_ihold_divider(handle, config.iholdDivider, deadline) != kStatus_Success)
164 {
165 TMCi_free_handle(handle);
166
167 snprintf(logMsg, sizeof(logMsg),
168 "[%s] Failed to set hold current divider - initialization failure", handle->label);
169 LOG_FATAL(logMsg);
170
171 return kStatus_Fail;
172 }
173
174 snprintf(logMsg, sizeof(logMsg), "[%s] TMC2209 initialized successfully", handle->label);
175 LOG_DEBUG(logMsg);
176
177 return kStatus_Success;
178}

180status_t TMCi_free_handle(TMC_Handle_t handle)
181{
182 if (handle == NULL)
183 {
184 return kStatus_Fail;
185 }
186
187 for (size_t i = 0; i < TMC_MAX_HANDLE_COUNT; i++)
188 {
189 if (&tmcHandleArray[i].handle == handle)
190 {
191 tmcHandleArray[i].used = 0;
192 return kStatus_Success;
193 }
194 }
195
196 return kStatus_Fail; // Handle not found in array
197}

199status_t TMCi_set_ihold_divider(TMC_Handle_t handle, uint8_t ihold, TickType_t deadline)
200{
201
202 if (handle == NULL)
203 {
204 return kStatus_Fail;
205 }
206
207 uint32_t ihold_irun;
208 static char logMsg[60];
209
210 if (TMCi_read(handle, TMC_IHOLD_IRUN_ADDR, &ihold_irun, deadline) != kStatus_Success)
211 {
212 snprintf(logMsg, sizeof(logMsg),
213 "[%s] Failed to read IHOLD register - driver communication error", handle->label);
214 LOG_ERROR(logMsg);
215 return kStatus_Fail;
216 }
217
218 ihold_irun &= ~(0b11111 << 0); // Clear IHOLD bits
219 ihold_irun |= (ihold & 0b11111) << 0; // Set new IHOLD bits
220
221 if (TMCi_write(handle, TMC_IHOLD_IRUN_ADDR, &ihold_irun, deadline) != kStatus_Success)
222 {
223 snprintf(logMsg, sizeof(logMsg),
224 "[%s] Failed to write IHOLD register - driver communication error", handle->label);
225 LOG_ERROR(logMsg);
226 return kStatus_Fail;
227 }
228
229 snprintf(logMsg, sizeof(logMsg), "[%s] Hold current divider set to %d", handle->label, ihold);
230 LOG_DEBUG(logMsg);
231
232 return kStatus_Success;
233}

272status_t TMCi_set_microstepping(TMC_Handle_t handle, TMC_MICROSTEPPING_t microstepping,
273 TickType_t deadline)
274{
275 if (handle == NULL)
276 {
277 return kStatus_Fail;
278 }
279
280 static char logMsg[60];
281
282 uint32_t chopConf;
283 TMCi_read(handle, TMC_CHOPCONF_ADDR, &chopConf, deadline);
284 chopConf &= ~(0b1111 << 24);
285 chopConf |= microstepping << 24;
286
287 if (kStatus_Success != TMCi_write(handle, TMC_CHOPCONF_ADDR, &chopConf, deadline))
288 {
289 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to write Microstepping to CHOPCONF register",
290 handle->label);
291 LOG_ERROR(logMsg);
292
293 return kStatus_Fail;
294 }
295 uint16_t microsteppingInteger;
296 TMC_microstepping_enum_to_uint_impl(microstepping, &microsteppingInteger);
297 snprintf(logMsg, sizeof(logMsg), "[%s] Microstepping set to %d", handle->label,
298 microsteppingInteger);
299 LOG_DEBUG(logMsg);
300
301 return kStatus_Success;
302}

327status_t TMCi_set_freewheeling(TMC_Handle_t handle, uint8_t enabled, TickType_t deadline)
328{
329 char logMsg[60];
330
331 if (handle == NULL)
332 {
333 return kStatus_Fail;
334 }
335
336 status_t iholdStatus;
337 if (enabled)
338 {
339 iholdStatus = TMCi_set_ihold_divider(handle, 0, deadline);
340 }
341 else
342 {
343 iholdStatus = TMCi_set_ihold_divider(handle, handle->iholdDivider, deadline);
344 }
345 if (iholdStatus != kStatus_Success)
346 {
347 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to set IHOLD for freewheeling",
348 handle->label);
349 LOG_ERROR(logMsg);
350 return kStatus_Fail;
351 }
352
353 uint32_t pwmconf = 0;
354 if (TMCi_read(handle, TMC_PWMCONF_ADDR, &pwmconf, deadline) != kStatus_Success)
355 {
356 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to read PWMCONF for freewheeling",
357 handle->label);
358 LOG_ERROR(logMsg);
359 return kStatus_Fail;
360 }
361 pwmconf &= ~(0b11 << 20);
362 if (enabled)
363 {
364 pwmconf |= 0b01 << 20; // Enable freewheeling
365 }
366 else
367 {
368 pwmconf &= ~(0b11 << 20); // Disable freewheeling
369 }
370
371 if (kStatus_Success != TMCi_write(handle, TMC_PWMCONF_ADDR, &pwmconf, deadline))
372 {
373 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to write Freewheeling to PWMCONF register",
374 handle->label);
375 LOG_ERROR(logMsg);
376 return kStatus_Fail;
377 }
378
379 handle->freewheeling = enabled;
380 snprintf(logMsg, sizeof(logMsg), "[%s] Freewheeling %s", handle->label,
381 enabled ? "enabled" : "disabled");
382 LOG_INFO(logMsg);
383 return kStatus_Success;
384}
