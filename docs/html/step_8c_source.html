139status_t STP_init()
140{
141 // Initialize acceleration table pool
142 for (int8_t i = 0; i < STP_ACCEL_TABLE_POOL_SIZE; i++)
143 {
144 accelTablePool[i].used = 0;
145 accelTablePool[i].tableSize = 0;
146 }
147
148 cmdQueue = xQueueCreate(STP_MOVEMENT_QUEUE_SIZE, sizeof(STP_CmdQueueItem_t));
149 if (cmdQueue == NULL)
150 {
151 return kStatus_Fail;
152 }
153 configQueue = xQueueCreate(STP_CONFIG_QUEUE_SIZE, sizeof(STP_StepperConfig_t));
154 if (configQueue == NULL)
155 {
156 return kStatus_Fail;
157 }
158
159 SIM->SCGC6 |= SIM_SCGC6_FTM3_MASK;
160
161 NVIC_SetPriority(FTM3_IRQn, 8);
162 NVIC_EnableIRQ(FTM3_IRQn);
163
164 FTM3->MOD = 0xFFFF;
165 FTM3->SC = FTM_SC_CLKS(2) | FTM_SC_PS(0);
166 return kStatus_Success;
167}

249// TODO check for correct step count
250void FTM3_IRQHandler(void)
251{
252 uint32_t intStatus;
253 /* Reading all interrupt flags of status register */
254 intStatus = FTM_GetStatusFlags(FTM3);
255 FTM_ClearStatusFlags(FTM3, intStatus);
256
257 /* Place your code here */
258 for (uint8_t i = 0; i < 8; i++)
259 {
260 STP_StepperHandle_t* stepperHandle;
261
262 if (intStatus & (0b1 << i))
263 {
264 stepperHandle = FTM3_ISR_handle_cache[i];
265
266 stepperHandle->movementHandle.currStepCount++;
267 uint32_t currentStepCount = stepperHandle->movementHandle.currStepCount;
268 uint32_t totalSteps = stepperHandle->movementHandle.totalSteps;
269 uint32_t accelSteps = stepperHandle->movementHandle.accelSteps;
270 uint32_t constVelocitySteps = stepperHandle->movementHandle.endVelocitySteps;
271
272 if (currentStepCount > totalSteps)
273 {
274 stepperHandle->ftmBase->CONTROLS[stepperHandle->ftmChannel].CnSC &=
275 ~(FTM_CnSC_CHIE_MASK | FTM_CnSC_ELSA_MASK | FTM_CnSC_ELSB_MASK);
276 stepperHandle->movementHandle.state = STP_MOVEMENT_FINISHED;
277 continue;
278 }
279 if (stepperHandle->movementHandle.state == STP_MOVEMENT_STARTED)
280 {
281 stepperHandle->movementHandle.state = STP_MOVEMENT_ACCELERATING;
282 stepperHandle->movementHandle.phaseStepCount = 0;
283 stepperHandle->movementHandle.accelInterpCounter = 0;
284 stepperHandle->movementHandle.accelTableIndex = 0;
285 }
286 else if (stepperHandle->movementHandle.state == STP_MOVEMENT_ACCELERATING &&
287 currentStepCount > stepperHandle->movementHandle.accelSteps)
288 {
289 if (stepperHandle->movementHandle.isTrapezoidal)
290 {
291 stepperHandle->movementHandle.state = STP_MOVEMENT_CONST_VELOCITY;
292 }
293 else
294 {
295 stepperHandle->movementHandle.state = STP_MOVEMENT_DECELRATING;
296 stepperHandle->movementHandle.phaseStepCount = 0;
297 stepperHandle->movementHandle.accelInterpCounter = 0;
298 stepperHandle->movementHandle.accelTableIndex =
299 stepperHandle->movementHandle.accelTableSize - 1;
300 }
301 }
302 else if (stepperHandle->movementHandle.state == STP_MOVEMENT_CONST_VELOCITY &&
303 currentStepCount > accelSteps + constVelocitySteps)
304 {
305 stepperHandle->movementHandle.state = STP_MOVEMENT_DECELRATING;
306 stepperHandle->movementHandle.phaseStepCount = 0;
307 stepperHandle->movementHandle.accelInterpCounter = 0;
308 stepperHandle->movementHandle.accelTableIndex =
309 stepperHandle->movementHandle.accelTableSize - 1;
310 }
311
312 // Increment phase step counter
313 stepperHandle->movementHandle.phaseStepCount++;
314
315 uint16_t actualDelay;
316 int8_t poolIndex = stepperHandle->movementHandle.accelTablePoolIndex;
317
318 if (stepperHandle->movementHandle.state == STP_MOVEMENT_ACCELERATING)
319 {
320 // Use counter-based repeat factor (no division!)
321 if (poolIndex >= 0 && poolIndex < STP_ACCEL_TABLE_POOL_SIZE &&
322 stepperHandle->movementHandle.accelTableIndex <
323 accelTablePool[poolIndex].tableSize)
324 {
325 actualDelay = accelTablePool[poolIndex]
326 .table[stepperHandle->movementHandle.accelTableIndex];
327 // Increment counter and advance table index when counter reaches factor
328 stepperHandle->movementHandle.accelInterpCounter++;
329 if (stepperHandle->movementHandle.accelInterpCounter >=
330 stepperHandle->movementHandle.accelInterpFactor)
331 {
332 stepperHandle->movementHandle.accelInterpCounter = 0;
333 stepperHandle->movementHandle.accelTableIndex++;
334 }
335 }
336 else
337 {
338 actualDelay = stepperHandle->movementHandle.endVelocityDelay;
339 }
340 }
341 else if (stepperHandle->movementHandle.state == STP_MOVEMENT_CONST_VELOCITY)
342 {
343 // Constant velocity: use endVelocityDelay directly
344 actualDelay = stepperHandle->movementHandle.endVelocityDelay;
345 }
346 else if (stepperHandle->movementHandle.state == STP_MOVEMENT_DECELRATING)
347 {
348 // Use counter-based repeat factor reading table backwards (no division!)
349 if (poolIndex >= 0 && poolIndex < STP_ACCEL_TABLE_POOL_SIZE &&
350 stepperHandle->movementHandle.accelTableIndex <
351 accelTablePool[poolIndex].tableSize)
352 {
353 actualDelay = accelTablePool[poolIndex]
354 .table[stepperHandle->movementHandle.accelTableIndex];
355 // Increment counter and decrement table index when counter reaches factor
356 stepperHandle->movementHandle.accelInterpCounter++;
357 if (stepperHandle->movementHandle.accelInterpCounter >=
358 stepperHandle->movementHandle.accelInterpFactor)
359 {
360 stepperHandle->movementHandle.accelInterpCounter = 0;
361 if (stepperHandle->movementHandle.accelTableIndex > 0)
362 {
363 stepperHandle->movementHandle.accelTableIndex--;
364 }
365 }
366 }
367 else
368 {
369 actualDelay = stepperHandle->movementHandle.endVelocityDelay;
370 }
371 }
372 else
373 {
374 actualDelay = stepperHandle->movementHandle.endVelocityDelay;
375 }
376 stepperHandle->ftmBase->CONTROLS[stepperHandle->ftmChannel].CnV += actualDelay;
377 }
378 }

401static status_t init_handle(const STP_StepperConfig_t* config)
402{
403 if (config == NULL)
404 {
405 return kStatus_Fail;
406 }
407
408 STP_StepperHandle_t* handle;
409 for (uint8_t i = 0; i < STP_MAX_HANDLE_COUNT; i++)
410 {
411 if (handles[i].used == 0)
412 {
413 handles[i].used = 1;
414 handle = &handles[i].handle;
415 break;
416 }
417 }
418
419 if (handle == NULL)
420 {
421 LOG_ERROR("Failed to init stepper handle. Too many handles.");
422 return kStatus_Fail;
423 }
424
425 handle->ftmBase = config->ftmBase;
426 handle->ftmChannel = config->ftmChannel;
427
428 handle->stepPort = config->stepPort;
429 handle->stepGPIO = config->stepGPIO;
430 handle->stepPin = config->stepPin;
431 handle->stepMuxFTM = config->stepMuxFTM;
432 handle->stepMuxGPIO = config->stepMuxGPIO;
433
434 handle->dirPort = config->dirPort;
435 handle->dirGPIO = config->dirGPIO;
436 handle->dirPin = config->dirPin;
437 handle->dirLogicHighClockwise = config->dirLogicHighClockwise;
438
439 handle->acceleration = config->acceleration;
440 handle->stepAngle = config->stepAngle;
441 handle->endVelocity = config->endVelocity;
442 handle->label = config->label;
443 handle->dirMux = config->dirMux;
444
445 handle->microstepping = 4;
446
447 if (reset_movement_handle(&handle->movementHandle) != kStatus_Success)
448 return kStatus_Fail;
449
450 if (init_step_pin(handle) != kStatus_Success)
451 return kStatus_Fail;
452
453 if (init_dir_pin(handle) != kStatus_Success)
454 return kStatus_Fail;
455
456 update_isr_handle_cache();
457
458 static char logMsg[90];
459
460 snprintf(logMsg, sizeof(logMsg), "[%s] stepper handle initialized: accel=%.1f, endVel=%.1f",
461 handle->label, handle->acceleration, handle->endVelocity);
462 LOG_INFO(logMsg);
463 return kStatus_Success;
464}

533static status_t plan_motion_profile(STP_StepperHandle_t* handle)
534{
535 if (handle == NULL)
536 {
537 return kStatus_Fail;
538 }
539
540 calculate_step_profile(handle);
541
542 uint16_t stepZeroDelay;
543 uint16_t endVelocityDelay;
544 calculate_delays(handle, &stepZeroDelay, &endVelocityDelay);
545
546 uint32_t interpFactor;
547 uint32_t tableSize;
548 calculate_acceleration_table_parameters(handle, &interpFactor, &tableSize);
549
550 int8_t poolIndex;
551 if (create_accel_table(handle, tableSize, stepZeroDelay, endVelocityDelay, &poolIndex) !=
552 kStatus_Success)
553 return kStatus_Fail;
554
555 // Store table information in movement handle
556 handle->movementHandle.accelTablePoolIndex = poolIndex;
557 handle->movementHandle.accelTableSize = tableSize;
558 handle->movementHandle.accelInterpFactor = (uint16_t)interpFactor;
559
560 handle->movementHandle.currStepCount = 0;
561 handle->movementHandle.state = STP_MOVEMENT_PLANNED;
562
563 static char logMsg[100];
564 if (handle->movementHandle.isTrapezoidal)
565 {
566 snprintf(logMsg, sizeof(logMsg),
567 "[%s] Motion planned: trapezoidal (accel:%lu const:%lu decel:%lu)", handle->label,
568 handle->movementHandle.accelSteps, handle->movementHandle.endVelocitySteps,
569 handle->movementHandle.decelSteps);
570 }
571 else
572 {
573 snprintf(logMsg, sizeof(logMsg), "[%s] Motion planned: triangular (accel:%lu decel:%lu)",
574 handle->label, handle->movementHandle.accelSteps,
575 handle->movementHandle.decelSteps);
576 }
577 LOG_DEBUG(logMsg);
578 return kStatus_Success;
579}

581static inline void calculate_step_profile(STP_StepperHandle_t* handle)
582{
583 uint32_t accelSteps =
584 (uint32_t)((handle->endVelocity * handle->endVelocity * handle->microstepping) /
585 (2 * handle->acceleration * handle->stepAngle));
586
587 if (accelSteps * 2 >= handle->movementHandle.totalSteps)
588 {
589
590 handle->movementHandle.isTrapezoidal = 0;
591
592 handle->movementHandle.accelSteps = handle->movementHandle.totalSteps / 2;
593
594 handle->movementHandle.decelSteps =
595 handle->movementHandle.totalSteps - handle->movementHandle.accelSteps;
596
597 handle->movementHandle.endVelocitySteps = 0;
598 }
599 else
600 {
601
602 handle->movementHandle.isTrapezoidal = 1;
603
604 handle->movementHandle.accelSteps = accelSteps;
605 handle->movementHandle.decelSteps = accelSteps;
606
607 handle->movementHandle.endVelocitySteps =
608 handle->movementHandle.totalSteps - 2 * accelSteps;
609 }
610}

612static inline void calculate_delays(STP_StepperHandle_t* handle, uint16_t* stepZeroDelay,
613 uint16_t* endVelocityDelay)
614{
615 *stepZeroDelay =
616 (uint16_t)(0.676 * STP_TIMER_FREQ_HZ *
617 sqrt(2.0 * handle->stepAngle / (handle->acceleration * handle->microstepping)));
618
619 *endVelocityDelay = (uint16_t)(handle->stepAngle * STP_TIMER_FREQ_HZ /
620 (handle->endVelocity * handle->microstepping));
621
622 handle->movementHandle.endVelocityDelay = *endVelocityDelay;
623}

625static inline void calculate_acceleration_table_parameters(STP_StepperHandle_t* handle,
626 uint32_t* interpFactor,
627 uint32_t* tableSize)
628{
629 *interpFactor = STP_ACCEL_TABLE_INTERP_FACTOR;
630
631 if (*interpFactor < 1)
632 *interpFactor = 1;
633
634 if (handle->movementHandle.accelSteps > STP_MAX_ACCEL_TABLE_SIZE * (*interpFactor))
635 {
636 *interpFactor = (handle->movementHandle.accelSteps + STP_MAX_ACCEL_TABLE_SIZE - 1) /
637 STP_MAX_ACCEL_TABLE_SIZE;
638 }
639
640 *tableSize = (handle->movementHandle.accelSteps + *interpFactor - 1) / *interpFactor;
641
642 if (*tableSize > STP_MAX_ACCEL_TABLE_SIZE)
643 *tableSize = STP_MAX_ACCEL_TABLE_SIZE;
644}

646static status_t create_accel_table(STP_StepperHandle_t* handle, uint32_t tableSize,
647 uint16_t stepZeroDelay, uint16_t endVelocityDelay,
648 int8_t* poolIndex)
649{
650
651 if (handle == NULL || poolIndex == NULL)
652 return kStatus_Fail;
653
654 status_t status;
655
656 status = allocate_accel_table(tableSize, poolIndex);
657
658 if (status != kStatus_Success)
659 return kStatus_Fail;
660
661 if (tableSize > 0)
662 {
663 status = calculate_acceleration_profile(accelTablePool[*poolIndex].table, tableSize,
664 stepZeroDelay, endVelocityDelay);
665
666 if (status != kStatus_Success)
667 {
668 free_accel_table(*poolIndex);
669 return kStatus_Fail;
670 }
671 }
672
673 return kStatus_Success;
674}

716static status_t calculate_acceleration_profile(uint16_t* lookupTable, uint32_t numSteps,
717 uint16_t stepZeroDelay, uint16_t endVelocityDelay)
718{
719 if (lookupTable == NULL || numSteps == 0)
720 {
721 return kStatus_Fail;
722 }
723
724 // Start with initial delay, scaled up for fixed-point arithmetic
725 uint32_t currentDelay = ((uint32_t)stepZeroDelay) << STP_DELAY_SCALE_BITS;
726 uint32_t minDelay = ((uint32_t)endVelocityDelay) << STP_DELAY_SCALE_BITS;
727
728 // Calculate each step delay using David Austin formula
729 for (uint32_t n = 1; n <= numSteps; n++)
730 {
731 // Store current delay (scaled down to uint16_t)
732 lookupTable[n - 1] = (uint16_t)(currentDelay >> STP_DELAY_SCALE_BITS);
733
734 // Calculate next delay: c_n = c_(n-1) - 2*c_(n-1) / (4*n + 1)
735 // Using fixed-point arithmetic to maintain fractional precision
736 uint32_t nextDelay = currentDelay - (2 * currentDelay / (4 * n + 1));
737
738 // Clamp to minimum delay (end velocity)
739 if (nextDelay < minDelay)
740 {
741 nextDelay = minDelay;
742 }
743
744 currentDelay = nextDelay;
745 }
746 return kStatus_Success;
747}

749static status_t start_steps(STP_StepperHandle_t* handle)
750{
751
752 static char logMsg[100];
753
754 if (step_pin_mux_ftm(handle) != kStatus_Success)
755 return kStatus_Fail;
756
757 if (set_direction_pin(handle) != kStatus_Success)
758 return kStatus_Fail;
759
760 handle->ftmBase->CONTROLS[handle->ftmChannel].CnSC &= ~FTM_CnSC_CHF_MASK;
761
762 // Use first value from lookup table
763 uint16_t initialDelay;
764 int8_t poolIndex = handle->movementHandle.accelTablePoolIndex;
765
766 if (poolIndex >= 0 && poolIndex < STP_ACCEL_TABLE_POOL_SIZE &&
767 accelTablePool[poolIndex].tableSize > 0)
768 {
769 initialDelay = accelTablePool[poolIndex].table[0];
770 }
771 else
772 {
773 snprintf(logMsg, sizeof(logMsg),
774 "[%s] Failed to retrieve initial delay from lookup table for %lu steps",
775 handle->label, handle->movementHandle.totalSteps);
776 LOG_ERROR(logMsg);
777 return kStatus_Fail;
778 }
779 handle->ftmBase->CONTROLS[handle->ftmChannel].CnV = handle->ftmBase->CNT + initialDelay;
780 handle->ftmBase->CONTROLS[handle->ftmChannel].CnSC |=
781 FTM_CnSC_MSA_MASK | FTM_CnSC_ELSA_MASK | FTM_CnSC_CHIE_MASK;
782
783 handle->movementHandle.state = STP_MOVEMENT_STARTED;
784
785 snprintf(logMsg, sizeof(logMsg), "[%s] Movement started: %lu steps", handle->label,
786 handle->movementHandle.totalSteps);
787 LOG_DEBUG(logMsg);
788 return kStatus_Success;
789}

838static status_t stop_steps(STP_StepperHandle_t* handle, uint8_t doDeceleration)
839{
840 if (handle == NULL)
841 {
842 return kStatus_Fail;
843 }
844
845 if (doDeceleration == 0)
846 {
847 NVIC_DisableIRQ(FTM3_IRQn);
848 handle->ftmBase->CONTROLS[handle->ftmChannel].CnSC &=
849 ~(FTM_CnSC_CHIE_MASK | FTM_CnSC_ELSA_MASK | FTM_CnSC_ELSB_MASK);
850 handle->ftmBase->CONTROLS[handle->ftmChannel].CnSC &= ~(FTM_CnSC_CHF_MASK);
851 handle->movementHandle.state = STP_MOVEMENT_STOPPED;
852 if (step_pin_mux_gpio(handle) != kStatus_Success)
853 return kStatus_Fail;
854 NVIC_EnableIRQ(FTM3_IRQn);
855 }
856 else
857 {
858 NVIC_DisableIRQ(FTM3_IRQn);
859 if (handle->movementHandle.state == STP_MOVEMENT_ACCELERATING)
860 {
861 handle->movementHandle.isTrapezoidal = 0;
862 handle->movementHandle.endVelocitySteps = 0;
863 handle->movementHandle.accelSteps = handle->movementHandle.currStepCount;
864 handle->movementHandle.decelSteps = handle->movementHandle.currStepCount;
865 handle->movementHandle.totalSteps = handle->movementHandle.currStepCount * 2;
866 }
867 else if (handle->movementHandle.state == STP_MOVEMENT_CONST_VELOCITY)
868 {
869 handle->movementHandle.endVelocitySteps =
870 handle->movementHandle.currStepCount - handle->movementHandle.accelSteps;
871 handle->movementHandle.totalSteps = handle->movementHandle.accelSteps +
872 handle->movementHandle.endVelocitySteps +
873 handle->movementHandle.decelSteps;
874 }
875 NVIC_EnableIRQ(FTM3_IRQn);
876 }
877
878 return kStatus_Success;
879}

898static void process_movement(void)
899{
900 static char logMsg[80];
901
902 // Check all handles for finished movements
903 for (size_t i = 0; i < STP_MAX_HANDLE_COUNT; i++)
904 {
905 if (handles[i].used == 1 &&
906 (handles[i].handle.movementHandle.state == STP_MOVEMENT_FINISHED ||
907 handles[i].handle.movementHandle.state == STP_MOVEMENT_STOPPED))
908 {
909
910 step_pin_mux_gpio(&handles[i].handle);
911
912 if (handles[i].handle.movementHandle.state == STP_MOVEMENT_FINISHED)
913 snprintf(logMsg, sizeof(logMsg), "[%s] Movement finished", handles[i].handle.label);
914 else if (handles[i].handle.movementHandle.state == STP_MOVEMENT_STOPPED)
915 snprintf(logMsg, sizeof(logMsg), "[%s] Movement stopped", handles[i].handle.label);
916 LOG_DEBUG(logMsg);
917 reset_movement_handle(&handles[i].handle.movementHandle);
918 }
919 }
920
921 STP_CmdQueueItem_t queueItem;
922 BaseType_t status;
923 status = xQueueReceive(cmdQueue, &queueItem, pdMS_TO_TICKS(10));
924 if (status != pdPASS)
925 {
926 return;
927 }
928 switch (queueItem.type)
929 {
930 case STP_CMD_MOVE:
931 {
932 LOG_DEBUG("Processing movement command");
933 queueItem.handle->movementHandle.totalSteps = queueItem.data.move.stepCount;
934 queueItem.handle->movementHandle.direction = queueItem.data.move.direction;
935 if (plan_motion_profile(queueItem.handle) == kStatus_Success)
936 {
937 start_steps(queueItem.handle);
938 }
939 break;
940 }
941 case STP_CMD_STOP:
942 {
943 LOG_DEBUG("Processing stop command");
944 stop_steps(queueItem.handle, queueItem.data.stop.doDeceleration);
945 break;
946 }
947 }
948}
