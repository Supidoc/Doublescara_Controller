233 SIM->SCGC6 |= SIM_SCGC6_FTM3_MASK;

236 NVIC_EnableIRQ(FTM3_IRQn);

240
241 gpio_pin_config_t debugPinConfig;
242 debugPinConfig.pinDirection = kGPIO_DigitalOutput;
243
244 GPIO_PinInit(isrDebugGPIO, isrDebugPin, &debugPinConfig);
245

247 GPIO_PinWrite(isrDebugGPIO, isrDebugPin, 0);

249 THE_init_cmd_handles(cmdHandles, STP_MAX_CMD_HANDLE_COUNT);

480static status_t send_cmd_async(STP_CmdQueueItem_t* queueItem, TickType_t deadline,
481 THE_CmdHandle_t* cmdHandle)
482{
483 if (queueItem == NULL)
484 {
485 return kStatus_Fail;
486 }
487 THE_CmdHandle_t internaleCmdHandle = NULL;
488
489 status_t allocStatus =
490 THE_get_cmd_handle(&internaleCmdHandle, cmdHandles, STP_MAX_CMD_HANDLE_COUNT);
491 if (allocStatus != kStatus_Success)
492 {
493 return kStatus_Fail;
494 }
495 if (cmdHandle != NULL)
496 {
497 THE_add_cmd_handle_ref(internaleCmdHandle);
498 }
499
500 queueItem->cmdHandle = internaleCmdHandle;
501
502 status_t sendStatus = THE_send_cmd(cmdQueue, queueItem, deadline, internaleCmdHandle);
503 if (sendStatus != kStatus_Success)
504 {
505 if (cmdHandle != NULL)
506 {
507 THE_remove_cmd_handle_ref(*cmdHandle);
508 *cmdHandle = NULL;
509 }
510 return kStatus_Fail;
511 }
512 if (cmdHandle != NULL)
513 {
514 *cmdHandle = internaleCmdHandle;
515 }
516 return kStatus_Success;
517}

519// TODO check for correct step count
520void FTM3_IRQHandler(void)
521{
522 uint32_t intStatus;
523 /* Reading all interrupt flags of status register */
524 intStatus = FTM_GetStatusFlags(FTM3);
525 FTM_ClearStatusFlags(FTM3, intStatus);
526
527 traceISR_ENTER();
528 GPIO_PinWrite(isrDebugGPIO, isrDebugPin, 1);
529
530 /* Place your code here */
531 for (uint8_t i = 0; i < 8; i++)
532 {
533 STP_Handle_t stepperHandle;
534
535 if (intStatus & (0b1 << i))
536 {
537 if (FTM3_ISR_handle_cache[i] == NULL)
538 {
539 continue;
540 }
541 stepperHandle = FTM3_ISR_handle_cache[i];
542
543 stepperHandle->movementHandle.currStepCount++;
544
545 stepperHandle->absolutePosition +=
546 (stepperHandle->movementHandle.direction == STP_COUNTERCLOCKWISE) ? 1 : -1;
547
548 uint32_t currentStepCount = stepperHandle->movementHandle.currStepCount;
549 uint32_t totalSteps = stepperHandle->movementHandle.totalSteps;
550 uint32_t accelSteps = stepperHandle->movementHandle.accelSteps;
551 uint32_t constVelocitySteps = stepperHandle->movementHandle.endVelocitySteps;
552
553 // Toggle Pin because letting the time do it leads to inconsistencies in stepcount
554 // because the start level for the timer output can't be defined
555 GPIO_PortToggle(stepperHandle->stepGPIO, 0b1 << stepperHandle->stepPin);
556
557 if (currentStepCount >= totalSteps)
558 {
559 stepperHandle->ftmBase->CONTROLS[stepperHandle->ftmChannel].CnSC &=
560 ~(FTM_CnSC_CHIE_MASK | FTM_CnSC_ELSA_MASK | FTM_CnSC_ELSB_MASK);
561
562 stepperHandle->movementHandle.state = STP_MOVEMENT_FINISHED;
563 continue;
564 }
565 if (stepperHandle->movementHandle.state == STP_MOVEMENT_STARTED)
566 {
567 stepperHandle->movementHandle.state = STP_MOVEMENT_ACCELERATING;
568 stepperHandle->movementHandle.phaseStepCount = 0;
569 stepperHandle->movementHandle.accelInterpCounter = 0;
570 stepperHandle->movementHandle.accelTableIndex = 0;
571 }
572 else if (stepperHandle->movementHandle.state == STP_MOVEMENT_ACCELERATING &&
573 currentStepCount > stepperHandle->movementHandle.accelSteps)
574 {
575 if (stepperHandle->movementHandle.isTrapezoidal)
576 {
577 stepperHandle->movementHandle.state = STP_MOVEMENT_CONST_VELOCITY;
578 }
579 else
580 {
581 stepperHandle->movementHandle.state = STP_MOVEMENT_DECELRATING;
582 stepperHandle->movementHandle.phaseStepCount = 0;
583 stepperHandle->movementHandle.accelInterpCounter = 0;
584 stepperHandle->movementHandle.accelTableIndex =
585 stepperHandle->movementHandle.accelTableSize - 1;
586 }
587 }
588 else if (stepperHandle->movementHandle.state == STP_MOVEMENT_CONST_VELOCITY &&
589 currentStepCount > accelSteps + constVelocitySteps)
590 {
591 stepperHandle->movementHandle.state = STP_MOVEMENT_DECELRATING;
592 stepperHandle->movementHandle.phaseStepCount = 0;
593 stepperHandle->movementHandle.accelInterpCounter = 0;
594 stepperHandle->movementHandle.accelTableIndex =
595 stepperHandle->movementHandle.accelTableSize - 1;
596 }
597
598 // Increment phase step counter
599 stepperHandle->movementHandle.phaseStepCount++;
600
601 uint16_t actualDelay;
602 int8_t poolIndex = stepperHandle->movementHandle.accelTablePoolIndex;
603
604 if (stepperHandle->movementHandle.state == STP_MOVEMENT_ACCELERATING)
605 {
606 // Use counter-based repeat factor (no division!)
607 if (poolIndex >= 0 && poolIndex < STP_ACCEL_TABLE_POOL_SIZE &&
608 stepperHandle->movementHandle.accelTableIndex <
609 accelTablePool[poolIndex].tableSize)
610 {
611 actualDelay = accelTablePool[poolIndex]
612 .table[stepperHandle->movementHandle.accelTableIndex];
613 // Increment counter and advance table index when counter reaches factor
614 stepperHandle->movementHandle.accelInterpCounter++;
615 if (stepperHandle->movementHandle.accelInterpCounter >=
616 stepperHandle->movementHandle.accelInterpFactor)
617 {
618 stepperHandle->movementHandle.accelInterpCounter = 0;
619 stepperHandle->movementHandle.accelTableIndex++;
620 }
621 }
622 else
623 {
624 actualDelay = stepperHandle->movementHandle.endVelocityDelay;
625 }
626 }
627 else if (stepperHandle->movementHandle.state == STP_MOVEMENT_CONST_VELOCITY)
628 {
629 // Constant velocity: use endVelocityDelay directly
630 actualDelay = stepperHandle->movementHandle.endVelocityDelay;
631 }
632 else if (stepperHandle->movementHandle.state == STP_MOVEMENT_DECELRATING)
633 {
634 // Use counter-based repeat factor reading table backwards (no division!)
635 if (poolIndex >= 0 && poolIndex < STP_ACCEL_TABLE_POOL_SIZE &&
636 stepperHandle->movementHandle.accelTableIndex <
637 accelTablePool[poolIndex].tableSize)
638 {
639 actualDelay = accelTablePool[poolIndex]
640 .table[stepperHandle->movementHandle.accelTableIndex];
641 // Increment counter and decrement table index when counter reaches factor
642 stepperHandle->movementHandle.accelInterpCounter++;
643 if (stepperHandle->movementHandle.accelInterpCounter >=
644 stepperHandle->movementHandle.accelInterpFactor)
645 {
646 stepperHandle->movementHandle.accelInterpCounter = 0;
647 if (stepperHandle->movementHandle.accelTableIndex > 0)
648 {
649 stepperHandle->movementHandle.accelTableIndex--;
650 }
651 }
652 }
653 else
654 {
655 actualDelay = stepperHandle->movementHandle.endVelocityDelay;
656 }
657 }
658 else
659 {
660 actualDelay = stepperHandle->movementHandle.endVelocityDelay;
661 }
662 stepperHandle->ftmBase->CONTROLS[stepperHandle->ftmChannel].CnV += actualDelay;
663 }
664 }
665
666 GPIO_PinWrite(isrDebugGPIO, isrDebugPin, 0);
667 traceISR_EXIT();
668
669/* Add for ARM errata 838869, affects Cortex-M4, Cortex-M4F
670 Store immediate overlapping exception return operation might vector to incorrect interrupt. */
671#if defined __CORTEX_M && (__CORTEX_M == 4U)
672 __DSB();
673#endif
674}

688 }

699 handle->ftmChannel = config.ftmChannel;
700
701 handle->stepPort = config.stepPort;
702 handle->stepGPIO = config.stepGPIO;

704
705 handle->dirPort = config.dirPort;
706 handle->dirPin = config.dirPin;
707 handle->dirLogicHighClockwise = config.dirLogicHighClockwise;
708
709 handle->acceleration = config.acceleration;
710 handle->endVelocity = config.endVelocity;
711 handle->label = config.label;
712 handle->absolutePosition = 0;
713
714 if (reset_movement_handle(&handle->movementHandle) != kStatus_Success)
715 {
716 return kStatus_Fail;
717 }
718
719 if (init_step_pin(handle) != kStatus_Success)
720 {
721 return kStatus_Fail;

723

726 return kStatus_Fail;

730
731 snprintf(logMsg, sizeof(logMsg),
732 "[%s] Stepper initialized successfully (accel=%.1f step/s², vel=%.1f step/s)",
733 handle->label, handle->acceleration, handle->endVelocity);
734 LOG_INFO(logMsg);
735
736 return kStatus_Success;
737}

834static inline void calculate_step_profile(STP_Handle_t handle)
835{
836 uint32_t accelSteps =
837 (uint32_t)((handle->endVelocity * handle->endVelocity) / (2 * handle->acceleration));
838
839 if (accelSteps * 2 >= handle->movementHandle.totalSteps)
840 {

844 handle->movementHandle.accelSteps = handle->movementHandle.totalSteps / 2;

846 handle->movementHandle.decelSteps =
847 handle->movementHandle.totalSteps - handle->movementHandle.accelSteps;
848
849 handle->movementHandle.endVelocitySteps = 0;
850 }
851 else
852 {
853
854 handle->movementHandle.isTrapezoidal = 1;
855
856 handle->movementHandle.accelSteps = accelSteps;
857 handle->movementHandle.decelSteps = accelSteps;

860 handle->movementHandle.totalSteps - 2 * accelSteps;
861 }

972static status_t calculate_acceleration_profile(uint16_t* lookupTable, uint32_t numSteps,
973 uint16_t stepZeroDelay, uint16_t endVelocityDelay)
974{
975 if (lookupTable == NULL || numSteps == 0)
976 {
977 return kStatus_Fail;
978 }
979
980 // Start with initial delay, scaled up for fixed-point arithmetic
981 uint32_t currentDelay = ((uint32_t)stepZeroDelay) << STP_DELAY_SCALE_BITS;
982 uint32_t minDelay = ((uint32_t)endVelocityDelay) << STP_DELAY_SCALE_BITS;
983
984 // Calculate each step delay using David Austin formula
985 for (uint32_t n = 1; n <= numSteps; n++)
986 {
987 // Store current delay (scaled down to uint16_t)
988 lookupTable[n - 1] = (uint16_t)(currentDelay >> STP_DELAY_SCALE_BITS);
989
990 // Calculate next delay: c_n = c_(n-1) - 2*c_(n-1) / (4*n + 1)
991 // Using fixed-point arithmetic to maintain fractional precision
992 uint32_t nextDelay = currentDelay - (2 * currentDelay / (4 * n + 1));
993
994 // Clamp to minimum delay (end velocity)
995 if (nextDelay < minDelay)
996 {
997 nextDelay = minDelay;
998 }
999
1000 currentDelay = nextDelay;
1001 }
1002 return kStatus_Success;
1003}

1104static status_t stop_steps(STP_Handle_t handle, uint8_t doDeceleration, THE_CmdHandle_t cmdHandle)
1105{
1106 if (handle == NULL)
1107 {
1108 return kStatus_Fail;
1109 }
1110
1111 THE_CmdHandle_t oldCmdHandle = NULL;
1112
1113 taskENTER_CRITICAL();
1114
1115 STP_MovementState_t st = handle->movementHandle.state;
1116 if (st == STP_MOVEMENT_IDLE || st == STP_MOVEMENT_FINISHED || st == STP_MOVEMENT_STOPPED)
1117 {
1118 taskEXIT_CRITICAL();
1119 return kStatus_Fail;
1120 }
1121
1122 oldCmdHandle = handle->movementHandle.cmdHandle;
1123
1124 if (doDeceleration == 0)
1125 {
1126 handle->ftmBase->CONTROLS[handle->ftmChannel].CnSC &=
1127 ~(FTM_CnSC_CHIE_MASK | FTM_CnSC_ELSA_MASK | FTM_CnSC_ELSB_MASK);

1129 handle->movementHandle.state = STP_MOVEMENT_STOPPED;
1130 }
1131 else
1132 {
1133 if (st == STP_MOVEMENT_PLANNED)
1134 {
1135 // nothing running yet -> stop immediately
1136 handle->movementHandle.state = STP_MOVEMENT_STOPPED;
1137 }
1138 else if (st == STP_MOVEMENT_ACCELERATING)
1139 {
1140 handle->movementHandle.isTrapezoidal = 0;
1141 handle->movementHandle.endVelocitySteps = 0;
1142 handle->movementHandle.accelSteps = handle->movementHandle.currStepCount;
1143 handle->movementHandle.decelSteps = handle->movementHandle.currStepCount;
1144 handle->movementHandle.totalSteps = handle->movementHandle.currStepCount * 2;
1145 }
1146 else if (st == STP_MOVEMENT_CONST_VELOCITY)
1147 {
1148 handle->movementHandle.endVelocitySteps =
1149 handle->movementHandle.currStepCount - handle->movementHandle.accelSteps;
1150 handle->movementHandle.totalSteps = handle->movementHandle.accelSteps +
1151 handle->movementHandle.endVelocitySteps +
1152 handle->movementHandle.decelSteps;
1153 }
1154 // if already decelerating: keep as-is
1155 }
1156
1157 handle->movementHandle.cmdHandle = cmdHandle;
1158 taskEXIT_CRITICAL();
1159
1160 if (oldCmdHandle != NULL && oldCmdHandle != cmdHandle)
1161 {
1162 THE_notify_task_failure(oldCmdHandle);
1163 THE_remove_cmd_handle_ref(oldCmdHandle);
1164 }
1165
1166 return kStatus_Success;

1239 queueItem.handle->movementHandle.direction =

1343 THE_notify_task_failure(queueItem.cmdHandle);
1344 }

1347 }
1348 case STP_CMD_SET_ACCELERATION:
1349 {
1350 LOG_DEBUG("Processing set acceleration command");

1361 THE_remove_cmd_handle_ref(queueItem.cmdHandle);

1363 }

1365 {
1366 THE_notify_task_failure(queueItem.cmdHandle);
1367 THE_remove_cmd_handle_ref(queueItem.cmdHandle);
1368 LOG_WARN("Received unknown command type");

1370 }
1371 }
