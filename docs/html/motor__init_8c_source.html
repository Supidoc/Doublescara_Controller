44status_t MTR_init_motor(MTR_MotorConfig_t config, TickType_t deadline,
45 MTR_HandlesArrayItem_t* handles, uint8_t maxHandles)
46{
47 static char logMsg[100];
48
49 if (handles == NULL || maxHandles == 0)
50 {
51 return kStatus_Fail;
52 }
53
54 if (config.stepAngle <= 0.0 || config.microstep == 0 || config.reductionFactor <= 0.0)
55 {
56 snprintf(
57 logMsg, sizeof(logMsg),
58 "[%s] Invalid motor configuration: stepAngle=%.4f, microstep=%d, reductionFactor=%.4f",
59 config.label, config.stepAngle, config.microstep, config.reductionFactor);
60 LOG_FATAL(logMsg);
61 return kStatus_Fail;
62 }
63
64 STP_StepperConfig_t stepperConfig;
65 STP_get_default_config(&stepperConfig);
66
67 stepperConfig.ftmBase = config.stepperConfig.ftmBase;
68 stepperConfig.ftmChannel = config.stepperConfig.ftmChannel;
69 stepperConfig.stepPort = config.stepperConfig.stepPort;
70 stepperConfig.stepGPIO = config.stepperConfig.stepGPIO;
71 stepperConfig.stepPin = config.stepperConfig.stepPin;
72 stepperConfig.dirPort = config.stepperConfig.dirPort;
73 stepperConfig.dirPin = config.stepperConfig.dirPin;
74 stepperConfig.dirLogicHighClockwise = config.stepperConfig.dirLogicHighClockwise;
75 stepperConfig.label = config.label;
76
77 stepperConfig.acceleration =
78 config.acceleration * config.reductionFactor * config.microstep / config.stepAngle;
79 stepperConfig.endVelocity =
80 config.endVelocity * config.reductionFactor * config.microstep / config.stepAngle;
81
82 CHD_CmdHandle_t stepCmdHandle = NULL;
83 if (STP_init_handle_async(stepperConfig, deadline, &stepCmdHandle) != kStatus_Success)
84 {
85 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to initialize stepper motor handle",
86 config.label);
87 LOG_ERROR(logMsg);
88 return kStatus_Fail;
89 }
90
91 TMC_Config_t tmcConfig;
92 TMC_get_default_config(&tmcConfig);
93
94 tmcConfig.label = config.label;
95 TMC_microstepping_uint_to_enum(config.microstep, &tmcConfig.microstepping);
96 tmcConfig.serialAdress = config.tmcConfig.serialAdress;
97 tmcConfig.uartHandle = config.tmcConfig.uartHandle;
98 tmcConfig.uartRTOSHandle = config.tmcConfig.uartRTOSHandle;
99 TMC_current_to_divider(config.tmcConfig.iHoldCurrentA, TMC_ROUND_NEAREST,
100 &tmcConfig.iholdDivider);
101 TMC_current_to_divider(config.tmcConfig.iRunCurrentA, TMC_ROUND_NEAREST,
102 &tmcConfig.irunDivider);
103
104 CHD_CmdHandle_t tmcCmdHandle = NULL;
105 if (TMC_init_handle_async(tmcConfig, deadline, &tmcCmdHandle) != kStatus_Success)
106 {
107 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to initialize TMC driver handle",
108 config.label);
109 LOG_ERROR(logMsg);
110 return kStatus_Fail;
111 }
112
113 if (wait_for_cmd_handle(stepCmdHandle, deadline) != kStatus_Success)
114 {
115 snprintf(logMsg, sizeof(logMsg), "[%s] Stepper initialization timed out or failed",
116 config.label);
117 LOG_ERROR(logMsg);
118 return kStatus_Fail;
119 }
120
121 STP_Handle_t stepperHandle = NULL;
122 if (STP_get_handle_by_label(config.label, &stepperHandle) != kStatus_Success ||
123 stepperHandle == NULL)
124 {
125 snprintf(logMsg, sizeof(logMsg),
126 "[%s] Failed to retrieve stepper handle after initialization", config.label);
127 LOG_FATAL(logMsg);
128 return kStatus_Fail;
129 }
130
131 if (wait_for_cmd_handle(tmcCmdHandle, deadline) != kStatus_Success)
132 {
133 snprintf(logMsg, sizeof(logMsg), "[%s] TMC initialization timed out or failed",
134 config.label);
135 LOG_ERROR(logMsg);
136 return kStatus_Fail;
137 }
138
139 TMC_Handle_t tmcHandle = NULL;
140 if (TMC_get_handle_by_label(config.label, &tmcHandle) != kStatus_Success || tmcHandle == NULL)
141 {
142 snprintf(logMsg, sizeof(logMsg), "[%s] Failed to retrieve TMC handle after initialization",
143 config.label);
144 LOG_FATAL(logMsg);
145 return kStatus_Fail;
146 }
147
148 uint8_t handleIndex = maxHandles;
149 for (uint8_t i = 0; i < maxHandles; i++)
150 {
151 if (handles[i].used == 0)
152 {
153 handles[i].used = 1;
154 handleIndex = i;
155 break;
156 }
157 }
158
159 if (handleIndex == maxHandles)
160 {
161 snprintf(logMsg, sizeof(logMsg),
162 "[%s] No free motor handle slots available (max %d motors)", config.label,
163 maxHandles);
164 LOG_FATAL(logMsg);
165 return kStatus_Fail;
166 }
167
168 MTR_MotorHandle_t newHandle = &handles[handleIndex].handle;
169 newHandle->stepperHandle = stepperHandle;
170 newHandle->tmcHandle = tmcHandle;
171 newHandle->stepAngle = config.stepAngle;
172 newHandle->microstep = config.microstep;
173 newHandle->reductionFactor = config.reductionFactor;
174 newHandle->label = config.label;
175 newHandle->freewheeling = 0;
176 newHandle->previousHoldCurrent = config.tmcConfig.iHoldCurrentA;
177
178 snprintf(logMsg, sizeof(logMsg),
179 "[%s] Motor initialized successfully (step=%.2f deg, microstep=%d, reduction=%.2f)",
180 config.label, config.stepAngle, config.microstep, config.reductionFactor);
181 LOG_DEBUG(logMsg);
182
183 return kStatus_Success;
184}

190static status_t wait_for_cmd_handle(CHD_CmdHandle_t cmdHandle, TickType_t deadline)
191{
192 if (cmdHandle == NULL)
193 {
194 return kStatus_Fail;
195 }
196
197 status_t status = SYW_cmd_wait_result(cmdHandle, deadline, NULL);
198 CHD_remove_cmd_handle_ref(cmdHandle);
199 return status;
200}
