120 -- SystemCoreClockUpdate()
121 ---------------------------------------------------------------------------- */
122
123void SystemCoreClockUpdate (void) {
124
125 uint32_t MCGOUTClock; /* Variable to store output clock frequency of the MCG module */
126 uint16_t Divider;
127 uint8_t tmpC7 = 0;
128
129 if ((MCG->C1 & MCG_C1_CLKS_MASK) == 0x00U) {
130 /* Output of FLL or PLL is selected */
131 if ((MCG->C6 & MCG_C6_PLLS_MASK) == 0x00U) {
132 /* FLL is selected */
133 if ((MCG->C1 & MCG_C1_IREFS_MASK) == 0x00U) {
134 /* External reference clock is selected */
135 switch (MCG->C7 & MCG_C7_OSCSEL_MASK) {
136 case 0x00U:
137 MCGOUTClock = CPU_XTAL_CLK_HZ; /* System oscillator drives MCG clock */
138 break;
139 case 0x01U:
140 MCGOUTClock = CPU_XTAL32k_CLK_HZ; /* RTC 32 kHz oscillator drives MCG clock */
141 break;
142 case 0x02U:
143 default:
144 MCGOUTClock = CPU_INT_IRC_CLK_HZ; /* IRC 48MHz oscillator drives MCG clock */
145 break;
146 }
147 tmpC7 = MCG->C7;
148 if (((MCG->C2 & MCG_C2_RANGE_MASK) != 0x00U) && ((tmpC7 & MCG_C7_OSCSEL_MASK) != 0x01U)) {
149 switch (MCG->C1 & MCG_C1_FRDIV_MASK) {
150 case 0x38U:
151 Divider = 1536U;
152 break;
153 case 0x30U:
154 Divider = 1280U;
155 break;
156 default:
157 Divider = (uint16_t)(32LU << ((MCG->C1 & MCG_C1_FRDIV_MASK) >> MCG_C1_FRDIV_SHIFT));
158 break;
159 }
160 } else {
161 Divider = (uint16_t)(1LU << ((MCG->C1 & MCG_C1_FRDIV_MASK) >> MCG_C1_FRDIV_SHIFT));
162 }
163 MCGOUTClock = (MCGOUTClock / Divider); /* Calculate the divided FLL reference clock */
164 } else {
165 MCGOUTClock = CPU_INT_SLOW_CLK_HZ; /* The slow internal reference clock is selected */
166 }
167 /* Select correct multiplier to calculate the MCG output clock */
168 switch (MCG->C4 & (MCG_C4_DMX32_MASK | MCG_C4_DRST_DRS_MASK)) {
169 case 0x00U:
170 MCGOUTClock *= 640U;
171 break;
172 case 0x20U:
173 MCGOUTClock *= 1280U;
174 break;
175 case 0x40U:
176 MCGOUTClock *= 1920U;
177 break;
178 case 0x60U:
179 MCGOUTClock *= 2560U;
180 break;
181 case 0x80U:
182 MCGOUTClock *= 732U;
183 break;
184 case 0xA0U:
185 MCGOUTClock *= 1464U;
186 break;
187 case 0xC0U:
188 MCGOUTClock *= 2197U;
189 break;
190 case 0xE0U:
191 MCGOUTClock *= 2929U;
192 break;
193 default:
194 MCGOUTClock *= 640U;
195 break;
196 }
197 } else { /* (!((MCG->C6 & MCG_C6_PLLS_MASK) == 0x00U)) */
198 /* PLL is selected */
199 Divider = (((uint16_t)MCG->C5 & MCG_C5_PRDIV0_MASK) + 0x01U);
200 MCGOUTClock = (uint32_t)(CPU_XTAL_CLK_HZ / Divider); /* Calculate the PLL reference clock */
201 Divider = (((uint16_t)MCG->C6 & MCG_C6_VDIV0_MASK) + 24U);
202 MCGOUTClock *= Divider; /* Calculate the MCG output clock */
203 } /* (!((MCG->C6 & MCG_C6_PLLS_MASK) == 0x00U)) */
204 } else if ((MCG->C1 & MCG_C1_CLKS_MASK) == 0x40U) {
205 /* Internal reference clock is selected */
206 if ((MCG->C2 & MCG_C2_IRCS_MASK) == 0x00U) {
207 MCGOUTClock = CPU_INT_SLOW_CLK_HZ; /* Slow internal reference clock selected */
208 } else { /* Fast internal reference clock selected */
209 Divider = (uint16_t)(0x01LU << ((MCG->SC & MCG_SC_FCRDIV_MASK) >> MCG_SC_FCRDIV_SHIFT));
210 MCGOUTClock = (uint32_t) (CPU_INT_FAST_CLK_HZ / Divider);
211 }
212 } else if ((MCG->C1 & MCG_C1_CLKS_MASK) == 0x80U) {
213 /* External reference clock is selected */
214 switch (MCG->C7 & MCG_C7_OSCSEL_MASK) {
215 case 0x00U:
216 MCGOUTClock = CPU_XTAL_CLK_HZ; /* System oscillator drives MCG clock */
217 break;
218 case 0x01U:
219 MCGOUTClock = CPU_XTAL32k_CLK_HZ; /* RTC 32 kHz oscillator drives MCG clock */
220 break;
221 case 0x02U:
222 default:
223 MCGOUTClock = CPU_INT_IRC_CLK_HZ; /* IRC 48MHz oscillator drives MCG clock */
224 break;
225 }
226 } else {
227 /* Reserved value */
228 return;
229 }
230 SystemCoreClock = (MCGOUTClock / (0x01U + ((SIM->CLKDIV1 & SIM_CLKDIV1_OUTDIV1_MASK) >> SIM_CLKDIV1_OUTDIV1_SHIFT)));
231}
