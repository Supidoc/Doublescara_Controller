45status_t MTRi_send_cmd_async(MTR_CmdQueueItem_t* queueItem, TickType_t deadline,
46 CHD_CmdHandle_t* cmdHandle)
47{
48 if (queueItem == NULL)
49 {
50 return kStatus_Fail;
51 }
52 CHD_CmdHandle_t internaleCmdHandle = NULL;
53
54 status_t allocStatus =
55 CHD_get_cmd_handle(&internaleCmdHandle, mtrCmdHandles, MTR_MAX_CMD_HANDLE_COUNT);
56 if (allocStatus != kStatus_Success)
57 {
58 return kStatus_Fail;
59 }
60 if (cmdHandle != NULL)
61 {
62 CHD_add_cmd_handle_ref(internaleCmdHandle);
63 }
64
65 queueItem->cmdHandle = internaleCmdHandle;
66
67 status_t sendStatus = CDP_send_cmd(mtrCmdQueue, queueItem, deadline, internaleCmdHandle);
68 if (sendStatus != kStatus_Success)
69 {
70 CHD_remove_cmd_handle_ref(internaleCmdHandle);
71 if (cmdHandle != NULL)
72 {
73 *cmdHandle = NULL;
74 }
75 return kStatus_Fail;
76 }
77 if (cmdHandle != NULL)
78 {
79 *cmdHandle = internaleCmdHandle;
80 }
81 return kStatus_Success;
82}

84void MTRi_process(void)
85{
86 MTR_CmdQueueItem_t queueItem;
87 BaseType_t queueStatus;
88 uint8_t activeTaskCount = 0;
89
90 status_t taskStatus = kStatus_Fail;
91 for (uint8_t i = 0; i < MTR_MAX_PARALLEL_TASKS; i++)
92 {
93 if (parallelTasks[i].used)
94 {
95 activeTaskCount++;
96 taskStatus =
97 SYW_cmd_check_all(parallelTasks[i].cmdHandles, parallelTasks[i].count, NULL);
98 if (taskStatus == kStatus_Busy)
99 {
100 continue;
101 }
102 else if (taskStatus == kStatus_Success)
103 {
104 CDP_notify_task_success(parallelTasks[i].returnHandle);
105 }
106 else if (taskStatus == kStatus_Timeout)
107 {
108 CDP_notify_task_timeout(parallelTasks[i].returnHandle);
109 }
110 else
111 {
112 CDP_notify_task_failure(parallelTasks[i].returnHandle);
113 }
114 CHD_remove_cmd_handle_ref(parallelTasks[i].returnHandle);
115 parallelTasks[i].used = 0;
116 }
117 }
118
119 TickType_t delay = activeTaskCount ? pdMS_TO_TICKS(2) : portMAX_DELAY;
120
121 queueStatus = xQueueReceive(mtrCmdQueue, &queueItem, delay);
122 if (queueStatus != pdPASS)
123 {
124 return;
125 }
126
127 MTR_ParallelTaskItem* taskItem = NULL;
128 for (uint8_t i = 0; i < MTR_MAX_PARALLEL_TASKS; i++)
129 {
130 if (parallelTasks[i].used == 0)
131 {
132 parallelTasks[i].used = 1;
133 taskItem = &parallelTasks[i];
134 break;
135 }
136 }
137
138 if (taskItem != NULL)
139 {
140 taskItem->count = 0;
141 taskItem->returnHandle = queueItem.cmdHandle;
142 MTRi_process_cmd(queueItem, taskItem);
143 }
144}

146status_t MTRi_process_cmd(MTR_CmdQueueItem_t queueItem, MTR_ParallelTaskItem* taskItem)
147{
148 if (taskItem == NULL)
149 {
150 return kStatus_Fail;
151 }
152
153 status_t cmdStatus = kStatus_Fail;
154
155 switch (queueItem.type)
156 {
157 case MTR_CMD_MOVE_ANGLE:
158 cmdStatus = MTRi_move_angle(queueItem.handle, queueItem.data.moveAngle.angle,
159 queueItem.deadline, taskItem);
160 break;
161
162 case MTR_CMD_MOVE_ABSOLUTE_ANGLE:
163 cmdStatus = MTRi_move_absolute_angle(queueItem.handle, queueItem.data.moveAngle.angle,
164 queueItem.deadline, taskItem);
165 break;
166
167 case MTR_CMD_MOVE_REVOLUTIONS:
168 cmdStatus =
169 MTRi_move_revolutions(queueItem.handle, queueItem.data.moveRevolutions.revolutions,
170 queueItem.deadline, taskItem);
171 break;
172
173 case MTR_CMD_SET_VELOCITY:
174 cmdStatus =
175 MTRi_set_velocity(queueItem.handle, queueItem.data.setVelocity.velocity_deg_per_sec,
176 queueItem.deadline, taskItem);
177 break;
178
179 case MTR_CMD_SET_ACCELERATION:
180 cmdStatus = MTRi_set_acceleration(
181 queueItem.handle, queueItem.data.setAcceleration.acceleration_deg_per_sec2,
182 queueItem.deadline, taskItem);
183 break;
184
185 case MTR_CMD_STOP:
186 cmdStatus = MTRi_stop_motor(queueItem.handle, queueItem.data.stop.decelerate,
187 queueItem.deadline, taskItem);
188 break;
189
190 case MTR_CMD_GET_CURRENT_ANGLE:
191 cmdStatus =
192 MTRi_get_current_angle(queueItem.handle, queueItem.data.getCurrentAngle.angle);
193 break;
194
195 case MTR_CMD_GET_MOVEMENT_STATE:
196 cmdStatus =
197 MTRi_get_movement_state(queueItem.handle, queueItem.data.getMovementState.state);
198 break;
199 case MTR_CMD_SYNCHRONIZED_MOVE:
200 cmdStatus = MTRi_synchronized_move(
201 queueItem.data.synchronizedMove.handles, queueItem.data.synchronizedMove.angles,
202 queueItem.data.synchronizedMove.count, queueItem.deadline);
203 break;
204
205 case MTR_CMD_SET_RUN_CURRENT:
206 cmdStatus = MTRi_set_run_current(
207 queueItem.handle, queueItem.data.setRunCurrent.current_a, queueItem.deadline);
208 break;
209
210 case MTR_CMD_SET_HOLD_CURRENT:
211 cmdStatus = MTRi_set_hold_current(
212 queueItem.handle, queueItem.data.setHoldCurrent.current_a, queueItem.deadline);
213 break;
214 case MTR_CMD_INIT_MOTOR:
215 cmdStatus = MTR_init_motor(queueItem.data.initMotor.config, queueItem.deadline,
216 motorHandles, MTR_MAX_MOTORS);
217 break;
218 case MTR_CMD_ENABLE_FREEWHEELING:
219 cmdStatus = MTRi_enable_freewheeling(queueItem.handle, queueItem.deadline);
220 break;
221 case MTR_CMD_DISABLE_FREEWHEELING:
222 cmdStatus = MTRi_disable_freewheeling(queueItem.handle, queueItem.deadline);
223 break;
224 default:
225 {
226 static char errMsg[100];
227 snprintf(errMsg, sizeof(errMsg), "Unknown motor command type: %d", queueItem.type);
228 LOG_ERROR(errMsg);
229 break;
230 }
231 }
232 if (taskItem->count == 1)
233 {
234 if (SYW_cmd_wait_result(taskItem->cmdHandles[0], queueItem.deadline, NULL) !=
235 kStatus_Success)
236 {
237 cmdStatus = kStatus_Fail;
238 }
239 CHD_remove_cmd_handle_ref(taskItem->cmdHandles[0]);
240 taskItem->count = 0;
241 }
242 if (cmdStatus == kStatus_Success)
243 {
244
245 CDP_notify_task_success(queueItem.cmdHandle);
246 taskItem->used = 0;
247 CHD_remove_cmd_handle_ref(queueItem.cmdHandle);
248 }
249 if (cmdStatus == kStatus_Timeout)
250 {
251 CDP_notify_task_timeout(queueItem.cmdHandle);
252 taskItem->used = 0;
253 CHD_remove_cmd_handle_ref(queueItem.cmdHandle);
254 }
255 else if (cmdStatus == kStatus_Fail)
256 {
257 CDP_notify_task_failure(queueItem.cmdHandle);
258 taskItem->used = 0;
259 CHD_remove_cmd_handle_ref(queueItem.cmdHandle);
260 }
261
262 return cmdStatus;
263}
