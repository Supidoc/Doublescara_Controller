36status_t DSPI_RTOS_Init(dspi_rtos_handle_t *handle,
37 SPI_Type *base,
38 const dspi_master_config_t *masterConfig,
39 uint32_t srcClock_Hz)
40{
41 if (handle == NULL)
42 {
43 return kStatus_InvalidArgument;
44 }
45
46 if (base == NULL)
47 {
48 return kStatus_InvalidArgument;
49 }
50
51 (void)memset(handle, 0, sizeof(dspi_rtos_handle_t));
52
53#if (configSUPPORT_STATIC_ALLOCATION == 1)
54 handle->mutex = xSemaphoreCreateMutexStatic(&handle->mutexBuffer);
55#else
56 handle->mutex = xSemaphoreCreateMutex();
57#endif
58 if (handle->mutex == NULL)
59 {
60 return kStatus_Fail;
61 }
62#if (configSUPPORT_STATIC_ALLOCATION == 1)
63 handle->event = xSemaphoreCreateBinaryStatic(&handle->semaphoreBuffer);
64#else
65 handle->event = xSemaphoreCreateBinary();
66#endif
67 if (handle->event == NULL)
68 {
69 vSemaphoreDelete(handle->mutex);
70 return kStatus_Fail;
71 }
72
73 handle->base = base;
74
75 DSPI_MasterInit(handle->base, masterConfig, srcClock_Hz);
76 DSPI_MasterTransferCreateHandle(handle->base, &handle->drv_handle, DSPI_RTOS_Callback, (void *)handle);
77
78 return kStatus_Success;
79}

106status_t DSPI_RTOS_Transfer(dspi_rtos_handle_t *handle, dspi_transfer_t *transfer)
107{
108 status_t status;
109
110 /* Lock resource mutex */
111 if (xSemaphoreTake(handle->mutex, portMAX_DELAY) != pdTRUE)
112 {
113 return kStatus_DSPI_Busy;
114 }
115
116 status = DSPI_MasterTransferNonBlocking(handle->base, &handle->drv_handle, transfer);
117 if (status != kStatus_Success)
118 {
119 (void)xSemaphoreGive(handle->mutex);
120 return status;
121 }
122
123 /* Wait for transfer to finish */
124 if (xSemaphoreTake(handle->event, portMAX_DELAY) != pdTRUE)
125 {
126 return kStatus_DSPI_Error;
127 }
128
129 /* Unlock resource mutex */
130 (void)xSemaphoreGive(handle->mutex);
131
132 /* Return status captured by callback function */
133 return handle->async_status;
134}
