559static inline ftm_clock_prescale_t FTM_CalculateCounterClkDiv(FTM_Type *base,
560 uint32_t counterPeriod_Hz,
561 uint32_t srcClock_Hz)
562{
563 uint8_t i;
564 assert((srcClock_Hz / 2U) > counterPeriod_Hz);
565 assert((srcClock_Hz / 128U / 0xFFFFU) <= counterPeriod_Hz);
566 for (i = 0U; i < (uint32_t)kFTM_Prescale_Divide_128; i++)
567 {
568 if ((srcClock_Hz / (1UL << i) / 0xFFFFU) < counterPeriod_Hz)
569 {
570 break;
571 }
572 }
573 return (ftm_clock_prescale_t)i;
574}
