49status_t CHD_get_cmd_handle(CHD_CmdHandle_t* cmdHandle, CHD_CmdHandleImpl_t* cmdHandles,
50 size_t maxHandles)
51{
52 if (cmdHandle == NULL)
53 {
54 return kStatus_Fail;
55 }
56
57 taskENTER_CRITICAL();
58 for (size_t i = 0; i < maxHandles; i++)
59 {
60 if (!cmdHandles[i].used)
61 {
62 cmdHandles[i].used = 1;
63 cmdHandles[i].ref_count = 0;
64 xEventGroupClearBits(cmdHandles[i].eventGroup,
65 (CHD_CMD_BIT_SUCCESS | CHD_CMD_BIT_FAILURE | CHD_CMD_BIT_TIMEOUT));
66 if (cmdHandle != NULL)
67 {
68 *cmdHandle = &cmdHandles[i];
69 }
70 taskEXIT_CRITICAL();
71 return kStatus_Success;
72 }
73 }
74 taskEXIT_CRITICAL();
75 *cmdHandle = NULL; // No available command handle
76 return kStatus_Fail;
77}
