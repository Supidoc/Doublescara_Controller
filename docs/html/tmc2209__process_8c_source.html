49void TMCi_process(void)
50{
51 TMC_CommandQueueItem_t queueItem = {0};
52 BaseType_t status;
53 status = xQueueReceive(tmcCmdQueue, &queueItem, pdMS_TO_TICKS(10));
54 if (status != pdPASS)
55 {
56 return;
57 }
58 status_t cmdStatus = kStatus_Success;
59 switch (queueItem.commandType)
60 {
61 case TMC_CMD_DEFAULT_INIT:
62 cmdStatus = TMCi_init_handle(queueItem.data.initHandle.config, queueItem.deadline);
63 break;
64 case TMC_CMD_SET_MICROSTEPPING:
65 cmdStatus = TMCi_set_microstepping(queueItem.handle,
66 queueItem.data.setMicrostepping.microstepping,
67 queueItem.deadline);
68 break;
69 case TMC_CMD_SET_IHOLD_DIVIDER:
70 cmdStatus = TMCi_set_ihold_divider(
71 queueItem.handle, queueItem.data.setIholdDivider.ihold, queueItem.deadline);
72 if (cmdStatus == kStatus_Success)
73 {
74 queueItem.handle->iholdDivider = queueItem.data.setIholdDivider.ihold;
75 }
76 break;
77 case TMC_CMD_SET_IRUN_DIVIDER:
78 cmdStatus = TMCi_set_irun_divider(queueItem.handle, queueItem.data.setIrunDivider.irun,
79 queueItem.deadline);
80 if (cmdStatus == kStatus_Success)
81 {
82 queueItem.handle->irunDivider = queueItem.data.setIrunDivider.irun;
83 }
84 break;
85 case TMC_CMD_ENABLE_FREEWHEELING:
86 cmdStatus = TMCi_set_freewheeling(queueItem.handle, 1, queueItem.deadline);
87 if (cmdStatus == kStatus_Success)
88 {
89 queueItem.handle->freewheeling = 1;
90 }
91 break;
92 case TMC_CMD_DISABLE_FREEWHEELING:
93 cmdStatus = TMCi_set_freewheeling(queueItem.handle, 0, queueItem.deadline);
94 if (cmdStatus == kStatus_Success)
95 {
96 queueItem.handle->freewheeling = 0;
97 }
98 break;
99 default:
100 cmdStatus = kStatus_Fail;
101 break;
102 }
103
104 if (queueItem.cmdHandle != NULL)
105 {
106 if (queueItem.deadline <= xTaskGetTickCount())
107 {
108 CDP_notify_task_timeout(queueItem.cmdHandle);
109 }
110 else if (cmdStatus == kStatus_Success)
111 {
112 CDP_notify_task_success(queueItem.cmdHandle);
113 }
114 else
115 {
116 CDP_notify_task_failure(queueItem.cmdHandle);
117 }
118
119 CHD_remove_cmd_handle_ref(queueItem.cmdHandle);
120 }
121}

123status_t TMCi_send_cmd_async(TMC_CommandQueueItem_t* queueItem, TickType_t deadline,
124 CHD_CmdHandle_t* cmdHandle)
125{
126 if (queueItem == NULL)
127 {
128 return kStatus_Fail;
129 }
130 CHD_CmdHandle_t internaleCmdHandle = NULL;
131
132 status_t allocStatus =
133 CHD_get_cmd_handle(&internaleCmdHandle, tmcCmdHandles, TMC_MAX_CMD_HANDLE_COUNT);
134 if (allocStatus != kStatus_Success)
135 {
136 return kStatus_Fail;
137 }
138 if (cmdHandle != NULL)
139 {
140 CHD_add_cmd_handle_ref(internaleCmdHandle);
141 }
142
143 queueItem->cmdHandle = internaleCmdHandle;
144
145 status_t sendStatus = CDP_send_cmd(tmcCmdQueue, queueItem, deadline, internaleCmdHandle);
146 if (sendStatus != kStatus_Success)
147 {
148 CHD_remove_cmd_handle_ref(internaleCmdHandle);
149 if (cmdHandle != NULL)
150 {
151 *cmdHandle = NULL;
152 }
153 return kStatus_Fail;
154 }
155 if (cmdHandle != NULL)
156 {
157 *cmdHandle = internaleCmdHandle;
158 }
159 return kStatus_Success;
160}
