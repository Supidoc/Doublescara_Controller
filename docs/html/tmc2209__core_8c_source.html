54 if (tmcCmdQueue == NULL)
55 {
56 return kStatus_Fail;

232status_t TMC_current_to_divider(float desired_current, TMC_RoundingMode_t rounding,
233 uint8_t* divider)
234{
235
236 float reference_current = 1.767766953f; // Reference current for TMC2209 at 100% (1.77A)
237 // calculated from datasheet Rev. 1.09 p. 53
238
239 /* Validate inputs */
240 if (divider == NULL)
241 {
242 return kStatus_Fail;
243 }
244
245 if (desired_current <= 0.0f)
246 {
247 return kStatus_Fail;
248 }
249
250 if (desired_current > reference_current)
251 {
252 return kStatus_Fail;
253 }
254
255 float raw_divider = (desired_current * 32 / reference_current) - 1.0f;
256
257 /* Apply rounding mode */
258 uint8_t rounded_divider;
259 switch (rounding)
260 {
261 case TMC_ROUND_FLOOR:
262 rounded_divider = (uint8_t)floorf(raw_divider);
263 break;
264 case TMC_ROUND_CEIL:
265 rounded_divider = (uint8_t)ceilf(raw_divider);
266 break;
267 case TMC_ROUND_NEAREST:
268 rounded_divider = (uint8_t)roundf(raw_divider);
269 break;
270 default:
271 return kStatus_Fail;
272 }
273
274 /* Validate divider is within valid range [0, 31] for 3-bit TMC2209 register */
275 if (rounded_divider > 31)
276 {
277 return kStatus_Fail;
278 }
279 if (rounded_divider < 0)
280 {
281 rounded_divider = 0;
282 }
283
284 *divider = rounded_divider;
285 return kStatus_Success;
286}

288status_t TMC_get_default_config(TMC_Config_t* config)
289{
290 if (config == NULL)
291 {
292 return kStatus_Fail;
293 }
294
295 config->uartRTOSHandle = &UART1_rtos_handle;
296 config->uartHandle = &UART1_uart_handle;
297 config->serialAdress = 0;
298 config->label = "TMC2209";
299 config->microstepping = TMC_MS_16; // Default to 1/16 microstepping
300 config->iholdDivider = 16; // Default IHOLD divider (50% of max current)
301 config->irunDivider = 16; // Default IRUN divider (50% of max current)
302
303 return kStatus_Success;

306status_t TMC_get_handle_by_label(const char* label, TMC_Handle_t* handle)
307{
308 if (label == NULL || handle == NULL)
309 {
310 return kStatus_Fail;
311 }
312
313 for (size_t i = 0; i < TMC_MAX_HANDLE_COUNT; i++)
314 {
315 if (tmcHandleArray[i].used && strcmp(tmcHandleArray[i].handle.label, label) == 0)
316 {
317 *handle = &tmcHandleArray[i].handle;
318 return kStatus_Success;
319 }
320 }
321
322 return kStatus_Fail; // No matching handle found
323}
